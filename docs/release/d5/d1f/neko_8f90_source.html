129 subroutine neko_init(C)
130 type(case_t), target, intent(inout), optional :: C
131 character(len=NEKO_FNAME_LEN) :: case_file
132 character(len=LOG_SIZE) :: log_buf
133 character(len=10) :: suffix
134 character(10) :: time
135 character(8) :: date
136 integer :: argc, nthrds, rw, sw
137
138 call date_and_time(time = time, date = date)
139
140 call comm_init
141 call neko_mpi_types_init
142 call jobctrl_init
143 call device_init
144
145 call neko_log%init()
146 call neko_field_registry%init()
147
148 call neko_log%header(neko_version, neko_build_info)
149
150 if (present(c)) then
151
152 argc = command_argument_count()
153
154 if ((argc .lt. 1) .or. (argc .gt. 1)) then
155 if (pe_rank .eq. 0) write(*,*) 'Usage: ./neko < case file >'
156 stop
157 end if
158
159 call get_command_argument(1, case_file)
160
161 call filename_suffix(case_file, suffix)
162
163 if (trim(suffix) .ne. 'case') then
164 call neko_error('Invalid case file')
165 end if
166
167 ! Check the device count against the number of MPI ranks
168 if (neko_bcknd_device .eq. 1) then
169 if (device_count() .ne. 1) then
170 call neko_error('Only one device is supported per MPI rank')
171 end if
172 end if
173
174 !
175 ! Job information
176 !
177 call neko_log%section("Job Information")
178 write(log_buf, '(A,A,A,A,1x,A,1x,A,A,A,A,A)') 'Start time: ',&
179 time(1:2), ':', time(3:4), &
180 '/', date(1:4), '-', date(5:6), '-', date(7:8)
181 call neko_log%message(log_buf, neko_log_quiet)
182 write(log_buf, '(a)') 'Running on: '
183 sw = 10
184 if (pe_size .lt. 1e1) then
185 write(log_buf(13:), '(i1,a)') pe_size, ' MPI '
186 if (pe_size .eq. 1) then
187 write(log_buf(19:), '(a)') 'rank'
188 sw = 9
189 else
190 write(log_buf(19:), '(a)') 'ranks'
191 end if
192 rw = 1
193 else if (pe_size .lt. 1e2) then
194 write(log_buf(13:), '(i2,a)') pe_size, ' MPI ranks'
195 rw = 2
196 else if (pe_size .lt. 1e3) then
197 write(log_buf(13:), '(i3,a)') pe_size, ' MPI ranks'
198 rw = 3
199 else if (pe_size .lt. 1e4) then
200 write(log_buf(13:), '(i4,a)') pe_size, ' MPI ranks'
201 rw = 4
202 else if (pe_size .lt. 1e5) then
203 write(log_buf(13:), '(i5,a)') pe_size, ' MPI ranks'
204 rw = 5
205 else
206 write(log_buf(13:), '(i6,a)') pe_size, ' MPI ranks'
207 rw = 6
208 end if
209
210 nthrds = 1
211 !$omp parallel
212 !$omp master
213 !$ nthrds = omp_get_num_threads()
214 !$omp end master
215 !$omp end parallel
216
217 if (nthrds .gt. 1) then
218 if (nthrds .lt. 1e1) then
219 write(log_buf(13 + rw + sw:), '(a,i1,a)') ', using ', &
220 nthrds, ' thrds each'
221 else if (nthrds .lt. 1e2) then
222 write(log_buf(13 + rw + sw:), '(a,i2,a)') ', using ', &
223 nthrds, ' thrds each'
224 else if (nthrds .lt. 1e3) then
225 write(log_buf(13 + rw + sw:), '(a,i3,a)') ', using ', &
226 nthrds, ' thrds each'
227 else if (nthrds .lt. 1e4) then
228 write(log_buf(13 + rw + sw:), '(a,i4,a)') ', using ', &
229 nthrds, ' thrds each'
230 end if
231 end if
232 call neko_log%message(log_buf, neko_log_quiet)
233
234 write(log_buf, '(a)') 'CPU type : '
235 call system_cpu_name(log_buf(13:))
236 call neko_log%message(log_buf, neko_log_quiet)
237
238 write(log_buf, '(a)') 'Bcknd type: '
239 if (neko_bcknd_sx .eq. 1) then
240 write(log_buf(13:), '(a)') 'SX-Aurora'
241 else if (neko_bcknd_xsmm .eq. 1) then
242 write(log_buf(13:), '(a)') 'CPU (libxsmm)'
243 else if (neko_bcknd_cuda .eq. 1) then
244 write(log_buf(13:), '(a)') 'Accelerator (CUDA)'
245 else if (neko_bcknd_hip .eq. 1) then
246 write(log_buf(13:), '(a)') 'Accelerator (HIP)'
247 else if (neko_bcknd_opencl .eq. 1) then
248 write(log_buf(13:), '(a)') 'Accelerator (OpenCL)'
249 else
250 write(log_buf(13:), '(a)') 'CPU'
251 end if
252 call neko_log%message(log_buf, neko_log_quiet)
253
254 if (neko_bcknd_hip .eq. 1 .or. neko_bcknd_cuda .eq. 1 .or. &
255 neko_bcknd_opencl .eq. 1) then
256 write(log_buf, '(a)') 'Dev. name : '
257 call device_name(log_buf(13:))
258 call neko_log%message(log_buf, neko_log_quiet)
259 end if
260
261 write(log_buf, '(a)') 'Real type : '
262 select case (rp)
263 case (real32)
264 write(log_buf(13:), '(a)') 'single precision'
265 case (real64)
266 write(log_buf(13:), '(a)') 'double precision'
267 case (real128)
268 write(log_buf(13:), '(a)') 'quad precision'
269 end select
270 call neko_log%message(log_buf, neko_log_quiet)
271
272 call neko_log%end()
273
274 !
275 ! Create case
276 !
277 call case_init(c, case_file)
278
279 !
280 ! Setup runtime statistics
281 !
282 call neko_rt_stats%init(c%params)
283
284
285 !
286 ! Create simulation components
287 !
288 call neko_simcomps%init(c)
289
290 end if
291
292 end subroutine neko_init
293
294 subroutine neko_finalize(C)
295 type(case_t), intent(inout), optional :: C
296
297 call neko_rt_stats%report()
298 call neko_rt_stats%free()
299
300 call neko_scratch_registry%free()
301
302 if (present(c)) then
303 call case_free(c)
304 end if
305
306 call neko_field_registry%free()
307 call device_finalize
308 call neko_mpi_types_free
309 call comm_free
310 end subroutine neko_finalize
311
312end module neko
