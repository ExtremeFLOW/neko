56 const int y = idx/(NX-2) + 1;

58
59 idx1 = l1 + x*NX + y*NX*NX + el;
60 idx2 = l2 + x*NX + y*NX*NX + el;
61 a1[idx1] = f1*a1[idx1] + f2*a2[idx2];

64 idx2 = (NX-1-l2) + x*NX + y*NX*NX + el;
