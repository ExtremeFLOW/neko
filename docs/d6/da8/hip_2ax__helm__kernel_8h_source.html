39 template< typename T, const int LX, const int CHUNKS >
40 __global__ void ax_helm_kernel_1d(T * __restrict__ w,
41  const T * __restrict__ u,
42  const T * __restrict__ dx,
43  const T * __restrict__ dy,
44  const T * __restrict__ dz,
45  const T * __restrict__ dxt,
46  const T * __restrict__ dyt,
47  const T * __restrict__ dzt,
48  const T * __restrict__ h1,
49  const T * __restrict__ g11,
50  const T * __restrict__ g22,
51  const T * __restrict__ g33,
52  const T * __restrict__ g12,
53  const T * __restrict__ g13,
54  const T * __restrict__ g23) {
55 
56  __shared__ T shdx[LX*LX];
57  __shared__ T shdy[LX*LX];
58  __shared__ T shdzt[LX*LX];
59 
60  __shared__ T shdxt[LX*LX];
61  __shared__ T shdyt[LX*LX];
62  __shared__ T shdz[LX*LX];
63 
64  __shared__ T shu[LX*LX*LX];
65  __shared__ T shur[LX*LX*LX];
66  __shared__ T shus[LX*LX*LX];
67  __shared__ T shut[LX*LX*LX];
68 
69  const int e = blockIdx.x;
70  const int iii = threadIdx.x;
71  const int nchunks = (LX * LX * LX - 1)/CHUNKS + 1;
72 
73  if (iii<LX*LX) {
74  shdx[iii] = dx[iii];
75  shdy[iii] = dy[iii];
76  shdz[iii] = dz[iii];
77  }
78 
79  {
80  int i = iii;
81  while (i < LX * LX * LX){
82  shu[i] = u[i+e*LX*LX*LX];
83  i = i + CHUNKS;
84  }
85  }
86 
87  __syncthreads();
88 
89  if (iii<LX*LX){
90  shdxt[iii] = dxt[iii];
91  shdyt[iii] = dyt[iii];
92  shdzt[iii] = dzt[iii];
93  }
94 
95  for (int n=0; n<nchunks; n++){
96  const int ijk = iii+n*CHUNKS;
97  const int jk = ijk/LX;
98  const int i = ijk-jk*LX;
99  const int k = jk/LX;
100  const int j = jk-k*LX;
101  if (i<LX && j<LX && k<LX && ijk < LX*LX*LX){
102  const T G00 = g11[ijk+e*LX*LX*LX];
103  const T G11 = g22[ijk+e*LX*LX*LX];
104  const T G22 = g33[ijk+e*LX*LX*LX];
105  const T G01 = g12[ijk+e*LX*LX*LX];
106  const T G02 = g13[ijk+e*LX*LX*LX];
107  const T G12 = g23[ijk+e*LX*LX*LX];
108  const T H1 = h1[ijk+e*LX*LX*LX];
109  T rtmp = 0.0;
110  T stmp = 0.0;
111  T ttmp = 0.0;
112 #pragma unroll
113  for (int l = 0; l<LX; l++){
114  rtmp = rtmp + shdx[i+l*LX] * shu[l+j*LX+k*LX*LX];
115  stmp = stmp + shdy[j+l*LX] * shu[i+l*LX+k*LX*LX];
116  ttmp = ttmp + shdz[k+l*LX] * shu[i+j*LX+l*LX*LX];
117  }
118  shur[ijk] = H1 * (G00 * rtmp + G01 * stmp + G02 * ttmp);
119  shus[ijk] = H1 * (G01 * rtmp + G11 * stmp + G12 * ttmp);
120  shut[ijk] = H1 * (G02 * rtmp + G12 * stmp + G22 * ttmp);
121  }
122  }
123 
124  __syncthreads();
125 
126  for (int n=0; n<nchunks; n++){
127  const int ijk = iii+n*CHUNKS;
128  const int jk = ijk/LX;
129  const int k = jk/LX;
130  const int j = jk-k*LX;
131  const int i = ijk-jk*LX;
132  if (i<LX && j<LX && k<LX && ijk <LX*LX*LX){
133  T wijke = 0.0;
134 #pragma unroll
135  for (int l = 0; l<LX; l++){
136  wijke = wijke
137  + shdxt[i+l*LX] * shur[l+j*LX+k*LX*LX]
138  + shdyt[j+l*LX] * shus[i+l*LX+k*LX*LX]
139  + shdzt[k+l*LX] * shut[i+j*LX+l*LX*LX];
140  }
141  w[ijk+e*LX*LX*LX] = wijke;
142  }
143  }
144 }

234  rw[l] += rut * shdz[k+l*LX];
235  wijke += shus[i+l*LX] * shdy[l + j*LX];
236  }

238  }
