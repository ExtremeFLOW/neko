277  subroutine probes_map(this, coef)
278  class(probes_t), intent(inout) :: this
279  type(coef_t), intent(in) :: coef
280 
281  real(kind=rp) :: rst_gslib_raw(3*this%n_probes)
282 
283  real(kind=rp) :: tol_dist = 5d-6
284  integer :: i, local_index
285 
286  ! Global mapping, finding the points
287  call this%global_interp%find_points(this%xyz, this%n_probes)
288 
289  ! Number of points owned by a rank
290  this%n_local_probes = 0
291 
292  do i=1,this%n_probes
293 
294  !
295  ! Reformat the rst array into point_t
296  !
297  this%rst(i)%x(1) = this%global_interp%rst(1,i)
298  this%rst(i)%x(2) = this%global_interp%rst(2,i)
299  this%rst(i)%x(3) = this%global_interp%rst(3,i)
300 
301  !
302  ! Count the number of local points
303  !
304  if (pe_rank .eq. this%global_interp%proc_owner(i)) this%n_local_probes = this%n_local_probes+1
305 
306  end do
307 
308  ! -------------------------- END GSLIB ------------------------------------
309 
310  !
311  ! Allocate local stuff
312  !
313  if (this%n_local_probes .ne. 0) then
314 
315  !
316  ! Allocate and associate the element ownership locally
317  !
318  allocate(this%local_el_owner(this%n_local_probes))
319  allocate(this%local_to_global(this%n_local_probes))
320  allocate(this%local_rst(this%n_local_probes))
321 
322  local_index = 1
323 
324  do i = 1, this%n_probes
325  if (pe_rank .eq. this%global_interp%proc_owner(i)) then
326  this%local_el_owner(local_index) = this%global_interp%el_owner(i)
327  this%local_to_global(local_index) = i
328  this%local_rst(local_index) = this%rst(i)%x
329  local_index = local_index + 1
330  end if
331  end do
332 
333  !
334  ! This could be another way to generate the local array
335  ! this%local_el_owner = pack(this%el_owner, this%proc_owner .eq. pe_rank)
336  !
337 
338  if (neko_bcknd_device .eq. 1) then
339  call device_map(this%local_el_owner, this%local_el_owner_d, &
340  this%n_local_probes)
341  call device_memcpy(this%local_el_owner, this%local_el_owner_d, &
342  this%n_local_probes, host_to_device, sync = .true.)
343  end if
344 
345  end if

347  end subroutine probes_map

354  subroutine probes_interpolate(this, t, tstep, write_output)
355  class(probes_t), intent(inout) :: this
356  real(kind=rp), intent(in) :: t
357  integer, intent(in) :: tstep
358  logical, intent(inout) :: write_output
359  real(kind=rp), allocatable :: tmp(:,:)
360  integer :: i, ierr, lx
361  integer :: size_outfields, size_weights
362 
364  if (this%controller%check(t, tstep, .false.)) then
365 
366  size_outfields = this%n_probes * this%n_fields
367 
368  ! Fill the out_field array with 0s as we are reducing later
369  ! with MPI_SUM
370  call rzero(this%out_fields, size_outfields)
371 
372  ! Do not allocate/compute if current has no local probes
373  if (this%n_local_probes .ne. 0) then
374 
375  lx = this%interpolator%Xh%lx
376  size_weights = this%n_local_probes * lx
377 
378  !
379  ! Only update weights if necessary
380  !
381  if (this%update_weights) then
382 
383  if (.not. allocated(this%weights_r)) allocate(this%weights_r(lx,this%n_local_probes))
384  if (.not. allocated(this%weights_s)) allocate(this%weights_s(lx,this%n_local_probes))
385  if (.not. allocated(this%weights_t)) allocate(this%weights_t(lx,this%n_local_probes))
386 
387  !
388  ! Build weights
389  !
390  call this%interpolator%compute_weights(real(this%rst(:)%x(1), rp), &
391  real(this%rst(:)%x(2), rp), real(this%rst(:)%x(3), rp), &
392  this%weights_r, this%weights_s, this%weights_t)
393 
394  !
395  ! Associate device pointers
396  !
397  if (neko_bcknd_device .eq. 1) then
398 
399  call device_map(this%weights_r, this%weights_r_d, size_weights)
400  call device_map(this%weights_s, this%weights_s_d, size_weights)
401  call device_map(this%weights_t, this%weights_t_d, size_weights)
402  call device_memcpy(this%weights_r, this%weights_r_d, size_weights, host_to_device, sync = .true.)
403  call device_memcpy(this%weights_s, this%weights_s_d, size_weights, host_to_device, sync = .true.)
404  call device_memcpy(this%weights_t, this%weights_t_d, size_weights, host_to_device, sync = .true.)
405 
406  end if
407 
408  this%update_weights = .false.
409 
410  end if
411 
412  allocate(tmp(this%n_local_probes, this%n_fields))
413 
414  !
415  ! Interpolate
416  !
417  tmp = this%interpolator%interpolate(this%local_rst, &
418  this%local_el_owner, this%sampled_fields, &
419  this%weights_r, this%weights_s, this%weights_t)
420 
421  !
422  ! Reconstruct the global array using global to local mapping
423  !
424  do i = 1, this%n_local_probes
425  this%out_fields(this%local_to_global(i),:) = tmp(i,:)
426  end do
427 
428  deallocate(tmp)
429 
430  end if
431 
432  ! Artificial way to gather all values to rank 0
433  if (pe_rank .ne. 0) then
434  call mpi_reduce(this%out_fields(1,1), this%out_fields(1,1), size_outfields, &
435  mpi_real_precision, mpi_sum, 0, neko_comm, ierr)
436  else
437  call mpi_reduce(mpi_in_place, this%out_fields(1,1), size_outfields, &
438  mpi_real_precision, mpi_sum, 0, neko_comm, ierr)
439  end if
440 
441  !! Turn on flag to write output
442  write_output = .true.
443 
444  !! Register the execution of the activity
445  call this%controller%register_execution()
446  end if
447 
448  end subroutine probes_interpolate
