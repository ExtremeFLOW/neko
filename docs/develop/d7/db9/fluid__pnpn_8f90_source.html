139  subroutine fluid_pnpn_init(this, msh, lx, params, user, material_properties)
140  class(fluid_pnpn_t), target, intent(inout) :: this
141  type(mesh_t), target, intent(inout) :: msh
142  integer, intent(inout) :: lx
143  type(json_file), target, intent(inout) :: params
144  type(user_t), intent(in) :: user
145  type(material_properties_t), target, intent(inout) :: material_properties
146  character(len=15), parameter :: scheme = 'Modular (Pn/Pn)'
147 
148  call this%free()
149 
150  ! Initialize base class
151  call this%scheme_init(msh, lx, params, .true., .true., scheme, user, &
152  material_properties)
153 
154  if (this%variable_material_properties .eqv. .true.) then
155  call neko_error("Variable material properties not supported.")
156  end if
157 
158  ! Setup backend dependent Ax routines
159  call ax_helm_factory(this%ax, full_formulation = .false.)
160 
161  ! Setup backend dependent prs residual routines
162  call pnpn_prs_res_factory(this%prs_res)
163 
164  ! Setup backend dependent vel residual routines
165  call pnpn_vel_res_factory(this%vel_res)
166 
167  ! Setup backend dependent summation of AB/BDF
168  call rhs_maker_sumab_fctry(this%sumab)
169 
170  ! Setup backend dependent summation of extrapolation scheme
171  call rhs_maker_ext_fctry(this%makeabf)
172 
173  ! Setup backend depenent contributions to F from lagged BD terms
174  call rhs_maker_bdf_fctry(this%makebdf)
175 
176  ! Initialize variables specific to this plan
177  associate(xh_lx => this%Xh%lx, xh_ly => this%Xh%ly, xh_lz => this%Xh%lz, &
178  dm_xh => this%dm_Xh, nelv => this%msh%nelv)
179 
180  call this%p_res%init(dm_xh, "p_res")
181  call this%u_res%init(dm_xh, "u_res")
182  call this%v_res%init(dm_xh, "v_res")
183  call this%w_res%init(dm_xh, "w_res")
184  call this%abx1%init(dm_xh, "abx1")
185  call this%aby1%init(dm_xh, "aby1")
186  call this%abz1%init(dm_xh, "abz1")
187  call this%abx2%init(dm_xh, "abx2")
188  call this%aby2%init(dm_xh, "aby2")
189  call this%abz2%init(dm_xh, "abz2")
190  this%abx1 = 0.0_rp
191  this%aby1 = 0.0_rp
192  this%abz1 = 0.0_rp
193  this%abx2 = 0.0_rp
194  this%aby2 = 0.0_rp
195  this%abz2 = 0.0_rp
196 
197  call this%du%init(dm_xh, 'du')
198  call this%dv%init(dm_xh, 'dv')
199  call this%dw%init(dm_xh, 'dw')
200  call this%dp%init(dm_xh, 'dp')
201 
202  end associate
203 
204  ! Initialize velocity surface terms in pressure rhs
205  call this%bc_prs_surface%init_base(this%c_Xh)
206  call this%bc_prs_surface%mark_zone(msh%inlet)
207  call this%bc_prs_surface%mark_zones_from_list(msh%labeled_zones,&
208  'v', this%bc_labels)
209  !This impacts the rhs of the pressure, need to check what is correct to add here
210  call this%bc_prs_surface%mark_zones_from_list(msh%labeled_zones,&
211  'd_vel_u', this%bc_labels)
212  call this%bc_prs_surface%mark_zones_from_list(msh%labeled_zones,&
213  'd_vel_v', this%bc_labels)
214  call this%bc_prs_surface%mark_zones_from_list(msh%labeled_zones,&
215  'd_vel_w', this%bc_labels)
216  call this%bc_prs_surface%finalize()
217  ! Initialize symmetry surface terms in pressure rhs
218  call this%bc_sym_surface%init_base(this%c_Xh)
219  call this%bc_sym_surface%mark_zone(msh%sympln)
220  call this%bc_sym_surface%mark_zones_from_list(msh%labeled_zones,&
221  'sym', this%bc_labels)
222  ! Same here, should du, dv, dw be marked here?
223  call this%bc_sym_surface%finalize()
224  ! Initialize dirichlet bcs for velocity residual
225  call this%bc_vel_res_non_normal%init_base(this%c_Xh)
226  call this%bc_vel_res_non_normal%mark_zone(msh%outlet_normal)
227  call this%bc_vel_res_non_normal%mark_zones_from_list(msh%labeled_zones,&
228  'on', this%bc_labels)
229  call this%bc_vel_res_non_normal%mark_zones_from_list(msh%labeled_zones,&
230  'on+dong', &
231  this%bc_labels)
232  call this%bc_vel_res_non_normal%finalize()
233  call this%bc_vel_res_non_normal%init(this%c_Xh)
234 
235  call this%bc_field_dirichlet_p%init_base(this%c_Xh)
236  call this%bc_field_dirichlet_p%mark_zones_from_list(msh%labeled_zones, &
237  'on+dong', this%bc_labels)
238  call this%bc_field_dirichlet_p%mark_zones_from_list(msh%labeled_zones, &
239  'o+dong', this%bc_labels)
240  call this%bc_field_dirichlet_p%mark_zones_from_list(msh%labeled_zones, &
241  'd_pres', this%bc_labels)
242  call this%bc_field_dirichlet_p%finalize()
243  call this%bc_field_dirichlet_p%set_g(0.0_rp)
244  call bc_list_init(this%bclst_dp)
245  call bc_list_add(this%bclst_dp, this%bc_field_dirichlet_p)
246  !Add 0 prs bcs
247  call bc_list_add(this%bclst_dp, this%bc_prs)
248 
249  call this%bc_field_dirichlet_u%init_base(this%c_Xh)
250  call this%bc_field_dirichlet_u%mark_zones_from_list( &
251  msh%labeled_zones, 'd_vel_u', this%bc_labels)
252  call this%bc_field_dirichlet_u%finalize()
253  call this%bc_field_dirichlet_u%set_g(0.0_rp)
254 
255  call this%bc_field_dirichlet_v%init_base(this%c_Xh)
256  call this%bc_field_dirichlet_v%mark_zones_from_list(msh%labeled_zones, &
257  'd_vel_v', &
258  this%bc_labels)
259  call this%bc_field_dirichlet_v%finalize()
260  call this%bc_field_dirichlet_v%set_g(0.0_rp)
261 
262  call this%bc_field_dirichlet_w%init_base(this%c_Xh)
263  call this%bc_field_dirichlet_w%mark_zones_from_list(msh%labeled_zones, &
264  'd_vel_w', &
265  this%bc_labels)
266  call this%bc_field_dirichlet_w%finalize()
267  call this%bc_field_dirichlet_w%set_g(0.0_rp)
268 
269  call this%bc_vel_res%init_base(this%c_Xh)
270  call this%bc_vel_res%mark_zone(msh%inlet)
271  call this%bc_vel_res%mark_zone(msh%wall)
272  call this%bc_vel_res%mark_zones_from_list(msh%labeled_zones, &
273  'v', this%bc_labels)
274  call this%bc_vel_res%mark_zones_from_list(msh%labeled_zones, &
275  'w', this%bc_labels)
276  call this%bc_vel_res%finalize()
277  call this%bc_vel_res%set_g(0.0_rp)
278  call bc_list_init(this%bclst_vel_res)
279  call bc_list_add(this%bclst_vel_res, this%bc_vel_res)
280  call bc_list_add(this%bclst_vel_res, this%bc_vel_res_non_normal)
281  call bc_list_add(this%bclst_vel_res, this%bc_sym)
282 
283  !Initialize bcs for u, v, w velocity components
284  call bc_list_init(this%bclst_du)
285  call bc_list_add(this%bclst_du, this%bc_sym%bc_x)
286  call bc_list_add(this%bclst_du, this%bc_vel_res_non_normal%bc_x)
287  call bc_list_add(this%bclst_du, this%bc_vel_res)
288  call bc_list_add(this%bclst_du, this%bc_field_dirichlet_u)
289 
290  call bc_list_init(this%bclst_dv)
291  call bc_list_add(this%bclst_dv, this%bc_sym%bc_y)
292  call bc_list_add(this%bclst_dv, this%bc_vel_res_non_normal%bc_y)
293  call bc_list_add(this%bclst_dv, this%bc_vel_res)
294  call bc_list_add(this%bclst_dv, this%bc_field_dirichlet_v)
295 
296  call bc_list_init(this%bclst_dw)
297  call bc_list_add(this%bclst_dw, this%bc_sym%bc_z)
298  call bc_list_add(this%bclst_dw, this%bc_vel_res_non_normal%bc_z)
299  call bc_list_add(this%bclst_dw, this%bc_vel_res)
300  call bc_list_add(this%bclst_dw, this%bc_field_dirichlet_w)
301 
302  !Intialize projection space thingy
303  call this%proj_prs%init(this%dm_Xh%size(), this%pr_projection_dim, &
304  this%pr_projection_activ_step)
305 
306  call this%proj_u%init(this%dm_Xh%size(), this%vel_projection_dim, &
307  this%vel_projection_activ_step)
308  call this%proj_v%init(this%dm_Xh%size(), this%vel_projection_dim, &
309  this%vel_projection_activ_step)
310  call this%proj_w%init(this%dm_Xh%size(), this%vel_projection_dim, &
311  this%vel_projection_activ_step)
312 
313 
314  ! Add lagged term to checkpoint
315  call this%chkp%add_lag(this%ulag, this%vlag, this%wlag)
316 
317  call advection_factory(this%adv, params, this%c_Xh)
318 
319  if (params%valid_path('case.fluid.flow_rate_force')) then
320  call this%vol_flow%init(this%dm_Xh, params)
321  end if
322 
323  end subroutine fluid_pnpn_init

325  subroutine fluid_pnpn_restart(this, dtlag, tlag)
326  class(fluid_pnpn_t), target, intent(inout) :: this
327  real(kind=rp) :: dtlag(10), tlag(10)
328  type(field_t) :: u_temp, v_temp, w_temp
329  integer :: i, n
330 
331  n = this%u%dof%size()
332  ! Make sure that continuity is maintained (important for interpolation)
333  ! Do not do this for lagged rhs
334  ! (derivatives are not necessairly coninous across elements)
335  call col2(this%u%x, this%c_Xh%mult, this%u%dof%size())
336  call col2(this%v%x, this%c_Xh%mult, this%u%dof%size())
337  call col2(this%w%x, this%c_Xh%mult, this%u%dof%size())
338  call col2(this%p%x, this%c_Xh%mult, this%u%dof%size())
339  do i = 1, this%ulag%size()
340  call col2(this%ulag%lf(i)%x, this%c_Xh%mult, this%u%dof%size())
341  call col2(this%vlag%lf(i)%x, this%c_Xh%mult, this%u%dof%size())

516  subroutine fluid_pnpn_step(this, t, tstep, dt, ext_bdf, dt_controller)
517  class(fluid_pnpn_t), target, intent(inout) :: this
518  real(kind=rp), intent(inout) :: t
519  integer, intent(inout) :: tstep
520  real(kind=rp), intent(in) :: dt
521  type(time_scheme_controller_t), intent(inout) :: ext_bdf
522  type(time_step_controller_t), intent(in) :: dt_controller
523  ! number of degrees of freedom
524  integer :: n
525  ! Solver results monitors (pressure + 3 velocity)
526  type(ksp_monitor_t) :: ksp_results(4)
527  ! Extrapolated velocity for the pressure residual
528  type(field_t), pointer :: u_e, v_e, w_e
529  ! Indices for tracking temporary fields
530  integer :: temp_indices(3)
531 
532  if (this%freeze) return
533 
534  n = this%dm_Xh%size()
535 
536  call profiler_start_region('Fluid', 1)
537  associate(u => this%u, v => this%v, w => this%w, p => this%p, &
538  du => this%du, dv => this%dv, dw => this%dw, dp => this%dp, &
539  u_res => this%u_res, v_res => this%v_res, w_res => this%w_res, &
540  p_res => this%p_res, ax => this%Ax, xh => this%Xh, &
541  c_xh => this%c_Xh, dm_xh => this%dm_Xh, gs_xh => this%gs_Xh, &
542  ulag => this%ulag, vlag => this%vlag, wlag => this%wlag, &
543  msh => this%msh, prs_res => this%prs_res, &
544  source_term => this%source_term, &
545  vel_res => this%vel_res, sumab => this%sumab, &
546  makeabf => this%makeabf, makebdf => this%makebdf, &
547  vel_projection_dim => this%vel_projection_dim, &
548  pr_projection_dim => this%pr_projection_dim, &
549  rho => this%rho, mu => this%mu, &
550  f_x => this%f_x, f_y => this%f_y, f_z => this%f_z, &
551  if_variable_dt => dt_controller%if_variable_dt, &
552  dt_last_change => dt_controller%dt_last_change)
553 
554  ! Get temporary arrays
555  call this%scratch%request_field(u_e, temp_indices(1))
556  call this%scratch%request_field(v_e, temp_indices(2))
557  call this%scratch%request_field(w_e, temp_indices(3))
558  call sumab%compute_fluid(u_e, v_e, w_e, u, v, w, &
559  ulag, vlag, wlag, ext_bdf%advection_coeffs, ext_bdf%nadv)
560 
561  ! Compute the source terms
562  call this%source_term%compute(t, tstep)
563 
564  ! Pre-multiply the source terms with the mass matrix.
565  if (neko_bcknd_device .eq. 1) then
566  call device_opcolv(f_x%x_d, f_y%x_d, f_z%x_d, c_xh%B_d, msh%gdim, n)
567  else
568  call opcolv(f_x%x, f_y%x, f_z%x, c_xh%B, msh%gdim, n)
569  end if
570 
571  ! Add the advection operators to the right-hand-side.
572  call this%adv%compute(u, v, w, &
573  f_x, f_y, f_z, &
574  xh, this%c_Xh, dm_xh%size())
575 
576  ! At this point the RHS contains the sum of the advection operator and
577  ! additional source terms, evaluated using the velocity field from the
578  ! previous time-step. Now, this value is used in the explicit time
579  ! scheme to advance both terms in time.
580  call makeabf%compute_fluid(this%abx1, this%aby1, this%abz1,&
581  this%abx2, this%aby2, this%abz2, &
582  f_x%x, f_y%x, f_z%x, &
583  rho, ext_bdf%advection_coeffs, n)
584 
585  ! Add the RHS contributions coming from the BDF scheme.
586  call makebdf%compute_fluid(ulag, vlag, wlag, f_x%x, f_y%x, f_z%x, &
587  u, v, w, c_xh%B, rho, dt, &
588  ext_bdf%diffusion_coeffs, ext_bdf%ndiff, n)
589 
590  call ulag%update()
591  call vlag%update()
592  call wlag%update()
593 
597  call this%user_field_bc_vel%update(this%user_field_bc_vel%field_list, &
598  this%user_field_bc_vel%bc_list, this%c_Xh, t, tstep, "fluid")
599 
600  call this%bc_apply_vel(t, tstep)
601  call this%bc_apply_prs(t, tstep)
602 
603  ! Compute pressure.
604  call profiler_start_region('Pressure residual', 18)
605  call prs_res%compute(p, p_res,&
606  u, v, w, &
607  u_e, v_e, w_e, &
608  f_x, f_y, f_z, &
609  c_xh, gs_xh, &
610  this%bc_prs_surface, this%bc_sym_surface,&
611  ax, ext_bdf%diffusion_coeffs(1), dt, &
612  mu, rho)
613 
614  call gs_xh%op(p_res, gs_op_add)
615  call bc_list_apply_scalar(this%bclst_dp, p_res%x, p%dof%size(), t, tstep)
616  call profiler_end_region
617 
618  call this%proj_prs%pre_solving(p_res%x, tstep, c_xh, n, dt_controller, &
619  'Pressure')
620 
621  call this%pc_prs%update()
622  call profiler_start_region('Pressure solve', 3)
623  ksp_results(1) = &
624  this%ksp_prs%solve(ax, dp, p_res%x, n, c_xh, this%bclst_dp, gs_xh)
625 
626  call profiler_end_region
627 
628  call this%proj_prs%post_solving(dp%x, ax, c_xh, &
629  this%bclst_dp, gs_xh, n, tstep, dt_controller)
630 
631  call field_add2(p, dp, n)
632 
633  ! Compute velocity.
634  call profiler_start_region('Velocity residual', 19)
635  call vel_res%compute(ax, u, v, w, &
636  u_res, v_res, w_res, &
637  p, &
638  f_x, f_y, f_z, &
639  c_xh, msh, xh, &
640  mu, rho, ext_bdf%diffusion_coeffs(1), &
641  dt, dm_xh%size())
642 
643  call gs_xh%op(u_res, gs_op_add)
644  call gs_xh%op(v_res, gs_op_add)
645  call gs_xh%op(w_res, gs_op_add)
646 
647  call bc_list_apply_vector(this%bclst_vel_res,&
648  u_res%x, v_res%x, w_res%x, dm_xh%size(),&
649  t, tstep)
650 
651  !We should implement a bc that takes three field_bcs and implements vector_apply
652  if (neko_bcknd_device .eq. 1) then
653  call this%bc_field_dirichlet_u%apply_scalar_dev(u_res%x_d, t, tstep)
654  call this%bc_field_dirichlet_v%apply_scalar_dev(v_res%x_d, t, tstep)
655  call this%bc_field_dirichlet_w%apply_scalar_dev(w_res%x_d, t, tstep)
656  else
657  call this%bc_field_dirichlet_u%apply_scalar(u_res%x, n, t, tstep)
658  call this%bc_field_dirichlet_v%apply_scalar(v_res%x, n, t, tstep)
659  call this%bc_field_dirichlet_w%apply_scalar(w_res%x, n, t, tstep)
660  end if
661 
662  call profiler_end_region
663 
664  call this%proj_u%pre_solving(u_res%x, tstep, c_xh, n, dt_controller)
665  call this%proj_v%pre_solving(v_res%x, tstep, c_xh, n, dt_controller)
666  call this%proj_w%pre_solving(w_res%x, tstep, c_xh, n, dt_controller)
667 
668  call this%pc_vel%update()
669 
670  call profiler_start_region("Velocity solve", 4)
671  ksp_results(2:4) = this%ksp_vel%solve_coupled(ax, du, dv, dw, &
672  u_res%x, v_res%x, w_res%x, n, c_xh, &
673  this%bclst_du, this%bclst_dv, this%bclst_dw, gs_xh)
674  call profiler_end_region
675 
676  call this%proj_u%post_solving(du%x, ax, c_xh, &
677  this%bclst_du, gs_xh, n, tstep, dt_controller)
678  call this%proj_v%post_solving(dv%x, ax, c_xh, &
679  this%bclst_dv, gs_xh, n, tstep, dt_controller)
680  call this%proj_w%post_solving(dw%x, ax, c_xh, &
681  this%bclst_dw, gs_xh, n, tstep, dt_controller)
682 
683  if (neko_bcknd_device .eq. 1) then
684  call device_opadd2cm(u%x_d, v%x_d, w%x_d, &
685  du%x_d, dv%x_d, dw%x_d, 1.0_rp, n, msh%gdim)
686  else
687  call opadd2cm(u%x, v%x, w%x, du%x, dv%x, dw%x, 1.0_rp, n, msh%gdim)
688  end if
689 
690  if (this%forced_flow_rate) then
691  call this%vol_flow%adjust( u, v, w, p, u_res, v_res, w_res, p_res, &
692  c_xh, gs_xh, ext_bdf, rho, mu,&
693  dt, this%bclst_dp, this%bclst_du, this%bclst_dv, &
694  this%bclst_dw, this%bclst_vel_res, ax, this%ksp_prs, &
695  this%ksp_vel, this%pc_prs, this%pc_vel, this%ksp_prs%max_iter, &
696  this%ksp_vel%max_iter)
697  end if
698 
699  call fluid_step_info(tstep, t, dt, ksp_results)
700 
701  call this%scratch%relinquish_field(temp_indices)
702 
703  end associate
704  call profiler_end_region
705  end subroutine fluid_pnpn_step
