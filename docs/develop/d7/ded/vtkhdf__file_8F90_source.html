102 subroutine vtkhdf_file_write(this, data, t)
103 class(vtkhdf_file_t), intent(inout) :: this
104 class(*), target, intent(in) :: data
105 real(kind=rp), intent(in), optional :: t
106 type(mesh_t), pointer :: msh
107 type(dofmap_t), pointer :: dof
108 type(field_ptr_t), allocatable :: fp(:)
109 integer :: ierr, mpi_info, mpi_comm, i, n_fields
110 integer(hid_t) :: plist_id, file_id, attr_id, vtkhdf_grp
111 integer(hid_t) :: filespace, memspace
112 integer(hsize_t), dimension(1) :: vdims
113 integer :: lx, ly, lz
114 integer :: local_points, local_cells, local_conn
115 integer :: total_points, total_cells, total_conn
116 integer :: point_offset
117 integer :: max_local_points
118 integer, allocatable :: part_points(:), part_cells(:), part_conns(:)
119 character(len=1024) :: fname
120 character(len=16) :: type_str
121 logical :: link_exists, file_exists
122 integer(kind=1) :: VTK_cell_type
123 integer :: counter
124
125 ! Determine mesh and field data
126 select type(data)
127 type is (field_t)
128 msh => data%msh
129 dof => data%dof
130 n_fields = 1
131 allocate(fp(1))
132 fp(1)%ptr => data
133 type is (field_list_t)
134 msh => data%msh(1)
135 dof => data%dof(1)
136 n_fields = data%size()
137 allocate(fp(n_fields))
138 do i = 1, n_fields
139 fp(i)%ptr => data%items(i)%ptr
140 end do
141 class default
142 call neko_error('Invalid data type for vtkhdf_file_write')
143 end select
144
145 ! Check conditions to ensure the input data is supported.
146 if (.not. associated(msh)) then
147 call neko_error('Mesh must be associated for vtkhdf_file_write')
148 end if
149 if (dof%Xh%lx < 2 .or. dof%Xh%ly < 2) then
150 call neko_error('VTKHDF linear output requires lx, ly >= 2')
151 end if
152 if (msh%gdim .eq. 3 .and. dof%Xh%lz < 2) then
153 call neko_error('VTKHDF linear output requires lz >= 2 in 3D')
154 end if
155
156 ! Ensure precision is set and are valid.
157 if (this%precision .gt. rp) then
158 this%precision = rp
159 call neko_warning('Requested precision is higher than working precision')
160 else if (this%precision .eq. 0) then
161 this%precision = rp
162 end if
163
164 if (msh%gdim .eq. 3) then
165 vtk_cell_type = 12 ! VTK_HEXAHEDRON
166 else
167 vtk_cell_type = 9 ! VTK_QUAD
168 end if
169
170 call this%increment_counter()
171 fname = trim(this%get_base_fname())
172 counter = this%get_counter()
173
174 mpi_info = mpi_info_null%mpi_val
175 mpi_comm = neko_comm%mpi_val
176
177 call h5open_f(ierr)
178 call h5pcreate_f(h5p_file_access_f, plist_id, ierr)
179 call h5pset_fapl_mpio_f(plist_id, mpi_comm, mpi_info, ierr)
180
181 inquire(file = fname, exist = file_exists)
182 if (file_exists) then
183 call h5fopen_f(fname, h5f_acc_rdwr_f, file_id, ierr, &
184 access_prp = plist_id)
185 else
186 call h5fcreate_f(fname, h5f_acc_trunc_f, &
187 file_id, ierr, access_prp = plist_id)
188 end if
189
190 ! Create/open VTKHDF root group with vtkhdf_version and type attributes
191 call h5lexists_f(file_id, "VTKHDF", link_exists, ierr)
192 if (link_exists) then
193 call h5gopen_f(file_id, "VTKHDF", vtkhdf_grp, ierr)
194 else
195 call h5gcreate_f(file_id, "VTKHDF", vtkhdf_grp, ierr)
196
197 ! Write Version attribute
198 vdims = 2
199 call h5screate_simple_f(1, vdims, filespace, ierr)
200 call h5acreate_f(vtkhdf_grp, "Version", h5t_native_integer, filespace, &
201 attr_id, ierr)
202 call h5awrite_f(attr_id, h5t_native_integer, vtkhdf_version, vdims, ierr)
203 call h5aclose_f(attr_id, ierr)
204 call h5sclose_f(filespace, ierr)
205
206 ! Write Type attribute "UnstructuredGrid" as a fixed-length string
207 type_str = "UnstructuredGrid"
208 vdims = 1
209 call h5screate_f(h5s_scalar_f, filespace, ierr)
210
211 call h5tcopy_f(h5t_fortran_s1, memspace, ierr)
212 call h5tset_size_f(memspace, int(len_trim(type_str), kind=hsize_t), ierr)
213 call h5tset_strpad_f(memspace, h5t_str_nullterm_f, ierr)
214
215 call h5acreate_f(vtkhdf_grp, "Type", memspace, filespace, attr_id, ierr)
216 call h5awrite_f(attr_id, memspace, [type_str], vdims, ierr)
217 call h5aclose_f(attr_id, ierr)
218
219 call h5tclose_f(memspace, ierr)
220 call h5sclose_f(filespace, ierr)
221 end if
222
223 if (present(t)) then
224 call vtkhdf_write_steps(vtkhdf_grp, counter, t)
225 end if
226
227 if (associated(msh)) then
228 call vtkhdf_write_mesh(vtkhdf_grp, dof, msh, vtk_cell_type, &
229 this%amr_enabled, counter, t)
230 end if
231
232 ! Write field data in PointData group
233 if (n_fields > 0) then
234 call vtkhdf_write_pointdata(vtkhdf_grp, fp, this%precision, counter, t)
235 end if
236
237 call h5gclose_f(vtkhdf_grp, ierr)
238 call h5pclose_f(plist_id, ierr)
239 call h5fclose_f(file_id, ierr)
240 call h5close_f(ierr)
241
242 if (allocated(fp)) deallocate(fp)
243
244 end subroutine vtkhdf_file_write

254 subroutine vtkhdf_build_connectivity(conn, vtk_type, msh, dof)
255 integer, intent(out) :: conn(:)
256 integer(kind=1), intent(in) :: vtk_type
257 type(mesh_t) :: msh
258 type(dofmap_t) :: dof
259 integer :: lx, ly, lz, nelv
260 integer :: ie, ii, jj, kk, base, idx, npts_per_cell
261
262 nelv = msh%nelv
263 lx = dof%Xh%lx
264 ly = dof%Xh%ly
265 lz = dof%Xh%lz
266 npts_per_cell = lx * ly * lz
267 idx = 0
268
269 do ie = 1, nelv
270 base = (ie - 1) * npts_per_cell
271
272 select case (vtk_type)
273 case (12) ! VTK_HEXAHEDRON
274 do ii = 1, lx - 1
275 do jj = 1, ly - 1
276 do kk = 1, lz - 1
277 conn(idx + 1) = base + &
278 (kk - 1) * lx * ly + (jj - 1) * lx + ii - 1
279 conn(idx + 2) = base + &
280 (kk - 1) * lx * ly + (jj - 1) * lx + (ii + 1) - 1
281 conn(idx + 3) = base + &
282 (kk - 1) * lx * ly + jj * lx + (ii + 1) - 1
283 conn(idx + 4) = base + &
284 (kk - 1) * lx * ly + jj * lx + ii - 1
285 conn(idx + 5) = base + &
286 kk * lx * ly + (jj - 1) * lx + ii - 1
287 conn(idx + 6) = base + &
288 kk * lx * ly + (jj - 1) * lx + (ii + 1) - 1
289 conn(idx + 7) = base + &
290 kk * lx * ly + jj * lx + (ii + 1) - 1
291 conn(idx + 8) = base + &
292 kk * lx * ly + jj * lx + ii - 1
293 idx = idx + 8
294 end do
295 end do
296 end do
297
298 case (9) ! VTK_QUAD
299 do jj = 1, ly - 1
300 do ii = 1, lx - 1
301 conn(idx + 1) = base + (jj - 1) * lx + ii - 1
302 conn(idx + 2) = base + (jj - 1) * lx + (ii + 1) - 1
303 conn(idx + 3) = base + jj * lx + (ii + 1) - 1
304 conn(idx + 4) = base + jj * lx + ii - 1
305 idx = idx + 4
306 end do
307 end do
308
309 case default
310 call neko_error('Unsupported VTK cell type')
311 end select
312 end do
313
314 end subroutine vtkhdf_build_connectivity

326 subroutine vtkhdf_write_mesh(vtkhdf_grp, dof, msh, VTK_cell_type, amr, &
327 counter, t)
328 type(dofmap_t), intent(in) :: dof
329 type(mesh_t), intent(in) :: msh
330 integer(hid_t), intent(in) :: vtkhdf_grp
331 integer(kind=1), intent(in) :: VTK_cell_type
332 logical, intent(in) :: amr
333 integer, intent(in) :: counter
334 real(kind=rp), intent(in), optional :: t
335
336 integer :: ierr, i, ii, jj, kk, el, local_idx
337 integer :: lx, ly, lz, npts_per_cell, nodes_per_cell, cells_per_element
338 integer :: local_points, local_cells, local_conn
339 integer :: total_points, total_cells, total_conn
340 integer :: point_offset, max_local_points
341 integer :: total_offsets, cell_offset, conn_offset, offsets_offset
342 integer :: max_local_cells, max_local_conn
343 integer(hid_t) :: xf_id, dset_id, dcpl_id, grp_id, attr_id
344 integer(hid_t) :: filespace, memspace
345 integer(hsize_t), dimension(1) :: dcount, vdims, maxdims, doffset, chunkdims
346 integer(hsize_t), dimension(2) :: dcount2, vdims2, maxdims2, doffset2
347 integer(kind=8) :: i8_value
348 logical :: link_exists
349 integer, dimension(3) :: component_sizes
350 integer, dimension(3) :: component_offsets
351 integer, dimension(3) :: component_max_sizes
352
353 lx = dof%Xh%lx
354 ly = dof%Xh%ly
355 lz = dof%Xh%lz
356 npts_per_cell = lx * ly * lz
357
358 select case(vtk_cell_type)
359 case(12)
360 cells_per_element = (lx - 1) * (ly - 1) * (lz - 1)
361 nodes_per_cell = 8
362 case(9)
363 cells_per_element = (lx - 1) * (ly - 1)
364 nodes_per_cell = 4
365 case default
366 call neko_error('Unsupported VTK cell type')
367 end select
368
369 ! --- Build the number of cells and the connectivity
370 local_points = dof%size()
371 local_cells = msh%nelv * cells_per_element
372 local_conn = local_cells * nodes_per_cell
373
374 total_points = dof%global_size()
375 total_cells = msh%glb_nelv * cells_per_element
376 total_conn = total_cells * nodes_per_cell
377
378 component_sizes = [local_points, local_cells, local_conn]
379 component_offsets = 0
380 component_max_sizes = 0
381
382 call mpi_exscan(component_sizes, component_offsets, 3, mpi_integer, &
383 mpi_sum, neko_comm, ierr)
384 call mpi_allreduce(component_sizes, component_max_sizes, 3, mpi_integer, &
385 mpi_max, neko_comm, ierr)
386
387 point_offset = component_offsets(1)
388 cell_offset = component_offsets(2)
389 conn_offset = component_offsets(3)
390 max_local_points = component_max_sizes(1)
391 max_local_cells = component_max_sizes(2)
392 max_local_conn = component_max_sizes(3)
393
394 offsets_offset = cell_offset + pe_rank
395 total_offsets = total_cells + pe_size
396
397 ! Create collective transfer property list
398 call h5pcreate_f(h5p_dataset_xfer_f, xf_id, ierr)
399 call h5pset_dxpl_mpio_f(xf_id, h5fd_mpio_collective_f, ierr)
400
401 ! --- Shared sizes for 1D datasets ---
402 dcount = 1_hsize_t
403 vdims = int(pe_size, hsize_t)
404 chunkdims = int(pe_size, hsize_t)
405 doffset = int(pe_rank, hsize_t)
406
407 ! --- Create filespaces and memspaces for 1D datasets ---
408 call h5screate_simple_f(1, vdims, filespace, ierr)
409 call h5screate_simple_f(1, dcount, memspace, ierr)
410 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
411 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
412 doffset, dcount, ierr)
413 call h5pset_chunk_f(dcpl_id, 1, chunkdims, ierr)
414
415 ! --- NumberOfPoints dataset (per-partition) ---
416 call h5lexists_f(vtkhdf_grp, "NumberOfPoints", link_exists, ierr)
417 if (.not. link_exists) then
418 call h5dcreate_f(vtkhdf_grp, "NumberOfPoints", h5t_native_integer, &
419 filespace, dset_id, ierr, dcpl_id = dcpl_id)
420 call h5dwrite_f(dset_id, h5t_native_integer, local_points, &
421 dcount, ierr, file_space_id = filespace, &
422 mem_space_id = memspace, xfer_prp = xf_id)
423 call h5dclose_f(dset_id, ierr)
424 end if
425
426 ! --- NumberOfCells dataset (per-partition) ---
427 call h5lexists_f(vtkhdf_grp, "NumberOfCells", link_exists, ierr)
428 if (.not. link_exists) then
429 call h5dcreate_f(vtkhdf_grp, "NumberOfCells", h5t_native_integer, &
430 filespace, dset_id, ierr, dcpl_id = dcpl_id)
431 call h5dwrite_f(dset_id, h5t_native_integer, local_cells, &
432 dcount, ierr, file_space_id = filespace, &
433 mem_space_id = memspace, xfer_prp = xf_id)
434 call h5dclose_f(dset_id, ierr)
435 end if
436
437 ! --- NumberOfConnectivityIds dataset (per-partition) ---
438 call h5lexists_f(vtkhdf_grp, "NumberOfConnectivityIds", link_exists, ierr)
439 if (.not. link_exists) then
440 call h5dcreate_f(vtkhdf_grp, "NumberOfConnectivityIds", &
441 h5t_native_integer, filespace, dset_id, ierr, dcpl_id = dcpl_id)
442 call h5dwrite_f(dset_id, h5t_native_integer, local_conn, &
443 dcount, ierr, file_space_id = filespace, &
444 mem_space_id = memspace, xfer_prp = xf_id)
445 call h5dclose_f(dset_id, ierr)
446 end if
447
448 ! --- Close the mempory and file spaces ---
449 call h5sclose_f(memspace, ierr)
450 call h5sclose_f(filespace, ierr)
451 call h5pclose_f(dcpl_id, ierr)
452
453 ! --- Points dataset (global coordinates) ---
454 call h5lexists_f(vtkhdf_grp, "Points", link_exists, ierr)
455 if (.not. link_exists) then
456
457 vdims2 = [3_hsize_t, int(total_points, hsize_t)]
458 maxdims2 = [3_hsize_t, h5s_unlimited_f]
459 chunkdims(1) = int(max(1, min(max_local_points, total_points)), hsize_t)
460 dcount2 = [3_hsize_t, int(local_points, hsize_t)]
461 doffset2 = [0_hsize_t, int(point_offset, hsize_t)]
462
463 call h5screate_simple_f(2, vdims2, filespace, ierr, maxdims2)
464 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
465 call h5pset_chunk_f(dcpl_id, 2, [3_hsize_t, chunkdims(1)], ierr)
466 call h5dcreate_f(vtkhdf_grp, "Points", h5t_native_double, &
467 filespace, dset_id, ierr, dcpl_id = dcpl_id)
468 call h5screate_simple_f(2, dcount2, memspace, ierr)
469 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
470 doffset2, dcount2, ierr)
471
472 block
473 real(kind=dp), allocatable :: coords(:,:)
474
475 allocate(coords(3, local_points))
476 do concurrent(local_idx = 1:local_points)
477 block
478 integer :: idx(4)
479 idx = nonlinear_index(local_idx, lx, ly, lz)
480
481 coords(1, local_idx) = dof%x(idx(1), idx(2), idx(3), idx(4))
482 coords(2, local_idx) = dof%y(idx(1), idx(2), idx(3), idx(4))
483 coords(3, local_idx) = dof%z(idx(1), idx(2), idx(3), idx(4))
484 end block
485 end do
486 call h5dwrite_f(dset_id, h5t_native_double, coords, dcount2, ierr, &
487 file_space_id = filespace, mem_space_id = memspace, &
488 xfer_prp = xf_id)
489 deallocate(coords)
490 end block
491
492 call h5sclose_f(memspace, ierr)
493 call h5dclose_f(dset_id, ierr)
494 call h5pclose_f(dcpl_id, ierr)
495 call h5sclose_f(filespace, ierr)
496 end if
497
498 ! --- Connectivity dataset ---
499 call h5lexists_f(vtkhdf_grp, "Connectivity", link_exists, ierr)
500 if (link_exists) call h5ldelete_f(vtkhdf_grp, "Connectivity", ierr)
501
502 vdims = int(total_conn, hsize_t)
503 maxdims = h5s_unlimited_f
504 chunkdims = int(max(1, min(max_local_conn, total_conn)), hsize_t)
505 dcount = int(local_conn, hsize_t)
506 doffset = int(conn_offset, hsize_t)
507
508 call h5screate_simple_f(1, vdims, filespace, ierr, maxdims)
509 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
510 call h5pset_chunk_f(dcpl_id, 1, chunkdims, ierr)
511 call h5dcreate_f(vtkhdf_grp, "Connectivity", h5t_native_integer, &
512 filespace, dset_id, ierr, dcpl_id = dcpl_id)
513 call h5screate_simple_f(1, dcount, memspace, ierr)
514 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
515 doffset, dcount, ierr)
516
517 block
518 integer, allocatable :: connectivity(:)
519
520 allocate(connectivity(local_conn))
521 call vtkhdf_build_connectivity(connectivity, vtk_cell_type, msh, dof)
522 call h5dwrite_f(dset_id, h5t_native_integer, connectivity, dcount, &
523 ierr, file_space_id = filespace, mem_space_id = memspace, &
524 xfer_prp = xf_id)
525 deallocate(connectivity)
526 end block
527
528 call h5sclose_f(memspace, ierr)
529 call h5dclose_f(dset_id, ierr)
530 call h5pclose_f(dcpl_id, ierr)
531 call h5sclose_f(filespace, ierr)
532
533 ! --- Offsets dataset ---
534 call h5lexists_f(vtkhdf_grp, "Offsets", link_exists, ierr)
535 if (link_exists) call h5ldelete_f(vtkhdf_grp, "Offsets", ierr)
536
537 vdims = int(total_offsets, hsize_t)
538 maxdims = h5s_unlimited_f
539 chunkdims = int(max(1, min(max_local_cells + 1, total_offsets)), hsize_t)
540 dcount = int(local_cells + 1, hsize_t)
541 doffset = int(offsets_offset, hsize_t)
542
543 call h5screate_simple_f(1, vdims, filespace, ierr, maxdims)
544 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
545 call h5pset_chunk_f(dcpl_id, 1, chunkdims, ierr)
546 call h5dcreate_f(vtkhdf_grp, "Offsets", h5t_native_integer, &
547 filespace, dset_id, ierr, dcpl_id = dcpl_id)
548 call h5screate_simple_f(1, dcount, memspace, ierr)
549 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
550 doffset, dcount, ierr)
551
552 block
553 integer, allocatable :: offsets(:)
554
555 allocate(offsets(local_cells + 1))
556 do concurrent(i = 1:local_cells)
557 offsets(i) = (i - 1) * nodes_per_cell
558 end do
559 offsets(local_cells + 1) = local_conn
560 call h5dwrite_f(dset_id, h5t_native_integer, offsets, dcount, ierr, &
561 file_space_id = filespace, mem_space_id = memspace, xfer_prp = xf_id)
562 deallocate(offsets)
563 end block
564
565 call h5sclose_f(memspace, ierr)
566 call h5dclose_f(dset_id, ierr)
567 call h5pclose_f(dcpl_id, ierr)
568 call h5sclose_f(filespace, ierr)
569
570 ! --- Types dataset (VTK cell types) ---
571 call h5lexists_f(vtkhdf_grp, "Types", link_exists, ierr)
572 if (link_exists) call h5ldelete_f(vtkhdf_grp, "Types", ierr)
573
574 vdims = int(total_cells, hsize_t)
575 maxdims = h5s_unlimited_f
576 chunkdims = int(max(1, min(max_local_cells, total_cells)), hsize_t)
577 dcount = int(local_cells, hsize_t)
578 doffset = int(cell_offset, hsize_t)
579
580 call h5screate_simple_f(1, vdims, filespace, ierr, maxdims)
581 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
582 call h5pset_chunk_f(dcpl_id, 1, chunkdims, ierr)
583 call h5dcreate_f(vtkhdf_grp, "Types", h5t_std_u8le, &
584 filespace, dset_id, ierr, dcpl_id = dcpl_id)
585 call h5screate_simple_f(1, dcount, memspace, ierr)
586 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
587 doffset, dcount, ierr)
588
589 block
590 integer(kind=1), allocatable :: cell_types(:)
591 allocate(cell_types(local_cells), source=vtk_cell_type)
592 call h5dwrite_f(dset_id, h5t_std_u8le, cell_types, dcount, ierr, &
593 file_space_id = filespace, mem_space_id = memspace, xfer_prp = xf_id)
594 deallocate(cell_types)
595 end block
596
597 call h5sclose_f(memspace, ierr)
598 call h5dclose_f(dset_id, ierr)
599 call h5pclose_f(dcpl_id, ierr)
600 call h5sclose_f(filespace, ierr)
601
602 if (present(t)) then
603 ! Open Steps group
604 call h5gopen_f(vtkhdf_grp, "Steps", grp_id, ierr)
605
606 ! --- NumberOfParts ---
607 call vtkhdf_write_i8_at(grp_id, "NumberOfParts", int(pe_size, kind=8), &
608 counter)
609
610 ! --- PartOffsets ---
611 i8_value = 0
612 if (amr) then
613 i8_value = int(counter - 1, kind=8) * int(pe_size, kind=8)
614 call vtkhdf_write_i8_at(grp_id, "PartOffsets", i8_value, counter)
615
616 i8_value = int(counter - 1, kind=8) * int(total_points, kind=8)
617 call vtkhdf_write_i8_at(grp_id, "PointOffsets", i8_value, counter)
618
619 i8_value = int(counter - 1, kind=8) * int(total_cells, kind=8)
620 call vtkhdf_write_i8_at(grp_id, "CellOffsets", i8_value, counter)
621
622 i8_value = int(counter - 1, kind=8) * int(total_conn, kind=8)
623 call vtkhdf_write_i8_at(grp_id, "ConnectivityIdOffsets", i8_value, &
624 counter)
625
626 else
627 i8_value = int(0, kind=8)
628 call vtkhdf_write_i8_at(grp_id, "PartOffsets", i8_value, counter)
629 call vtkhdf_write_i8_at(grp_id, "PointOffsets", i8_value, counter)
630 call vtkhdf_write_i8_at(grp_id, "CellOffsets", i8_value, counter)
631 call vtkhdf_write_i8_at(grp_id, "ConnectivityIdOffsets", i8_value, &
632 counter)
633 end if
634
635 call h5gclose_f(grp_id, ierr)
636 end if
637
638 call h5pclose_f(xf_id, ierr)

648 subroutine vtkhdf_write_steps(vtkhdf_grp, counter, t)
649 integer(hid_t), intent(in) :: vtkhdf_grp
650 integer, intent(in) :: counter
651 real(kind=rp), intent(in) :: t
652
653 integer(hid_t) :: xf_id
654 integer :: ierr
655 integer(hid_t) :: grp_id, dset_id, dcpl_id, filespace, memspace, attr_id
656 integer(hsize_t), dimension(1) :: step_dims, step_maxdims
657 integer(hsize_t), dimension(1) :: step_count, step_offset, chunkdims, ddim
658 real(kind=dp), dimension(1) :: time_value
659 integer(kind=8) :: i8_value
660 logical :: link_exists, attr_exists
661
662 ! Create collective transfer property list
663 call h5pcreate_f(h5p_dataset_xfer_f, xf_id, ierr)
664 call h5pset_dxpl_mpio_f(xf_id, h5fd_mpio_collective_f, ierr)

670 else
671 call h5gcreate_f(vtkhdf_grp, "Steps", grp_id, ierr)
672 end if
673
674 ! --- Values dataset (time values, real type) ---
675 call h5lexists_f(grp_id, "Values", link_exists, ierr)
676 if (link_exists) then
677 call h5dopen_f(grp_id, "Values", dset_id, ierr)

680 ierr)

682
683 ! We have not written this timestep yet, expand the array
684 if (step_dims(1) .eq. counter) then
685 step_dims(1) = int(counter + 1, hsize_t)
686 call h5dset_extent_f(dset_id, step_dims, ierr)
687 else if (step_dims(1) .lt. counter) then
688 call neko_error("VTKHDF: Time steps written out of order.")
689 end if
690 else
691 step_dims(1) = 1_hsize_t
692 step_maxdims(1) = h5s_unlimited_f
693 chunkdims(1) = 1_hsize_t
694
695 call h5screate_simple_f(1, step_dims, filespace, ierr, step_maxdims)
696 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
697 call h5pset_chunk_f(dcpl_id, 1, chunkdims, ierr)
698 call h5dcreate_f(grp_id, "Values", h5t_native_double, &
699 filespace, dset_id, ierr, dcpl_id = dcpl_id)
700 call h5sclose_f(filespace, ierr)
701 call h5pclose_f(dcpl_id, ierr)
702 end if
703
704 step_count(1) = 1_hsize_t
705 step_offset(1) = int(counter, hsize_t)
706
707 call h5dget_space_f(dset_id, filespace, ierr)
708 call h5screate_simple_f(1, step_count, memspace, ierr)
709 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
710 step_offset, step_count, ierr)
711
712 time_value(1) = real(t, kind=dp)
713 call h5dwrite_f(dset_id, h5t_native_double, time_value, step_count, ierr, &
714 file_space_id = filespace, mem_space_id = memspace, xfer_prp = xf_id)
715
716 call h5sclose_f(memspace, ierr)
717 call h5sclose_f(filespace, ierr)
718 call h5dclose_f(dset_id, ierr)
719
720 ! --- NSteps attribute ---
721 ddim(1) = 1_hsize_t
722 call h5aexists_f(grp_id, "NSteps", attr_exists, ierr)
723 if (attr_exists) then
724 call h5aopen_f(grp_id, "NSteps", attr_id, ierr)
725 else
726 call h5screate_f(h5s_scalar_f, filespace, ierr)
727 call h5acreate_f(grp_id, "NSteps", h5t_native_integer, filespace, &
728 attr_id, ierr, h5p_default_f, h5p_default_f)
729 call h5sclose_f(filespace, ierr)
730 end if
731
732 call h5awrite_f(attr_id, h5t_native_integer, counter + 1, ddim, ierr)
733
734 call h5aclose_f(attr_id, ierr)
735 call h5gclose_f(grp_id, ierr)
736 call h5pclose_f(xf_id, ierr)
737
738 end subroutine vtkhdf_write_steps

816 subroutine vtkhdf_write_pointdata(vtkhdf_grp, fp, precision, counter, t)
817 integer(hid_t), intent(in) :: vtkhdf_grp
818 type(field_ptr_t), intent(in) :: fp(:)
819 integer, intent(in), optional :: precision
820 integer, intent(in) :: counter
821 real(kind=rp), intent(in), optional :: t
822
823 logical, allocatable :: field_written(:)
824 integer(kind=8) :: time_offset
825 integer :: nelv
826 integer :: local_points, point_offset
827 integer :: lx, ly, lz
828 integer :: max_local_points, total_points
829 integer(hid_t) :: precision_hdf
830 integer(hid_t) :: xf_id
831 integer :: ierr, i, j, ie, ii, jj, kk, local_idx
832 integer :: n_fields, npts_per_cell
833 integer(hid_t) :: pointdata_grp, grp_id, step_grp_id
834 integer(hid_t) :: dset_id, dcpl_id, attr_id, filespace, memspace
835 integer(hsize_t), dimension(1) :: dcount, doffset, chunkdims
836 integer(hsize_t), dimension(2) :: dcount2, doffset2
837 integer(hsize_t), dimension(1) :: pd_dims1, pd_maxdims1
838 integer(hsize_t), dimension(2) :: pd_dims2, pd_maxdims2
839 type(field_t), pointer :: fld, u, v, w
840 character(len=128) :: field_name
841 logical :: link_exists, is_vector
842
843 ! Create collective transfer property list
844 call h5pcreate_f(h5p_dataset_xfer_f, xf_id, ierr)
845 call h5pset_dxpl_mpio_f(xf_id, h5fd_mpio_collective_f, ierr)
846 precision_hdf = vtkhdf_file_determine_real(precision)
847
848 n_fields = size(fp)
849
850 ! --- Build the number of cells and the connectivity
851 local_points = fp(1)%ptr%dof%size()
852 total_points = fp(1)%ptr%dof%global_size()
853 nelv = fp(1)%ptr%msh%nelv
854 lx = fp(1)%ptr%dof%Xh%lx
855 ly = fp(1)%ptr%dof%Xh%ly
856 lz = fp(1)%ptr%dof%Xh%lz
857
858 point_offset = 0
859 max_local_points = 0
860 call mpi_exscan(local_points, point_offset, 1, mpi_integer, &
861 mpi_sum, neko_comm, ierr)
862 call mpi_allreduce(local_points, max_local_points, 1, mpi_integer, &
863 mpi_max, neko_comm, ierr)
864 npts_per_cell = lx * ly * lz
865
866 ! Sync all the fields
867 do i = 1, n_fields
868 if (associated(fp(i)%ptr)) then
869 call fp(i)%ptr%copy_from(device_to_host, sync = i .eq. n_fields)
870 end if
871 end do
872
873 allocate(field_written(n_fields), source=.false.)
874
875 ! Create or open PointData group
876 call h5lexists_f(vtkhdf_grp, "PointData", link_exists, ierr)
877 if (link_exists) then
878 call h5gopen_f(vtkhdf_grp, "PointData", pointdata_grp, ierr)
879 else
880 call h5gcreate_f(vtkhdf_grp, "PointData", pointdata_grp, ierr)
881 end if
882
883 ! Read how many steps have been written so far
884 if (present(t)) then
885 call h5gopen_f(vtkhdf_grp, "Steps", step_grp_id, ierr)
886 time_offset = int(counter, kind=8) * int(total_points, kind=8)
887 else
888 time_offset = 0_8
889 end if
890
891 do i = 1, n_fields
892 if (field_written(i)) cycle
893 fld => fp(i)%ptr
894
895 field_name = fld%name
896 if (field_name .eq. 'p') field_name = 'Pressure'
897
898 ! Determine if this is a velocity component to group as a vector
899 is_vector = (field_name .eq. 'u' .or. &
900 field_name .eq. 'v' .or. &
901 field_name .eq. 'w')
902
903 if (is_vector) then
904 u => null()
905 v => null()
906 w => null()
907 do j = 1, n_fields
908 select case (fp(j)%ptr%name)
909 case ('u')
910 u => fp(j)%ptr
911 field_written(j) = .true.
912 case ('v')
913 v => fp(j)%ptr
914 field_written(j) = .true.
915 case ('w')
916 w => fp(j)%ptr
917 field_written(j) = .true.
918 end select
919 end do
920 field_name = 'Velocity'
921 end if
922
923 ! Write PointDataOffsets under Steps for temporal output
924 if (present(t)) then
925 call h5lexists_f(step_grp_id, "PointDataOffsets", link_exists, ierr)
926 if (link_exists) then
927 call h5gopen_f(step_grp_id, "PointDataOffsets", grp_id, ierr)
928 else
929 call h5gcreate_f(step_grp_id, "PointDataOffsets", grp_id, ierr)
930 end if
931 call vtkhdf_write_i8_at(grp_id, trim(field_name), time_offset, &
932 counter)
933 call h5gclose_f(grp_id, ierr)
934 end if
935
936 if (is_vector) then
937 ! Create or extend 2D dataset (3 x total_points)
938 call h5lexists_f(pointdata_grp, trim(field_name), link_exists, ierr)
939 if (link_exists .and. present(t)) then
940 call h5dopen_f(pointdata_grp, trim(field_name), dset_id, ierr)
941 call h5dget_space_f(dset_id, filespace, ierr)
942 call h5sget_simple_extent_dims_f(filespace, pd_dims2, &
943 pd_maxdims2, ierr)
944 call h5sclose_f(filespace, ierr)
945
946 if (pd_dims2(2) .eq. counter * total_points) then
947 pd_dims2(2) = int((counter + 1) * total_points, hsize_t)
948 call h5dset_extent_f(dset_id, pd_dims2, ierr)
949 else if (pd_dims2(2) .lt. counter * total_points) then
950 call neko_error("VTKHDF: Data written out of order.")
951 end if
952
953 else if (.not. link_exists) then
954 pd_dims2 = [3_hsize_t, int(total_points, hsize_t)]
955 pd_maxdims2 = [3_hsize_t, h5s_unlimited_f]
956 chunkdims(1) = max(1_hsize_t, int(max_local_points, hsize_t))
957
958 call h5screate_simple_f(2, pd_dims2, filespace, ierr, pd_maxdims2)
959 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
960 call h5pset_chunk_f(dcpl_id, 2, [3_hsize_t, chunkdims(1)], ierr)
961 call h5dcreate_f(pointdata_grp, trim(field_name), precision_hdf, &
962 filespace, dset_id, ierr, dcpl_id = dcpl_id)
963 call h5sclose_f(filespace, ierr)
964 call h5pclose_f(dcpl_id, ierr)
965 end if
966
967 dcount2 = [3_hsize_t, int(local_points, hsize_t)]
968 doffset2 = [0_hsize_t, int(time_offset + point_offset, hsize_t)]
969
970 call h5dget_space_f(dset_id, filespace, ierr)
971 call h5screate_simple_f(2, dcount2, memspace, ierr)
972 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
973 doffset2, dcount2, ierr)
974
975 if (precision .eq. sp) then
976 call write_vector_single(dset_id, u, v, w, dcount2, ierr, &
977 filespace, memspace, xf_id)
978 else if (precision .eq. dp) then
979 call write_vector_double(dset_id, u, v, w, dcount2, ierr, &
980 filespace, memspace, xf_id)
981 end if
982
983 call h5sclose_f(filespace, ierr)
984 call h5sclose_f(memspace, ierr)
985 call h5dclose_f(dset_id, ierr)
986
987 else
988 ! Write scalar field as 1D dataset
989 call h5lexists_f(pointdata_grp, trim(field_name), link_exists, ierr)
990 if (link_exists) then
991 call h5dopen_f(pointdata_grp, trim(field_name), dset_id, ierr)
992 call h5dget_space_f(dset_id, filespace, ierr)
993 call h5sget_simple_extent_dims_f(filespace, pd_dims1, &
994 pd_maxdims1, ierr)
995 call h5sclose_f(filespace, ierr)
996
997 if (pd_dims1(1) .eq. counter * total_points) then
998 pd_dims1(1) = int((counter + 1) * total_points, hsize_t)
999 call h5dset_extent_f(dset_id, pd_dims1, ierr)
1000 else if (pd_dims1(1) .lt. counter * total_points) then
1001 call neko_error("VTKHDF: Data written out of order.")
1002 end if
1003 else
1004 pd_dims1(1) = int(total_points, hsize_t)
1005 pd_maxdims1(1) = h5s_unlimited_f
1006 chunkdims(1) = max(1_hsize_t, int(max_local_points, hsize_t))
1007
1008 call h5screate_simple_f(1, pd_dims1, filespace, ierr, pd_maxdims1)
1009 call h5pcreate_f(h5p_dataset_create_f, dcpl_id, ierr)
1010 call h5pset_chunk_f(dcpl_id, 1, chunkdims, ierr)
1011 call h5dcreate_f(pointdata_grp, trim(field_name), precision_hdf, &
1012 filespace, dset_id, ierr, dcpl_id = dcpl_id)
1013 call h5sclose_f(filespace, ierr)
1014 call h5pclose_f(dcpl_id, ierr)
1015 end if
1016
1017 dcount(1) = int(local_points, hsize_t)
1018 doffset(1) = int(time_offset + point_offset, hsize_t)
1019
1020 call h5dget_space_f(dset_id, filespace, ierr)
1021 call h5screate_simple_f(1, dcount, memspace, ierr)
1022 call h5sselect_hyperslab_f(filespace, h5s_select_set_f, &
1023 doffset, dcount, ierr)
1024
1025 if (precision .eq. rp) then
1026 call h5dwrite_f(dset_id, precision_hdf, fld%x, dcount, ierr, &
1027 file_space_id = filespace, mem_space_id = memspace, &
1028 xfer_prp = xf_id)
1029
1030 else if (precision .eq. sp) then
1031 block
1032 real(kind=rp), pointer :: x(:)
1033 real(kind=sp), allocatable :: data_1d(:)
1034
1035 x(1:local_points) => fld%x
1036 allocate(data_1d(local_points))
1037 do concurrent(local_idx = 1:local_points)
1038 data_1d(local_idx) = real(x(local_idx), sp)
1039 end do
1040
1041 call h5dwrite_f(dset_id, precision_hdf, data_1d, dcount, ierr, &
1042 file_space_id = filespace, mem_space_id = memspace, &
1043 xfer_prp = xf_id)
1044 deallocate(data_1d)
1045 end block
1046
1047 else if (precision .eq. dp) then
1048 block
1049 real(kind=rp), pointer :: x(:)
1050 real(kind=dp), allocatable :: data_1d(:)
1051
1052 x(1:local_points) => fld%x
1053 allocate(data_1d(local_points))
1054 do concurrent(local_idx = 1:local_points)
1055 data_1d(local_idx) = real(x(local_idx), dp)
1056 end do
1057
1058 call h5dwrite_f(dset_id, precision_hdf, data_1d, dcount, ierr, &
1059 file_space_id = filespace, mem_space_id = memspace, &
1060 xfer_prp = xf_id)
1061 deallocate(data_1d)
1062 end block
1063
1064 end if
1065
1066 call h5sclose_f(filespace, ierr)
1067 call h5sclose_f(memspace, ierr)
1068 call h5dclose_f(dset_id, ierr)
1069 end if
1070 end do
1071
1072 if (present(t)) call h5gclose_f(step_grp_id, ierr)
1073 call h5gclose_f(pointdata_grp, ierr)
1074 call h5pclose_f(xf_id, ierr)
1075
1076 deallocate(field_written)
1077
1078 end subroutine vtkhdf_write_pointdata
