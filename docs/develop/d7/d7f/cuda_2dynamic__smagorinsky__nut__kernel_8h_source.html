43__global__ void s_abs_compute(T * __restrict__ s_abs,
44 const T * __restrict__ s11,
45 const T * __restrict__ s22,
46 const T * __restrict__ s33,
47 const T * __restrict__ s12,
48 const T * __restrict__ s13,

51
52 const int idx = blockIdx.x * blockDim.x + threadIdx.x;
53 const int str = blockDim.x * gridDim.x;
54
55 for (int i = idx; i < n; i += str) {
56 const T s11_r = s11[i];

64 s22_r * s22_r +
65 s33_r * s33_r) +
66 4.0 * (s12_r * s12_r +
67 s13_r * s13_r +
68 s23_r * s23_r));

70 }

202template< typename T>
203__global__ void mij_nut_compute_part2(const T * __restrict__ m11,
204 const T * __restrict__ m22,
205 const T * __restrict__ m33,
206 const T * __restrict__ m12,
207 const T * __restrict__ m13,
208 const T * __restrict__ m23,
209 T * __restrict__ l11,
210 T * __restrict__ l22,
211 T * __restrict__ l33,
212 T * __restrict__ l12,
213 T * __restrict__ l13,
214 T * __restrict__ l23,
215 const T * __restrict__ fsabss11,
216 const T * __restrict__ fsabss22,
217 const T * __restrict__ fsabss33,
218 const T * __restrict__ fsabss12,
219 const T * __restrict__ fsabss13,
220 const T * __restrict__ fsabss23,
221 T * __restrict__ num,
222 T * __restrict__ den,
223 T * __restrict__ c_dyn,
224 const T * __restrict__ delta,
225 const T * __restrict__ s_abs,
226 T * __restrict__ nut,
227 const T alpha,
228 const int n){
229
230 const int idx = blockIdx.x * blockDim.x + threadIdx.x;
231 const int str = blockDim.x * gridDim.x;
232
233 T num_curr, den_curr;
234
235 for (int i = idx; i < n; i += str) {
236 const T delta_r = delta[i];
237 const T delta2 = delta_r * delta_r;
238 T c_dyn_r;
239
240 const T m11_r = delta2 * (m11[i] - fsabss11[i]);
241 const T m22_r = delta2 * (m22[i] - fsabss22[i]);
242 const T m33_r = delta2 * (m33[i] - fsabss33[i]);
243 const T m12_r = delta2 * (m12[i] - fsabss12[i]);
244 const T m13_r = delta2 * (m13[i] - fsabss13[i]);
245 const T m23_r = delta2 * (m23[i] - fsabss23[i]);
246
247 num_curr = m11_r * l11[i]
248 + m22_r * l22[i]
249 + m33_r * l33[i]
250 + 2.0 * (m12_r * l12[i]
251 + m13_r * l13[i]
252 + m23_r * l23[i]);
253
254 den_curr = m11_r * m11_r
255 + m22_r * m22_r
256 + m33_r * m33_r
257 + 2.0 * (m12_r * m12_r
258 + m13_r * m13_r
259 + m23_r * m23_r);
260
261 const T num_r = alpha * num[i] + (1.0 - alpha) * num_curr;
262 const T den_r = alpha * den[i] + (1.0 - alpha) * den_curr;
263
264 num[i] = num_r;
265 den[i] = den_r;
266
267 if (den_r > 0.0) {
268 c_dyn_r = 0.5 * num_r / den_r;
269 } else {
270 c_dyn_r = 0.0;
271 }
272
273 c_dyn_r = max(c_dyn_r, 0.0);
274 c_dyn[i] = c_dyn_r;
275 nut[i] = c_dyn_r * delta2 * s_abs[i];
276
277 }
278}
