69 subroutine compute_ri_b(bc_type, g_dot_n, hi, ti, ts, magu, kappa, q, Ri_b)
70 character(len=*), intent(in) :: bc_type
71 real(kind=rp), intent(in) :: hi, ti, ts
72 real(kind=rp), intent(in) :: magu, kappa, g_dot_n
73 real(kind=rp), intent(inout) :: q, ri_b
74
75 select case (bc_type)
76 case ("neumann")
77 ri_b = - g_dot_n*hi / ti*q / (magu**3*kappa**2)
78 case ("dirichlet")
79 ri_b = g_dot_n*hi/ti*(ti - ts)/magu**2
80 case default
81 call neko_error("Invalid specified temperature b.c. type ('neumann' or 'dirichlet'?)")
82 end select
83 end subroutine compute_ri_b

86 subroutine assign_bc_value(bc_type,bc_value,q,ts,ti,kappa,utau,z0h,hi)
87 character(len=*), intent(in) :: bc_type
88 real(kind=rp), intent(in) :: hi, ti, kappa, utau, z0h,bc_value
89 real(kind=rp), intent(inout) :: q,ts
90
91 select case (bc_type)
92 case ("neumann")
93 ! ts not used
94 q = bc_value
95 case ("dirichlet")
96 ts = bc_value
97 q = kappa*utau*(ts - ti)/log(hi/z0h)
98 case default
99 call neko_error("Invalid specified temperature b.c. type ('neumann' or 'dirichlet'?)")
100 end select
101 end subroutine assign_bc_value

104 subroutine set_stability_regime(Ri_b,Ri_threshold)
105 real(kind=rp), intent(in) :: ri_b, ri_threshold
106
107 if (ri_b > ri_threshold) then
108 tau_ptr => tau_stable
109 heat_flux_ptr => heat_flux_stable
110 elseif (ri_b < -ri_threshold) then
111 tau_ptr => tau_convective
112 heat_flux_ptr => heat_flux_convective
113 else
114 tau_ptr => tau_neutral
115 heat_flux_ptr => heat_flux_neutral
116 end if
117 end subroutine set_stability_regime

121 subroutine richardson_compute_cpu(u, v, w, temp, ind_r, ind_s, ind_t, ind_e, &
122 n_x, n_y, n_z, h, tau_x, tau_y, tau_z, n_nodes, lx, nelv, &
123 kappa, mu, rho, g_vec, Pr, z0, z0h_in, bc_type, bc_value, tstep, &
124 Ri_b_diagn, L_ob_diagn, utau_diagn, magu_diagn, ti_diagn, ts_diagn,&
125 q_diagn, h_x_idx, h_y_idx, h_z_idx)
126 integer, intent(in) :: n_nodes, lx, nelv, tstep
127 real(kind=rp), dimension(lx, lx, lx, nelv), intent(in) :: u, v, w, temp
128 integer, intent(in), dimension(n_nodes) :: ind_r, ind_s, ind_t, ind_e
129 real(kind=rp), dimension(n_nodes), intent(in) :: n_x, n_y, n_z, h
130 real(kind=rp), intent(in) :: kappa, z0, z0h_in, bc_value, mu, rho, pr
131 real(kind=rp), dimension(3), intent(in) :: g_vec
132 real(kind=rp) :: g_dot_n
133 character(len=*), intent(in) :: bc_type
134 real(kind=rp), dimension(n_nodes), intent(inout) :: tau_x, tau_y, tau_z
135 integer :: i
136 real(kind=rp) :: ui, vi, wi, hi
137 real(kind=rp) :: normu, z0h
138 real(kind=rp) :: l
139 real(kind=rp), parameter :: tol = 0.001_rp
140 real(kind=rp), parameter :: nr_step = 0.001_rp
141 real(kind=rp), parameter :: ri_threshold = 0.0001_rp
142 character(len=LOG_SIZE) :: log_buf
143 real(kind=rp) :: utau, ri_b, l_ob, magu, q, ti, ts
144 real(kind=rp), dimension(n_nodes), intent(inout) :: ri_b_diagn, l_ob_diagn
145 real(kind=rp), dimension(n_nodes), intent(inout) :: utau_diagn, magu_diagn
146 real(kind=rp), dimension(n_nodes), intent(inout) :: ti_diagn, ts_diagn
147 real(kind=rp), dimension(n_nodes), intent(inout) :: q_diagn
148 integer, dimension(n_nodes), intent(in) :: h_x_idx
149 integer, dimension(n_nodes), intent(in) :: h_y_idx
150 integer, dimension(n_nodes), intent(in) :: h_z_idx
151
152 do i=1, n_nodes
153 ! Sample the variables
154 ui = u(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
155 vi = v(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
156 wi = w(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
157 ti = temp(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
158 hi = h(i)
159
160 ! Project on horizontal directions
161 normu = ui * n_x(i) + vi * n_y(i) + wi * n_z(i)
162 ui = ui - normu * n_x(i)
163 vi = vi - normu * n_y(i)
164 wi = wi - normu * n_z(i)
165
166 ! Compute velocity magnitude
167 magu = sqrt(ui**2 + vi**2 + wi**2)
168 magu = max(magu, 1.0e-6_rp)
169 utau = magu*kappa / log(hi/z0)
170
171 ! Compute thermal roughness length from Zilitinkevich, 1995
172 if (z0h_in < 0) then
173 ! z0h_in is interpreted as -C_Zil (Zilitinkevich constant) for z0h
174 z0h = z0 * exp(z0h_in*sqrt((utau*z0)/(mu/rho)))
175 else
176 z0h = z0h_in
177 end if
178
179 ! Get q, ts based on bc_type
180 ! Maybe redundant, but needed to initialise Rib
181 call assign_bc_value(bc_type,bc_value,q,ts,ti,kappa,utau,z0h,hi)
182
183 ! Compute g along the normal (generalisation for hills and similar)
184 g_dot_n = abs(g_vec(1)*n_x(i) + g_vec(2)*n_y(i) + g_vec(3)*n_z(i))
185
186 ! Compute Richardson and set stability accordingly
187 call compute_ri_b(bc_type, g_dot_n, hi, ti, ts, magu, kappa, q, ri_b)
188 call set_stability_regime(ri_b, ri_threshold)
189
190 ! Set length scale
191 l = kappa * hi
192 ! Compute u*
193 utau = sqrt(tau_ptr(magu, ri_b, hi, z0, l, kappa))
194 select case (bc_type)
195 case ("neumann")
196 !!! TEMPORARY: neutral log-law approximation
197 ts = ti - (q * pr * log(hi/z0h)) / (max(utau, 1e-6_rp) * kappa)
198 q = q
199 case ("dirichlet")
200 ! Compute q
201 q = heat_flux_ptr(ti, ts, ri_b, hi, magu, z0h, pr, l, utau, kappa)
202 case default
203 call neko_error("Invalid specified temperature b.c. type ('neumann' or 'dirichlet'?)")
204 end select
205
206 ! Distribute according to the velocity vector and bound magu to avoid 0 division
207 magu = max(magu, 1.0e-6_rp)
208 tau_x(i) = -rho*utau**2 * ui / magu
209 tau_y(i) = -rho*utau**2 * vi / magu
210 tau_z(i) = -rho*utau**2 * wi / magu
211 if (abs(ri_b) <= ri_threshold) then
212 ! Neutral (L_ob undefined)
213 l_ob = 1e10_rp
214 else
215 l_ob = -(ts*utau**3)/(kappa*g_dot_n*q)
216 end if
217
218 ri_b_diagn(i) = ri_b
219 l_ob_diagn(i) = l_ob
220 utau_diagn(i) = utau
221 magu_diagn(i) = magu
222 ti_diagn(i) = ti
223 ts_diagn(i) = temp(ind_r(i)-h_x_idx(i), ind_s(i)-h_y_idx(i), ind_t(i)-h_z_idx(i), ind_e(i))
224 q_diagn(i) = q
225 end do
226
227 end subroutine richardson_compute_cpu

231 function tau_stable(magu, Ri_b, h, z0, l, kappa) result(tau)
232 real(kind=rp), intent(in) :: magu, ri_b, h, z0, l, kappa
233 real(kind=rp) :: tau
234
235 tau = magu**2/(log(h/z0)**2) * f_tau_stable(ri_b)/ &
236 f_tau_stable(0.0_rp) * (l/h)**2
237 end function tau_stable

239 function heat_flux_stable(ti, ts, Ri_b, h, magu, z0h, Pr,&
240 l, utau, kappa) result(heat_flux)
241 real(kind=rp), intent(in) :: ts, ti, ri_b, h, magu
242 real(kind=rp), intent(in) :: z0h, pr, l, utau, kappa
243 real(kind=rp) :: heat_flux
244
245 heat_flux = (ti - ts)/(log(h/z0h)) * &
246 f_theta_stable(ri_b)/abs(f_theta_stable(0.0_rp)) * &
247 (l/h) * utau/pr
248 end function heat_flux_stable

250 function f_tau_stable(Ri_b) result(f_tau)
251 real(kind=rp), intent(in) :: ri_b
252 real(kind=rp) :: f_tau
253
254 f_tau = 0.17 * (0.25 + 0.75 / (1.0 + 4.0*ri_b))
255 end function f_tau_stable

257 function f_theta_stable(Ri_b) result(f_theta)
258 real(kind=rp), intent(in) :: ri_b
259 real(kind=rp) :: f_theta
260
261 f_theta = -0.145 / (1.0 + 4.0 * ri_b)

266 function tau_convective(magu, Ri_b, h, z0, l, kappa) result(tau)
267 real(kind=rp), intent(in) :: magu, ri_b, h, z0, l, kappa
268 real(kind=rp) :: tau
269 real(kind=rp) :: a, b, c
270
271 a = kappa / log(h/z0)
272 b = 2.0
273 c = 7.4 * a**2 * b * (h/z0)**0.5
274
275 tau = a**2 * magu**2 * f_tau_convective(ri_b, c)
276 end function tau_convective

278 function heat_flux_convective(ti, ts, Ri_b, h, magu, z0h, Pr,&
279 l, utau, kappa) result(heat_flux)
280 real(kind=rp), intent(in) :: ts, ti, ri_b, h, magu
281 real(kind=rp), intent(in) :: z0h, pr, l, utau, kappa
282 real(kind=rp) :: heat_flux
283 real(kind=rp) :: a, b, c
284
285 a = kappa / log(h/z0h)
286 b = 2.0
287 c = 5.3 * a**2 * b * (h/z0h)**0.5
288
289 heat_flux = - a**2 / 0.74 * magu * &
290 (ti - ts) * f_theta_convective(ri_b, c)
291
292 end function heat_flux_convective

294 function f_tau_convective(Ri_b, c) result(f_tau)
295 real(kind=rp), intent(in) :: ri_b, c
296 real(kind=rp) :: f_tau
297
298 f_tau = 1.0 - 2*ri_b / (1.0 + c * abs(ri_b)**0.5)
299 end function f_tau_convective

301 function f_theta_convective(Ri_b, c) result(f_theta)
302 real(kind=rp), intent(in) :: ri_b, c
303 real(kind=rp) :: f_theta
304
305 f_theta = 1.0 - 2*ri_b / (1.0 + c * abs(ri_b)**0.5)
306 end function f_theta_convective
