143  subroutine gradient_jump_penalty_init(this, params, dofmap, coef, a, b)
144  implicit none

146  type(json_file), target, intent(inout) :: params

150 
151  integer :: i, j
152  real(kind=rp), allocatable :: zg(:) ! Quadrature points
153 
154  call this%free()
155 
156  this%p = dofmap%xh%lx - 1
157  this%lx = dofmap%xh%lx
158 
159  if (this%p .gt. 1) then
160  this%tau = -a * (this%p + 1) ** (-b)
161  else
162  this%tau = -a
163  end if
164 
165  this%coef => coef
166  this%n = this%lx ** 3 * this%coef%msh%nelv
167  this%n_large = (this%lx + 2) ** 3 * this%coef%msh%nelv
168 
169  allocate(this%n_facet(this%coef%msh%nelv))
170  do i = 1, this%coef%msh%nelv
171  select type (ep => this%coef%msh%elements(i)%e)
172  type is (hex_t)
173  this%n_facet(i) = 6
174  type is (quad_t)
175  call neko_error("Only Hexahedral element is &
176  &supported now for gradient jump penalty")
177  end select
178  end do
179  this%n_facet_max = maxval(this%n_facet)
180 
181  allocate(this%h2(this%lx + 2, this%lx + 2, &
182  this%lx + 2, this%coef%msh%nelv))

184  do i = 1, this%coef%msh%nelv
185  select type (ep => this%coef%msh%elements(i)%e)
186  type is (hex_t)
187  call eval_h2_hex(this%h2(:, :, :, i), this%lx, ep)
188  type is (quad_t)
189  call neko_error("Gradient jump penalty error: mesh size &
190  &evaluation is not supported for quad_t")
191  end select
192  end do
193 
194  allocate(zg(this%lx))
195  allocate(this%dphidxi(this%lx, this%lx))
196 
197  zg = dofmap%xh%zg(:,1)
198  do i = 1, dofmap%xh%lx
199  do j = 1, dofmap%xh%lx
200  this%dphidxi(j,i) = this%coef%Xh%dx(j,i)
201  end do
202  end do
203 
204  allocate(this%penalty(this%lx, this%lx, this%lx, this%coef%msh%nelv))
205  allocate(this%grad1(this%lx, this%lx, this%lx, this%coef%msh%nelv))
206  allocate(this%grad2(this%lx, this%lx, this%lx, this%coef%msh%nelv))
207  allocate(this%grad3(this%lx, this%lx, this%lx, this%coef%msh%nelv))
208 
209  allocate(this%penalty_facet(this%lx + 2, this%lx + 2, &
210  this%lx + 2, this%coef%msh%nelv))
211  allocate(this%G(this%lx + 2, this%lx + 2, &
212  this%lx + 2, this%coef%msh%nelv))
213  allocate(this%flux1(this%lx + 2, this%lx + 2, &
214  this%lx + 2, this%coef%msh%nelv))
215  allocate(this%flux2(this%lx + 2, this%lx + 2, &
216  this%lx + 2, this%coef%msh%nelv))
217  allocate(this%flux3(this%lx + 2, this%lx + 2, &
218  this%lx + 2, this%coef%msh%nelv))
219  allocate(this%volflux1(this%lx + 2, this%lx + 2, &
220  this%lx + 2, this%coef%msh%nelv))
221  allocate(this%volflux2(this%lx + 2, this%lx + 2, &
222  this%lx + 2, this%coef%msh%nelv))
223  allocate(this%volflux3(this%lx + 2, this%lx + 2, &
224  this%lx + 2, this%coef%msh%nelv))

229  allocate(this%n2(this%lx + 2, this%lx + 2, &

240  this%coef%nx(:, :, 2, :)

253  this%coef%ny(:, :, 4, :)

320  subroutine eval_h2_hex(h2_el, n, ep)
321  integer, intent(in) :: n
322  real(kind=rp), intent(inout) :: h2_el(n + 2, n + 2, n + 2)
323  type(hex_t), target, intent(in) :: ep
324 
325  integer :: i
326  type(point_t), pointer :: p1, p2, p3, p4, p5, p6, p7, p8
327 
328  p1 => ep%p(1)
329  p2 => ep%p(2)
330  p3 => ep%p(3)
331  p4 => ep%p(4)
332  p5 => ep%p(5)
333  p6 => ep%p(6)
334  p7 => ep%p(7)
335  p8 => ep%p(8)
336  h2_el = 0.0_rp
337 
338  h2_el(1, 2 : n + 1, 2 : n + 1) = dist2_facets_hex(p1, p5, p7, p3, &
339  p2, p6, p8, p4)
340  h2_el(n + 2, 2 : n + 1, 2 : n + 1) = h2_el(1, 2, 2)
341 
342  h2_el(2 : n + 1, 1, 2 : n + 1) = dist2_facets_hex(p1, p2, p6, p5, &
343  p3, p4, p8, p7)
344  h2_el(2 : n + 1, n + 2, 2 : n + 1) = h2_el(2, 1, 2)
345 
346  h2_el(2 : n + 1, 2 : n + 1, 1) = dist2_facets_hex(p1, p2, p4, p3, &
347  p5, p6, p8, p7)
348  h2_el(2 : n + 1, 2 : n + 1, n + 2) = h2_el(2, 2, 1)
349 
350  end subroutine eval_h2_hex

355  function dist2_facets_hex(p11, p12, p13, p14, &
356  p21, p22, p23, p24) result(dist2)
357  type(point_t), intent(in) :: p11, p12, p13, p14
358  type(point_t), intent(in) :: p21, p22, p23, p24
359  real(kind=rp) :: dist2
360 
361  dist2 = 0.0_rp
362 
363  dist2 = dist2 + dist_vertex_to_plane_3d(p11, p21, p22, p23)
364  dist2 = dist2 + dist_vertex_to_plane_3d(p11, p21, p22, p24)
365  dist2 = dist2 + dist_vertex_to_plane_3d(p11, p21, p23, p24)
366  dist2 = dist2 + dist_vertex_to_plane_3d(p11, p22, p23, p24)
367 
368  dist2 = dist2 + dist_vertex_to_plane_3d(p12, p21, p22, p23)
369  dist2 = dist2 + dist_vertex_to_plane_3d(p12, p21, p22, p24)
370  dist2 = dist2 + dist_vertex_to_plane_3d(p12, p21, p23, p24)
371  dist2 = dist2 + dist_vertex_to_plane_3d(p12, p22, p23, p24)
372 
373  dist2 = dist2 + dist_vertex_to_plane_3d(p13, p21, p22, p23)
374  dist2 = dist2 + dist_vertex_to_plane_3d(p13, p21, p22, p24)
375  dist2 = dist2 + dist_vertex_to_plane_3d(p13, p21, p23, p24)
376  dist2 = dist2 + dist_vertex_to_plane_3d(p13, p22, p23, p24)
377 
378  dist2 = dist2 + dist_vertex_to_plane_3d(p14, p21, p22, p23)
379  dist2 = dist2 + dist_vertex_to_plane_3d(p14, p21, p22, p24)
380  dist2 = dist2 + dist_vertex_to_plane_3d(p14, p21, p23, p24)
381  dist2 = dist2 + dist_vertex_to_plane_3d(p14, p22, p23, p24)
382 
383  dist2 = dist2 / 16.0_rp
384  dist2 = dist2 * dist2
385 
386  end function dist2_facets_hex

391  function dist_vertex_to_plane_3d(pv, p1, p2, p3) result(dist)
392  type(point_t), intent(in) :: pv, p1, p2, p3
394  real(kind=rp), dimension(3) :: u12, u13, uv1
395  real(kind=rp) :: norm_u12, norm_u13
397  real(kind=rp) :: un(3)
399  real(kind=rp) :: dist
400  integer :: i
401 
402  ! Set up u12 and u13
403  norm_u12 = 0.0_rp
404  norm_u13 = 0.0_rp
405  do i = 1, 3
406  u12(i) = p2%x(i) - p1%x(i)
407  norm_u12 = norm_u12 + u12(i) * u12(i)
408  u13(i) = p3%x(i) - p1%x(i)
409  norm_u13 = norm_u13 + u13(i) * u13(i)
410  uv1(i) = pv%x(i) - p1%x(i)
411  end do
412  norm_u12 = sqrt(norm_u12)
413  norm_u13 = sqrt(norm_u13)
414 
415  ! Normalized cross product of u12 and u13 to get un
416  un(1) = u12(2) * u13(3) - u12(3) * u13(2)
417  un(2) = - u12(1) * u13(3) + u12(3) * u13(1)
418  un(3) = u12(1) * u13(2) - u12(2) * u13(1)
419  do i = 1, 3
420  un(i) = un(i) / norm_u12 / norm_u13
421  end do
422 
423  ! Project of uv1 onto un, noting un is a unit vector
424  dist = 0.0_rp
425  do i = 1, 3
426  dist = dist + uv1(i) * un(i)
427  end do
428 
429  end function dist_vertex_to_plane_3d

432  subroutine facet_factor_init(this)
433  class(gradient_jump_penalty_t), intent(inout) :: this
435  real(kind=rp) :: wa(this%lx, this%lx, this%lx, this%coef%msh%nelv)
436 
437  allocate(this%facet_factor(this%lx + 2, this%lx + 2, &
438  this%lx + 2, this%coef%msh%nelv))
439 
440  associate(facet_factor => this%facet_factor, &
441  lx => this%lx, area => this%coef%area, &
442  nelv => this%coef%msh%nelv, &
443  jacinv => this%coef%jacinv, n => this%n, &
444  n_large => this%n_large, h2 => this%h2, &
445  tau => this%tau)
446 
447  ! Assemble facet_factor for facet 1 and 2
448  call add4(wa, this%coef%drdx, this%coef%drdy, this%coef%drdz, n)
449  call col2(wa, jacinv, n)
450  facet_factor(1, 2: lx + 1, 2: lx + 1, :) = -1.0_rp * wa(1, :, :, :)
451  facet_factor(lx + 2, 2: lx + 1, 2: lx + 1, :) = wa(lx, :, :, :)
452 
453  ! Assemble facet_factor for facet 3 and 4
454  call add4(wa, this%coef%dsdx, this%coef%dsdy, this%coef%dsdz, n)
455  call col2(wa, jacinv, n)
456  facet_factor(2: lx + 1, 1, 2: lx + 1, :) = -1.0_rp * wa(:, 1, :, :)
457  facet_factor(2: lx + 1, lx + 2, 2: lx + 1, :) = wa(:, lx, :, :)
458 
459  ! Assemble facet_factor for facet 5 and 6
460  call add4(wa, this%coef%dsdx, this%coef%dsdy, this%coef%dsdz, n)
461  call col2(wa, jacinv, n)
462  facet_factor(2: lx + 1, 2: lx + 1, 1, :) = -1.0_rp * wa(:, :, 1, :)
463  facet_factor(2: lx + 1, 2: lx + 1, lx + 2, :) = wa(:, :, lx, :)
464 
465  ! Multiplied by h^2 * tau
466  call col2(facet_factor, h2, n_large)
467  call cmult(facet_factor, tau, this%n_large)
468 
469  ! Multiplied by the quadrant weight
470  call col2(facet_factor(1, 2: lx + 1, 2: lx + 1, :), &
471  area(:, :, 1, :), lx * lx * nelv)
472  call col2(facet_factor(lx + 2, 2: lx + 1, 2: lx + 1, :), &
473  area(:, :, 2, :), lx * lx * nelv)
474  call col2(facet_factor(2: lx + 1, 1, 2: lx + 1, :), &
475  area(:, :, 3, :), lx * lx * nelv)
476  call col2(facet_factor(2: lx + 1, lx + 2, 2: lx + 1, :), &
477  area(:, :, 4, :), lx * lx * nelv)
478  call col2(facet_factor(2: lx + 1, 2: lx + 1, 1, :), &
479  area(:, :, 5, :), lx * lx * nelv)
480  call col2(facet_factor(2: lx + 1, 2: lx + 1, lx + 2, :), &
481  area(:, :, 6, :), lx * lx * nelv)
482 
483  end associate
484  end subroutine facet_factor_init
