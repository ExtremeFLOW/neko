434 subroutine wall_model_find_points(this)
435 class(wall_model_t), intent(inout) :: this
436 integer :: n_nodes, fid, idx(4), i, linear
437 real(kind=rp) :: normal(3), p(3), x, y, z, xw, yw, zw, magp
438 real(kind=rp) :: hmin, hmax
439 type(field_t), pointer :: h_field
440 type(file_t) :: h_file
441 character(len=LOG_SIZE), allocatable :: log_msg
442
443 n_nodes = this%msk(0)
444 this%n_nodes = n_nodes
445
446 call neko_field_registry%add_field(this%coef%dof, "sampling_height", &
447 ignore_existing=.true.)
448
449 h_field => neko_field_registry%get_field_by_name("sampling_height")
450
451 do i = 1, n_nodes
452 linear = this%msk(i)
453 fid = this%facet(i)
454 idx = nonlinear_index(linear, this%coef%Xh%lx, this%coef%Xh%ly,&
455 this%coef%Xh%lz)
456 normal = this%coef%get_normal(idx(1), idx(2), idx(3), idx(4), fid)
457
458 this%n_x%x(i) = normal(1)
459 this%n_y%x(i) = normal(2)
460 this%n_z%x(i) = normal(3)
461
462 ! inward normal
463 normal = -normal
464
465 select case (fid)
466 case (1)
467 this%ind_r(i) = idx(1) + this%h_index
468 this%ind_s(i) = idx(2)
469 this%ind_t(i) = idx(3)
470 case (2)
471 this%ind_r(i) = idx(1) - this%h_index
472 this%ind_s(i) = idx(2)
473 this%ind_t(i) = idx(3)
474 case (3)
475 this%ind_r(i) = idx(1)
476 this%ind_s(i) = idx(2) + this%h_index
477 this%ind_t(i) = idx(3)
478 case (4)
479 this%ind_r(i) = idx(1)
480 this%ind_s(i) = idx(2) - this%h_index
481 this%ind_t(i) = idx(3)
482 case (5)
483 this%ind_r(i) = idx(1)
484 this%ind_s(i) = idx(2)
485 this%ind_t(i) = idx(3) + this%h_index
486 case (6)
487 this%ind_r(i) = idx(1)
488 this%ind_s(i) = idx(2)
489 this%ind_t(i) = idx(3) - this%h_index
490 case default
491 call neko_error("The face index is not correct ")
492 end select
493 this%ind_e(i) = idx(4)
494
495 ! Location of the wall node
496 xw = this%dof%x(idx(1), idx(2), idx(3), idx(4))
497 yw = this%dof%y(idx(1), idx(2), idx(3), idx(4))
498 zw = this%dof%z(idx(1), idx(2), idx(3), idx(4))
499
500 ! Location of the sampling point
501 x = this%dof%x(this%ind_r(i), this%ind_s(i), this%ind_t(i), &
502 this%ind_e(i))
503 y = this%dof%y(this%ind_r(i), this%ind_s(i), this%ind_t(i), &
504 this%ind_e(i))
505 z = this%dof%z(this%ind_r(i), this%ind_s(i), this%ind_t(i), &
506 this%ind_e(i))
507
508 ! Vector from the sampling point to the wall
509 p(1) = x - xw
510 p(2) = y - yw
511 p(3) = z - zw
512
513 ! Total distance to the sampling point
514 magp = sqrt(p(1)**2 + p(2)**2 + p(3)**2)
515
516 ! Project on the normal direction to get h
517 this%h%x(i) = p(1)*normal(1) + p(2)*normal(2) + p(3)*normal(3)
518
519 h_field%x(linear,1,1,1) = this%h%x(i)
520
521 ! Look at how much the total distance distance from the normal and warn
522 ! if significant
523 if ((this%h%x(i) - magp) / magp > 0.1) then
524 write(log_msg,*) "Significant misalignment between wall normal and"
525 call neko_log%message(log_msg, neko_log_debug)
526 write(log_msg,*) "sampling point direction at wall node", xw, yw, zw
527 call neko_log%message(log_msg, neko_log_debug)
528 end if
529 end do
530
531! hmin = glmin(this%h%x, n_nodes)
532! hmax = glmax(this%h%x, n_nodes)
533! if (pe_rank .eq. 0) then
534! write(*, "(A, F10.4, F10.4)") " h min / max:", hmin, hmax
535! end if
536
537 if (neko_bcknd_device .eq. 1) then
538 call device_memcpy(this%h%x, this%h%x_d, n_nodes, host_to_device,&
539 sync = .false.)
540 call device_memcpy(this%n_x%x, this%n_x%x_d, n_nodes, host_to_device, &
541 sync = .false.)
542 call device_memcpy(this%n_y%x, this%n_y%x_d, n_nodes, host_to_device, &
543 sync = .false.)
544 call device_memcpy(this%n_z%x, this%n_z%x_d, n_nodes, host_to_device, &
545 sync = .true.)
546 end if
547
548 ! Each wall_model bc will do a write unfortunately... But very helpful
549 ! for setup debugging.
550 call h_file%init("sampling_height.fld")
551 call h_file%write(h_field)
552 end subroutine wall_model_find_points

554 subroutine wall_model_compute_mag_field(this)
555 class(wall_model_t), intent(inout) :: this
556 integer :: i, m
557 real(kind=rp) :: magtau
558
559 m = this%msk(0)
560 if (m > 0) then
561 if (neko_bcknd_device .eq. 1) then
562 call wall_model_compute_mag_field_device(this%tau_x%x_d, &
563 this%tau_y%x_d, &
564 this%tau_z%x_d, &
565 this%tau_field%x_d, &
566 this%msk_d, m)
567 else
568 do i = 1, m
569 magtau = sqrt(this%tau_x%x(i)**2 + &
570 this%tau_y%x(i)**2 + &
571 this%tau_z%x(i)**2)
572 this%tau_field%x(this%msk(i),1,1,1) = magtau
573 end do
574 end if
575 end if
576
577 end subroutine wall_model_compute_mag_field
