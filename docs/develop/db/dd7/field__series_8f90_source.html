59 subroutine field_series_init(this, f, len)
60 class(field_series_t), intent(inout) :: this
61 type(field_t), intent(inout), target :: f
62 integer :: len
63 character(len=80) :: name
64 character(len=5) :: id_str
65 integer :: i
66
67 call this%free()
68
69 this%f => f
