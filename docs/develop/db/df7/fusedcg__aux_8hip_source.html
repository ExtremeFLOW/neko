86 real hip_fusedcg_part2(void *a, void *b, void *c,
87 void *alpha_d , real *alpha, int *p_cur, int * n) {
88
89 const dim3 nthrds(1024, 1, 1);
90 const dim3 nblcks(((*n)+1024 - 1)/ 1024, 1, 1);
91 const int nb = ((*n) + 1024 - 1)/ 1024;
92 const hipStream_t stream = (hipStream_t) glb_cmd_queue;
93
94 if (fusedcg_buf != NULL && fusedcg_buf_len < nb) {
95 HIP_CHECK(hipHostFree(fusedcg_buf));
96 HIP_CHECK(hipFree(fusedcg_buf_d));
97 fusedcg_buf = NULL;
98 }
99
100 if (fusedcg_buf == NULL){
101 HIP_CHECK(hipHostMalloc(&fusedcg_buf, 2*sizeof(real)));
102 HIP_CHECK(hipMalloc(&fusedcg_buf_d, nb*sizeof(real)));
103 fusedcg_buf_len = nb;
104 }
105
106 /* Store alpha(p_cur) in pinned memory */
107 fusedcg_buf[1] = (*alpha);
108
109 /* Update alpha_d(p_cur) = alpha(p_cur) */
110 real *alpha_d_p_cur = ((real *) alpha_d) + ((*p_cur - 1));
111 HIP_CHECK(hipMemcpyAsync(alpha_d_p_cur, &fusedcg_buf[1], sizeof(real),
112 hipMemcpyHostToDevice, stream));
113
114 hipLaunchKernelGGL(HIP_KERNEL_NAME(fusedcg_part2_kernel<real>),
115 nblcks, nthrds, 0, stream,
116 (real *) a, (real *) b, (real *) c,
117 *alpha, fusedcg_buf_d, * n);
118 HIP_CHECK(hipGetLastError());
119
120 hipLaunchKernelGGL(HIP_KERNEL_NAME(reduce_kernel<real>),
121 1, 1024, 0, stream, fusedcg_buf_d, nb);
122 HIP_CHECK(hipGetLastError());
123
124#ifdef HAVE_RCCL
125 device_nccl_allreduce(fusedcg_buf_d, fusedcg_buf_d, 1, sizeof(real),
126 DEVICE_NCCL_SUM, stream);
127 HIP_CHECK(hipMemcpyAsync(fusedcg_buf, fusedcg_buf_d, sizeof(real),
128 hipMemcpyDeviceToHost, stream));
129 HIP_CHECK(hipStreamSynchronize(stream));
130#elif HAVE_DEVICE_MPI
131 HIP_CHECK(hipStreamSynchronize(stream));
132 device_mpi_allreduce(fusedcg_buf_d, fusedcg_buf, 1,
133 sizeof(real), DEVICE_MPI_SUM);
134#else
135 HIP_CHECK(hipMemcpyAsync(fusedcg_buf, fusedcg_buf_d, sizeof(real),
136 hipMemcpyDeviceToHost, stream));
137 HIP_CHECK(hipStreamSynchronize(stream));
138#endif
139
140 return fusedcg_buf[0];
141 }
