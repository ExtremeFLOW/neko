66 subroutine stl_file_read(this, data)
67 class(stl_file_t) :: this
68 class(*), target, intent(inout) :: data
69 type(tri_mesh_t), pointer :: tri_msh => null()
70 type(mpi_status) :: status
71 type(mpi_file) :: fh
72 type(point_t), target :: p1, p2, p3
73 type(stl_hdr_t) :: stl_hdr
74 type(stl_triangle_t), allocatable :: stl_tri(:)
75 integer :: i, p_idx, ierr
76
77 call this%check_exists()
78
79 select type (data)
80 type is (tri_mesh_t)
81 tri_msh => data
82 class default
83 call neko_log%error('Invalid data')
84 end select
85
86 call mpi_file_open(neko_comm, trim(this%get_fname()), &
87 mpi_mode_rdonly, mpi_info_null, fh, ierr)
88 call mpi_file_read_all(fh, stl_hdr, 1, mpi_stl_header, status, ierr)
89
90 if (stl_hdr%hdr(1:6) .eq. 'solid') then
91 call neko_log%error('Invalid STL file (ASCII)')
92 end if
93
94 call tri_msh%init(stl_hdr%ntri)
95 allocate(stl_tri(stl_hdr%ntri))
96
97 call mpi_file_read_all(fh, stl_tri, stl_hdr%ntri, &
98 mpi_stl_triangle, status, ierr)
99
100 p_idx = 0
101 do i = 1, stl_hdr%ntri
102 p_idx = p_idx + 1
103 call p1%init(dble(stl_tri(i)%v1), p_idx)
104 p_idx = p_idx + 1
105 call p2%init(dble(stl_tri(i)%v2), p_idx)
106 p_idx = p_idx + 1
107 call p3%init(dble(stl_tri(i)%v3), p_idx)
108 call tri_msh%add_element(p1, p2, p3)
109 end do
110
111 deallocate(stl_tri)
112
113 call mpi_file_close(fh, ierr)
114
115 end subroutine stl_file_read
