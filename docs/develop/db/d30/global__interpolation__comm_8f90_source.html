238 subroutine glb_intrp_comm_sendrecv(this, send, recv, n_send, n_recv)
239 class(glb_intrp_comm_t), intent(inout) :: this
240 integer, intent(in) :: n_send, n_recv
241 real(kind=rp), dimension(n_send), intent(inout) :: send
242 real(kind=rp), dimension(n_recv), intent(inout) :: recv
243 type(c_ptr) :: null_ptr = c_null_ptr
244 integer :: i, j, ierr, src, dst, thrdid
245 integer, pointer :: sp(:)
246 integer :: nreqs
247
248 thrdid = 0
249 !$ thrdid = omp_get_thread_num()
250
251 !
252 ! Issue non-blocking receives
253 !
254 do i = 1, size(this%recv_pe)
255 ! We should not need this extra associate block, ant it works
256 ! great without it for GNU, Intel, NEC and Cray, but throws an
257 ! ICE with NAG.
258 ! Issue recv requests, we will later check that these have finished
259 ! in nbwait
260 associate(recv_data => this%recv_buf(i)%data)
261 call mpi_irecv(recv_data, size(recv_data), &
262 mpi_real_precision, this%recv_pe(i), thrdid, &
263 this%comm, this%recv_buf(i)%request, ierr)
264 end associate
265 this%recv_buf(i)%flag = .false.
266 end do
267
268 !
269 ! Issue non-blocking sends
270 !
271 do i = 1, size(this%send_pe)
272 dst = this%send_pe(i)
273 ! Gather data from u into buffers according to indices in send_dof
274 ! We want to send contigous data to each process in send_pe
275 sp => this%send_dof(dst)%array()
276 do concurrent(j = 1:this%send_dof(dst)%size())
277 this%send_buf(i)%data(j) = send(sp(j))
278 end do
279 ! We should not need this extra associate block, ant it works
280 ! great without it for GNU, Intel, NEC and Cray, but throws an
281 ! ICE with NAG.
282 associate(send_data => this%send_buf(i)%data)
283 call mpi_isend(send_data, this%send_dof(dst)%size(), &
284 mpi_real_precision, this%send_pe(i), thrdid, &
285 this%comm, this%send_buf(i)%request, ierr)
286 end associate
287 this%send_buf(i)%flag = .false.
288 end do
289
290 !
291 ! Wait for non-blocking operations
292 !
293
294 nreqs = size(this%recv_pe)
295
296 do while (nreqs .gt. 0)
297 do i = 1, size(this%recv_pe)
298 if (.not. this%recv_buf(i)%flag) then
299 ! Check if we have recieved the data we want
300 call mpi_test(this%recv_buf(i)%request, this%recv_buf(i)%flag, &
301 this%recv_buf(i)%status, ierr)
302 ! If it has been received
303 if (this%recv_buf(i)%flag) then
304 ! One more request has been succesful
305 nreqs = nreqs - 1
307 src = this%recv_pe(i)
308 sp => this%recv_dof(src)%array()
309 !NEC$ IVDEP
310 do concurrent(j = 1:this%recv_dof(src)%size())
311 recv(sp(j)) = this%recv_buf(i)%data(j)
312 end do
313 end if
314 end if
315 end do
316 end do
317 ! Finally, check that the non-blocking sends this rank have issued have also
318 ! completed successfully
319
320 nreqs = size(this%send_pe)
321 do while (nreqs .gt. 0)
322 do i = 1, size(this%send_pe)
323 if (.not. this%send_buf(i)%flag) then
324 call mpi_test(this%send_buf(i)%request, this%send_buf(i)%flag, &
325 mpi_status_ignore, ierr)
326 if (this%send_buf(i)%flag) nreqs = nreqs - 1
327 end if
328 end do
329 end do
330
331 end subroutine glb_intrp_comm_sendrecv
