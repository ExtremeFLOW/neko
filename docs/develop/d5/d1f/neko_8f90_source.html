154 subroutine neko_init(C)
155 type(case_t), target, intent(inout), optional :: C
156 character(len=NEKO_FNAME_LEN) :: case_file, args
157 character(len=LOG_SIZE) :: log_buf
158 character(len=10) :: suffix
159 character(10) :: time
160 character(8) :: date
161 integer :: argc, i
162
163 call date_and_time(time = time, date = date)
164
165 call comm_init
166 call neko_mpi_types_init
167 call jobctrl_init
168 call device_init
169
170 call neko_log%init()
171 call neko_registry%init()
172 call neko_scratch_registry%init()
173
174 call neko_log%header(neko_version, neko_build_info)
175
176 if (present(c)) then
177
178 !
179 ! Command line arguments
180 !
181 argc = command_argument_count()
182
183 if (argc .lt. 1) then
184 if (pe_rank .eq. 0) write(*,*) 'Usage: ./neko <case file>'
185 stop
186 end if
187
188 call get_command_argument(1, case_file)
189
190 call filename_suffix(case_file, suffix)
191
192 if (trim(suffix) .ne. 'case' .and. trim(suffix) .ne. 'json') then
193 call neko_error('Invalid case file')
194 end if
195
196 if (argc .gt. 1) then
197 write(log_buf, '(a)') 'Running with command line arguments: '
198 call neko_log%message(log_buf, neko_log_quiet)
199 do i = 2, argc
200 call get_command_argument(i, args)
201 call neko_log%message(args, neko_log_quiet)
202 end do
203 end if
204
205 !
206 ! Job information
207 !
208 call neko_job_info(date, time)
209
210 !
211 ! Create case
212 !
213 call c%init(case_file)
214
215 !
216 ! Setup runtime statistics
217 !
218 call neko_rt_stats%init(c%params)
219
220
221 !
222 ! Create simulation components
223 !
224 call neko_simcomps%init(c)
225
226 end if
227
228 end subroutine neko_init
229
231 subroutine neko_solve(C)
232 type(case_t), target, intent(inout) :: C
233 type(time_step_controller_t) :: dt_controller
234 type(json_file) :: dt_params
235 real(kind=dp) :: tstep_loop_start_time
236
237 call json_get(c%params, 'case.time', dt_params)
238 call dt_controller%init(dt_params)
239
240 call c%time%reset()
241 call simulation_init(c, dt_controller)
242
243 call profiler_start
244 tstep_loop_start_time = mpi_wtime()
245
246 do while (.not. (c%time%is_done() .or. jobctrl_time_limit()))
247 call simulation_step(c, dt_controller, tstep_loop_start_time)
248 end do
249 call profiler_stop
250
251 call simulation_finalize(c)
252
253 end subroutine neko_solve
254
256 subroutine neko_finalize(C)
257 type(case_t), intent(inout), optional :: c
258
259 call neko_rt_stats%report()
260 call neko_rt_stats%free()
261
262 call neko_scratch_registry%free()
263
264 if (present(c)) then
265 call c%free()
266 end if
267
268 call neko_registry%free()
269 call neko_user_access%free()
270 call device_finalize
271 call neko_mpi_types_free
272 call comm_free
273 end subroutine neko_finalize

278 subroutine neko_job_info(date, time)
279 character(10), optional, intent(in) :: time
280 character(8), optional, intent(in) :: date
281 character(len=LOG_SIZE) :: log_buf
282 integer :: nthrds, rw, sw
283
284 call neko_log%section("Job Information")
285
286 if (present(time) .and. present(date)) then
287 write(log_buf, '(A,A,A,A,1x,A,1x,A,A,A,A,A)') 'Start time: ', &
288 time(1:2), ':', time(3:4), &
289 '/', date(1:4), '-', date(5:6), '-', date(7:8)
290 call neko_log%message(log_buf, neko_log_quiet)
291 end if
292 write(log_buf, '(a)') 'Running on: '
293 sw = 10
294 if (pe_size .lt. 1e1) then
295 write(log_buf(13:), '(i1,a)') pe_size, ' MPI '
296 if (pe_size .eq. 1) then
297 write(log_buf(19:), '(a)') 'rank'
298 sw = 9
299 else
300 write(log_buf(19:), '(a)') 'ranks'
301 end if
302 rw = 1
303 else if (pe_size .lt. 1e2) then
304 write(log_buf(13:), '(i2,a)') pe_size, ' MPI ranks'
305 rw = 2
306 else if (pe_size .lt. 1e3) then
307 write(log_buf(13:), '(i3,a)') pe_size, ' MPI ranks'
308 rw = 3
309 else if (pe_size .lt. 1e4) then
310 write(log_buf(13:), '(i4,a)') pe_size, ' MPI ranks'
311 rw = 4
312 else if (pe_size .lt. 1e5) then
313 write(log_buf(13:), '(i5,a)') pe_size, ' MPI ranks'
314 rw = 5
315 else
316 write(log_buf(13:), '(i6,a)') pe_size, ' MPI ranks'
317 rw = 6
318 end if
319 nthrds = 1
320 !$omp parallel
321 !$omp master
322 !$ nthrds = omp_get_num_threads()
323 !$omp end master
324 !$omp end parallel
325 if (nthrds .gt. 1) then
326 if (nthrds .lt. 1e1) then
327 write(log_buf(13 + rw + sw:), '(a,i1,a)') ', using ', &
328 nthrds, ' thrds each'
329 else if (nthrds .lt. 1e2) then
330 write(log_buf(13 + rw + sw:), '(a,i2,a)') ', using ', &
331 nthrds, ' thrds each'
332 else if (nthrds .lt. 1e3) then
333 write(log_buf(13 + rw + sw:), '(a,i3,a)') ', using ', &
334 nthrds, ' thrds each'
335 else if (nthrds .lt. 1e4) then
336 write(log_buf(13 + rw + sw:), '(a,i4,a)') ', using ', &
337 nthrds, ' thrds each'
338 end if
339 end if
340 call neko_log%message(log_buf, neko_log_quiet)
341
342 write(log_buf, '(a)') 'CPU type : '
343 call system_cpu_name(log_buf(13:))
344 call neko_log%message(log_buf, neko_log_quiet)
345
346 write(log_buf, '(a)') 'Bcknd type: '
347 if (neko_bcknd_sx .eq. 1) then
348 write(log_buf(13:), '(a)') 'SX-Aurora'
349 else if (neko_bcknd_xsmm .eq. 1) then
350 write(log_buf(13:), '(a)') 'CPU (libxsmm)'
351 else if (neko_bcknd_cuda .eq. 1) then
352 write(log_buf(13:), '(a)') 'Accelerator (CUDA)'
353 else if (neko_bcknd_hip .eq. 1) then
354 write(log_buf(13:), '(a)') 'Accelerator (HIP)'
355 else if (neko_bcknd_opencl .eq. 1) then
356 write(log_buf(13:), '(a)') 'Accelerator (OpenCL)'
357 else
358 write(log_buf(13:), '(a)') 'CPU'
359 end if
360 call neko_log%message(log_buf, neko_log_quiet)
361
362 if (neko_bcknd_hip .eq. 1 .or. neko_bcknd_cuda .eq. 1 .or. &
363 neko_bcknd_opencl .eq. 1) then
364 write(log_buf, '(a)') 'Dev. name : '
365 call device_name(log_buf(13:))
366 call neko_log%message(log_buf, neko_log_quiet)
367 end if
368 write(log_buf, '(a)') 'Real type : '
369 select case (rp)
370 case (real32)
371 write(log_buf(13:), '(a)') 'single precision'
372 case (real64)
373 write(log_buf(13:), '(a)') 'double precision'
374 case (real128)
375 write(log_buf(13:), '(a)') 'quad precision'
376 end select
377 call neko_log%message(log_buf, neko_log_quiet)
378 call neko_log%end()
379 end subroutine neko_job_info
380end module neko
