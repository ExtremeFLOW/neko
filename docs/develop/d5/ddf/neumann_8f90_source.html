79  ! Store non-linear index
80  integer :: idx(4)
81 
82  m = this%msk(0)
83  do i = 1, m
84  k = this%msk(i)
85  facet = this%facet(i)
86  idx = nonlinear_index(k, this%coef%Xh%lx, this%coef%Xh%lx,&
87  this%coef%Xh%lx)

89  case (1,2)
