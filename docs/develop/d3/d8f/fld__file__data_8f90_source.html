260 subroutine fld_file_data_generate_interpolator(this, global_interp, to_dof, &
261 to_msh, tolerance)
262 class(fld_file_data_t), intent(in) :: this
263 type(global_interpolation_t), intent(inout) :: global_interp
264 type(dofmap_t), intent(in), target :: to_dof
265 type(mesh_t), intent(in), target :: to_msh
266 real(kind=rp), intent(in) :: tolerance
267
268 ! --- variables for interpolation
269 type(space_t) :: fld_xh
270 real(kind=rp), allocatable :: x_coords(:,:,:,:), y_coords(:,:,:,:), &
271 z_coords(:,:,:,:)
272 real(kind=rp) :: center_x, center_y, center_z
273 integer :: e, i
274 ! ---
275
276 type(space_t), pointer :: to_Xh
277 to_xh => to_dof%Xh
278
279 ! Safeguard in case we didn't read mesh information
280 if (.not. allocated(this%x%x) .or. &
281 .not. allocated(this%y%x) .or. &
282 .not. allocated(this%z%x)) call neko_error("Unable to retrieve &
283 &mesh information from fld data.")
284
285 ! Create a space based on the fld data
286 call fld_xh%init(gll, this%lx, this%ly, this%lz)
287
288 ! These are the coordinates of our current dofmap
289 ! that we use for the interpolation
290 allocate(x_coords(to_xh%lx, to_xh%ly, to_xh%lz, to_msh%nelv))
291 allocate(y_coords(to_xh%lx, to_xh%ly, to_xh%lz, to_msh%nelv))
292 allocate(z_coords(to_xh%lx, to_xh%ly, to_xh%lz, to_msh%nelv))
293
298 do e = 1, to_msh%nelv
299 center_x = 0d0
300 center_y = 0d0
301 center_z = 0d0
302 do i = 1, to_xh%lxyz
303 center_x = center_x + to_dof%x(i, 1, 1, e)
304 center_y = center_y + to_dof%y(i, 1, 1, e)
305 center_z = center_z + to_dof%z(i, 1, 1, e)
306 end do
307 center_x = center_x / to_xh%lxyz
308 center_y = center_y / to_xh%lxyz
309 center_z = center_z / to_xh%lxyz
310 do i = 1, to_xh%lxyz
311 x_coords(i, 1, 1, e) = to_dof%x(i, 1, 1, e) - &
312 tolerance * (to_dof%x(i, 1, 1, e) - center_x)
313 y_coords(i, 1, 1, e) = to_dof%y(i, 1, 1, e) - &
314 tolerance * (to_dof%y(i, 1, 1, e) - center_y)
315 z_coords(i, 1, 1, e) = to_dof%z(i, 1, 1, e) - &
316 tolerance * (to_dof%z(i, 1, 1, e) - center_z)
317 end do
318 end do
319
320 ! The initialization is done based on the variables created from
321 ! fld data
322 call global_interp%init(this%x%x, this%y%x, this%z%x, this%gdim, &
323 this%nelv, fld_xh, tol = tolerance)
324 call global_interp%find_points(x_coords, y_coords, z_coords, &
325 to_dof%size())
326
327 deallocate(x_coords)
328 deallocate(y_coords)
329 deallocate(z_coords)
330 call fld_xh%free()
331
332 end subroutine fld_file_data_generate_interpolator
