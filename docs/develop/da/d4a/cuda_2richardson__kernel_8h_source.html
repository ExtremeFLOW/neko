41/*
42* Similarity laws and corrections for the STABLE regime:
43* Based on Mauritsen et al. 2007
44*/
45
46template<typename T>
47__device__ T f_tau_stable(T Ri_b)
48{
49 return 0.17 * (0.25 + 0.75 / (1.0 + 4.0 * Ri_b));
50}
51
52template<typename T>
53__device__ T f_theta_stable(T Ri_b)
54{
55 return -0.145 / (1.0 + 4.0 * Ri_b);
56}
57
58template<typename T>
59__device__ T tau_stable(T magu, T Ri_b, T h, T z0, T kappa)
60{
61 T log_hz0 = log(h / z0);
62 return (magu * magu) / (log_hz0 * log_hz0) * (f_tau_stable<T>(Ri_b) / f_tau_stable<T>(0.0)) * (kappa * kappa);
63}
64
65template<typename T>
66__device__ T heat_flux_stable(T ti, T ts, T Ri_b, T h, T z0h, T utau, T kappa, T Pr)
67{
68 return (ti - ts) / log(h / z0h) * (f_theta_stable<T>(Ri_b) / fabs(f_theta_stable<T>(0.0))) * kappa * (utau / Pr);
69}
70
71/*
72* Similarity laws and corrections for the UNSTABLE (convective) regime:
73* Based on Louis 1979
74*/
75
76template<typename T>
77__device__ T f_tau_convective(T Ri_b, T c)
78{
79 return 1.0 - (2.0 * Ri_b) / (1.0 + c * sqrt(fabs(Ri_b)));
80}
81
82template<typename T>
83__device__ T f_theta_convective(T Ri_b, T c)
84{
85 // Functionally identical to f_tau_convective in Louis 1979
86 return 1.0 - (2.0 * Ri_b) / (1.0 + c * sqrt(fabs(Ri_b)));
87}
88
89template<typename T>
90__device__ T tau_convective(T magu, T Ri_b, T h, T z0, T kappa)
91{
92 T a = kappa / log(h / z0);
93 T b = 2.0;
94 T c = 7.4 * (a * a) * b * sqrt(h / z0);
95
96 return (a * a) * (magu * magu) * f_tau_convective<T>(Ri_b, c);
97}
98
99template<typename T>
100__device__ T heat_flux_convective(T ti, T ts, T Ri_b, T h, T magu, T z0h, T kappa)
101{
102 T a = kappa / log(h / z0h);
103 T b = 2.0;
104 T c = 5.3 * (a * a) * b * sqrt(h / z0h);
105
106 return -(a * a) / 0.74 * magu * (ti - ts) * f_theta_convective<T>(Ri_b, c);
107}

109/*
110* Similarity laws and corrections for the NEUTRAL regime:
111*/
112
113template<typename T>
114__device__ T tau_neutral(T magu, T h, T z0, T kappa)
115{
116 T val = (kappa * magu) / log(h / z0);
117 return val * val;
118}
119
120template<typename T>
121__device__ T heat_flux_neutral(T ti, T ts, T h, T z0h, T utau, T kappa)
122{
123 return kappa * utau * (ti - ts) / log(h / z0h);
124}

126/*
127 * CUDA kernel for the Richardson wall model.
128 */
129template<typename T, int BC_TYPE>
130__global__ void richardson_compute(
131 const T* __restrict__ u_d,
132 const T* __restrict__ v_d,
133 const T* __restrict__ w_d,
134 const T* __restrict__ temp_d,
135 const T* __restrict__ h_d,
136 const T* __restrict__ n_x_d,
137 const T* __restrict__ n_y_d,
138 const T* __restrict__ n_z_d,
139 const int* __restrict__ ind_r_d,
140 const int* __restrict__ ind_s_d,
141 const int* __restrict__ ind_t_d,
142 const int* __restrict__ ind_e_d,
143 T* __restrict__ tau_x_d,
144 T* __restrict__ tau_y_d,
145 T* __restrict__ tau_z_d,
146 int n_nodes,
147 int lx,
148 T kappa,
149 T mu,
150 T rho,
151 T g1,
152 T g2,
153 T g3,
154 T Pr,
155 T z0,
156 T z0h_in,
157 T bc_value,
158 T* __restrict__ Ri_b_diagn,
159 T* __restrict__ L_ob_diagn,
160 T* __restrict__ utau_diagn,
161 T* __restrict__ magu_diagn,
162 T* __restrict__ ti_diagn,
163 T* __restrict__ ts_diagn,
164 T* __restrict__ q_diagn,
165 const int* __restrict__ h_x_idx,
166 const int* __restrict__ h_y_idx,
167 const int* __restrict__ h_z_idx
168) {
169 const int idx = blockIdx.x * blockDim.x + threadIdx.x;
170 const int str = blockDim.x * gridDim.x;
171 if(idx >= n_nodes) return;
172
173 const T Ri_threshold = 1e-4;
174
175 // Use 64-bit integer to prevent overflow for large polynomial order / element counts
176 for (int i = idx; i < n_nodes; i += str) {
177 const int index = (ind_e_d[i] - 1) * lx * lx * lx +
178 (ind_t_d[i] - 1) * lx * lx +
179 (ind_s_d[i] - 1) * lx +
180 (ind_r_d[i] - 1); // 'long long' might be needed to avoid
181 // overflowing 32-bits of 'int'
182
183 T ui = u_d[index];
184 T vi = v_d[index];
185 T wi = w_d[index];
186 T ti = temp_d[index];
187 T hi = h_d[i];
188
189 // Extract the local normal vector
190 T nx = n_x_d[i];
191 T ny = n_y_d[i];
192 T nz = n_z_d[i];
193
194 // Get the tangential component
195 T normu = ui * nx + vi * ny + wi * nz;
196 ui -= normu * nx;
197 vi -= normu * ny;
198 wi -= normu * nz;
199
200 T magu = sqrt(ui*ui + vi*vi + wi*wi);
201 magu = fmax(magu, (T)1e-6);
202
203 T utau = kappa * magu / log(hi/z0);
204
205 // Zilitinkevich 1995 correlation for thermal roughness
206 T z0h;
207 if (z0h_in < 0.0) {
208 // Note that this uses previous timestep's utau, hence
209 // lags behind by one dt. usually very negligible
210 z0h = z0 * exp(z0h_in * sqrt((utau*z0)/(mu/rho)));
211 } else {
212 z0h = z0h_in;
213 }
214
215 T ts = 0;
216 T q = 0;
217
218 // Initialize variables based on Boundary Condition
219 if constexpr (BC_TYPE == 0) { // Neumann
220 q = bc_value;
221 } else { // Dirichlet
222 ts = bc_value;
223 q = kappa * utau * (ts - ti) / log(hi/z0h);
224 }
225
226 T Ri_b;
227 T g_dot_n = fabs(g1*nx + g2*ny + g3*nz);
228
229 // Compute Bulk Richardson Number
230 if constexpr (BC_TYPE == 0) {
231 Ri_b = -g_dot_n * hi / ti * q / (magu * magu * magu * kappa * kappa);
232 } else {
233 Ri_b = g_dot_n * hi / ti * (ti - ts) / (magu * magu);
234 }
235
236 T tau_mag = 0;
237
238 // Stability regime branching
239 if (Ri_b > Ri_threshold) { // Stable
240 tau_mag = tau_stable<T>(magu, Ri_b, hi, z0, kappa);
241 utau = sqrt(tau_mag);
242 if constexpr (BC_TYPE == 1) {
243 q = heat_flux_stable<T>(ti, ts, Ri_b, hi, z0h, utau, kappa, Pr);
244 }
245 }
246 else if (Ri_b < -Ri_threshold) { // Convective
247 tau_mag = tau_convective<T>(magu, Ri_b, hi, z0, kappa);
248 utau = sqrt(tau_mag);
249 if constexpr (BC_TYPE == 1) {
250 q = heat_flux_convective<T>(ti, ts, Ri_b, hi, magu, z0h, kappa);
251 }
252 }
253 else { // Neutral
254 tau_mag = tau_neutral<T>(magu, hi, z0, kappa);
255 utau = sqrt(tau_mag);
256 if constexpr (BC_TYPE == 1) {
257 q = heat_flux_neutral<T>(ti, ts, hi, z0h, utau, kappa);
258 }
259 }
260
261 // Apply spatial distribution
262 tau_x_d[i] = -rho*tau_mag * ui / magu;
263 tau_y_d[i] = -rho*tau_mag * vi / magu;
264 tau_z_d[i] = -rho*tau_mag * wi / magu;
265
266 // Note: L_ob calculation is omitted from the kernel as it is a pure
267 // diagnostic variable and writing it to global memory would require
268 // passing an extra L_ob_d array pointer if GPU diagnostics are needed.
269
270 const int index_ts = (ind_e_d[i] - 1) * lx * lx * lx +
271 (ind_t_d[i] - 1 - h_z_idx[i]) * lx * lx +
272 (ind_s_d[i] - 1 - h_y_idx[i]) * lx +
273 (ind_r_d[i] - 1 - h_x_idx[i]);
274 Ri_b_diagn[i] = Ri_b;
275 L_ob_diagn[i] = 9999;
276 utau_diagn[i] = utau;
277 magu_diagn[i] = magu;
278 ti_diagn[i] = ti;
279 ts_diagn[i] = temp_d[index_ts];
280 q_diagn[i] = q;
281 }
282}
