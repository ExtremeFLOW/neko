147 nblcks, nthrds, 0, stream,
148 u_d, buf_d + offset,
149 dof_d + offset, n);
150
151 break;
152 case GS_OP_MIN:
153 hipLaunchKernelGGL(HIP_KERNEL_NAME(gs_unpack_min_kernel<real>),
154 nblcks, nthrds, 0, stream,
155 u_d, buf_d + offset,
156 dof_d + offset, n);
157 break;
158 case GS_OP_MAX:
159 hipLaunchKernelGGL(HIP_KERNEL_NAME(gs_unpack_max_kernel<real>),
160 nblcks, nthrds, 0, stream,
161 u_d, buf_d + offset,
162 dof_d + offset, n);
163 break;
164 default:
165 printf("%s: unknown gs op %d\n", __FILE__, op);
166 abort();
