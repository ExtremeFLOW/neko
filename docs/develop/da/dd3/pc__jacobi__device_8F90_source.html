98 subroutine device_jacobi_init(this, coef, dof, gs_h)
99 class(device_jacobi_t), intent(inout) :: this
100 type(coef_t), intent(in), target :: coef

103

157 subroutine device_jacobi_update(this)
158 class(device_jacobi_t), intent(inout) :: this
159 integer :: lz, ly, lx
160 associate(dof => this%dof, coef => this%coef, xh => this%dof%Xh, &
161 gs_h => this%gs_h, nelv => this%dof%msh%nelv)
162
163 lx = xh%lx
164 ly = xh%ly
165 lz = xh%lz
166
167#ifdef HAVE_HIP
168 call hip_jacobi_update(this%d_d, xh%dxt_d, xh%dyt_d, xh%dzt_d, &
169 coef%G11_d, coef%G22_d, coef%G33_d, &
170 coef%G12_d, coef%G13_d, coef%G23_d, &
171 nelv, lx)
172#elif HAVE_CUDA
173 call cuda_jacobi_update(this%d_d, xh%dxt_d, xh%dyt_d, xh%dzt_d, &
174 coef%G11_d, coef%G22_d, coef%G33_d, &
175 coef%G12_d, coef%G13_d, coef%G23_d, &
176 nelv, lx)
177#elif HAVE_OPENCL
178 call opencl_jacobi_update(this%d_d, xh%dxt_d, xh%dyt_d, xh%dzt_d, &
179 coef%G11_d, coef%G22_d, coef%G33_d, &
180 coef%G12_d, coef%G13_d, coef%G23_d, &
181 nelv, lx)
182#endif
183
184 call device_col2(this%d_d, coef%h1_d, coef%dof%size())
185
186 if (coef%ifh2) then
187 call device_addcol3(this%d_d, coef%h2_d, coef%B_d, coef%dof%size())
188 end if
189
190 call gs_h%op(this%d, dof%size(), gs_op_add, this%gs_event)
191 call device_event_sync(this%gs_event)
192
193 call device_invcol1(this%d_d, dof%size())
194 end associate
195 end subroutine device_jacobi_update
