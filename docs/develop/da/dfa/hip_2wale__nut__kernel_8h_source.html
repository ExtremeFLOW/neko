127 2.0 * (sd12*sd12 + sd13*sd13 + sd23*sd23);

131 max((sqrt(Sij_Sij*Sij_Sij*Sij_Sij*Sij_Sij*Sij_Sij)
132 + sqrt(sqrt(Sdij_Sdij*Sdij_Sdij*Sdij_Sdij*Sdij_Sdij*Sdij_Sdij))),
133 eps);
134
135
136 nut[i] = c * c * delta_r * delta_r
137 * OP_wale * mult[i];
