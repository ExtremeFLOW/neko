57 !! A registry of named `vector_t` objects.
58 !! @details Same as field_registry_t but for vectors. Stores named
59 !! vectors, but raw vectors are returned by the `get_` methods.
60 !! Supports adding aliases to vectors as well, so that the same vector can be
61 !! accessed by different names.

208 subroutine vector_registry_add_vector(this, n, name, ignore_existing)
209 class(vector_registry_t), intent(inout) :: this
210 integer, intent(in) :: n
211 character(len=*), target, intent(in) :: name
212 logical, optional, intent(in) :: ignore_existing
213 logical :: ignore_existing_
214
215 ignore_existing_ = .false.
216 if (present(ignore_existing)) then
217 ignore_existing_ = ignore_existing
218 end if
219
220 if (this%vector_exists(name)) then
221 if (ignore_existing_) then
222 return
223 else
224 call neko_error("Vector with name " // name // &
225 " is already registered")
226 end if
227 end if
228
229 if (this%n_vectors() == size(this%vectors)) then
230 call this%expand()
231 end if
232
233 this%n_vectors_ = this%n_vectors_ + 1
234
235 ! Initialize the named vector at the appropriate index
236 call this%vectors(this%n_vectors_)%init(name, n)

238 end subroutine vector_registry_add_vector

243 subroutine vector_registry_add_alias(this, alias, name)
244 class(vector_registry_t), intent(inout) :: this
245 character(len=*), target, intent(in) :: alias
246 character(len=*), target, intent(in) :: name
247
248 if (this%vector_exists(alias)) then
249 call neko_error("Cannot create alias. Vector " // alias // &
250 " already exists in the registry")
251 end if
252
253 if (this%vector_exists(name)) then
254 if (this%n_aliases_ == size(this%aliases)) then
255 call this%expand_aliases()
256 end if
257
258 this%n_aliases_ = this%n_aliases_ + 1
259
260 call this%aliases(this%n_aliases_)%initialize()
261 call this%aliases(this%n_aliases_)%add("alias", alias)
262 call this%aliases(this%n_aliases_)%add("target", name)
263 else
264 call neko_error("Cannot create alias. Vector " // name // &
265 " could not be found in the registry")
266 end if
267 end subroutine vector_registry_add_alias

317 recursive function vector_registry_get_vector_by_name(this, name) result(f)
318 class(vector_registry_t), target, intent(in) :: this
319 character(len=*), intent(in) :: name
320 character(len=:), allocatable :: alias
321 character(len=:), allocatable :: alias_target
322 type(vector_t), pointer :: f
323 logical :: found
324 integer :: i
325 type(json_file), pointer :: alias_json ! need this for some reason
326
327 found = .false.
328
329 do i = 1, this%n_vectors()
330 if (this%vectors(i)%name == trim(name)) then
331 f => this%vectors(i)%vector
332 found = .true.
333 exit
334 end if
335 end do
336
337 do i = 1, this%n_aliases()
338 alias_json => this%aliases(i)
339 call json_get(alias_json, "alias", alias)
340 if (alias == trim(name)) then
341 call json_get(alias_json, "target", alias_target)
342 f => this%get_vector_by_name(alias_target)
343 found = .true.
344 exit
345 end if
346 end do
347
348 if (.not. found) then
349 if (pe_rank .eq. 0) then
350 write(error_unit,*) "Current vector_registry contents:"
351
352 do i=1, this%n_vectors()
353 write(error_unit,*) "- ", this%vectors(i)%name
354 end do
355 end if
356 call neko_error("Vector " // name // &
357 " could not be found in the registry")
358 end if
359 end function vector_registry_get_vector_by_name

362 function vector_registry_vector_exists(this, name) result(found)
363 class(vector_registry_t), target, intent(in) :: this
364 character(len=*), intent(in) :: name
365 character(len=:), allocatable :: alias
366 logical :: found
367 integer :: i
368 type(json_file), pointer :: alias_json
369
370 found = .false.
371 do i=1, this%n_vectors()
372 if (this%vectors(i)%name == name) then
373 found = .true.
374 exit
375 end if
376 end do
377
378 do i=1, this%n_aliases()
379 alias_json => this%aliases(i)
380 call json_get(alias_json, "alias", alias)
381 if (alias == name) then
382 found = .true.
383 exit
384 end if
385 end do
