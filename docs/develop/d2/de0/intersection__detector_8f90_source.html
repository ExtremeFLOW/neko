58 subroutine intersect_detector_init(this, msh, padding)
59 class(intersect_detector_t), intent(inout) :: this
60 type(mesh_t), target, intent(in) :: msh
61 real(kind=dp), intent(in), optional :: padding
62 type(hex_t), allocatable :: elements(:)
63 integer :: i
64
65 call this%free()
66 this%msh => msh
67
68 call this%search_tree%init(msh%nelv)
69
71 allocate(elements(msh%nelv))
72 do i = 1, msh%nelv
73 select type(el => msh%elements(i)%e)
74 type is (hex_t)
75 elements(i) = el
76 class default
77 call neko_error('Unsupported element type')
78 end select
79 end do
80
81 if (present(padding)) then
82 call this%search_tree%build(elements, padding)
83 else
84 call this%search_tree%build(elements)
85 end if
86
87 deallocate(elements)
88
89 if (this%search_tree%get_size() .ne. (msh%nelv)) then
90 call neko_error("Error building the search tree.")
91 end if
92
93 end subroutine intersect_detector_init

104
105 end subroutine intersect_detector_free
