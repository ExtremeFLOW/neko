194 recursive subroutine tamg_matvec_impl(this, vec_out, vec_in, lvl, lvl_out)
195 class(tamg_hierarchy_t), intent(inout) :: this
196 real(kind=rp), intent(inout) :: vec_out(:)
197 real(kind=rp), intent(inout) :: vec_in(:)
198 integer, intent(in) :: lvl
199 integer, intent(in) :: lvl_out
200 integer :: i, n, e
201
202 vec_out = 0d0
203
204 if (lvl .eq. 0) then
206 n = size(vec_in)
208 call this%gs_h%op(vec_in, n, gs_op_add)
209 do i = 1, n
210 vec_in(i) = vec_in(i) * this%coef%mult(i,1,1,1)
211 end do
213 call this%ax%compute(vec_out, vec_in, this%coef, this%msh, this%Xh)
215 call this%gs_h%op(vec_out, n, gs_op_add)
216 call bc_list_apply(this%blst, vec_out, n)
218 else
219 if (lvl_out .ge. lvl) then
222 associate( wrk_in => this%lvl(lvl)%wrk_in, wrk_out => this%lvl(lvl)%wrk_out)
223 wrk_in = 0d0
224 wrk_out = 0d0
225 do n = 1, this%lvl(lvl)%nnodes
226 associate(node => this%lvl(lvl)%nodes(n))
227 do i = 1, node%ndofs
228 wrk_in( node%dofs(i) ) = wrk_in( node%dofs(i) ) + vec_in( node%gid ) * node%interp_p( i )
229 end do
230 end associate
231 end do
232
233 call this%matvec_impl(wrk_out, wrk_in, lvl-1, lvl_out)
234
236 do n = 1, this%lvl(lvl)%nnodes
237 associate(node => this%lvl(lvl)%nodes(n))
238 do i = 1, node%ndofs
239 vec_out( node%gid ) = vec_out(node%gid ) + wrk_out( node%dofs(i) ) * node%interp_r( i )
240 end do
241 end associate
242 end do
243 end associate
244 else if (lvl_out .lt. lvl) then
246 call this%matvec_impl(vec_out, vec_in, lvl-1, lvl_out)
247 else
248 call neko_error("TAMG: matvec level numbering problem.")
249 end if
250 end if
251 end subroutine tamg_matvec_impl

255 recursive subroutine tamg_matvec_flat_impl(this, vec_out, vec_in, lvl_blah, lvl_out)
256 class(tamg_hierarchy_t), intent(inout) :: this
257 real(kind=rp), intent(inout) :: vec_out(:)
258 real(kind=rp), intent(inout) :: vec_in(:)
259 integer, intent(in) :: lvl_blah
260 integer, intent(in) :: lvl_out
261 integer :: i, n, cdof, lvl
262
263 lvl = lvl_out
264 if (lvl .eq. 0) then
266 n = size(vec_in)
268 call this%gs_h%op(vec_in, n, gs_op_add)
269 do i = 1, n
270 vec_in(i) = vec_in(i) * this%coef%mult(i,1,1,1)
271 end do
273 call this%ax%compute(vec_out, vec_in, this%coef, this%msh, this%Xh)
275 call this%gs_h%op(vec_out, n, gs_op_add)
276 call bc_list_apply(this%blst, vec_out, n)
278 else
279
280 associate( wrk_in => this%lvl(1)%wrk_in, wrk_out => this%lvl(1)%wrk_out)
281 n = size(wrk_in)
282 wrk_out = 0d0
283 vec_out = 0d0
284
286 do i = 1, n
287 cdof = this%lvl(lvl)%map_f2c_dof(i)
288 wrk_in(i) = vec_in( cdof )
289 end do
290
292 call this%gs_h%op(wrk_in, n, gs_op_add)
293 do i = 1, n
294 wrk_in(i) = wrk_in(i) * this%coef%mult(i,1,1,1)
295 end do
297 call this%ax%compute(wrk_out, wrk_in, this%coef, this%msh, this%Xh)
299 call this%gs_h%op(wrk_out, n, gs_op_add)
300 call bc_list_apply(this%blst, wrk_out, n)
302
304 do i = 1, n
305 cdof = this%lvl(lvl)%map_f2c_dof(i)
306 vec_out(cdof) = vec_out(cdof) + wrk_out( i )
307 end do
308 end associate
309 end if
