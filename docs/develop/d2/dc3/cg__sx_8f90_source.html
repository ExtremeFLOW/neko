130 function sx_cg_solve(this, Ax, x, f, n, coef, blst, gs_h, niter) result(ksp_results)
131 class(sx_cg_t), intent(inout) :: this
132 class(ax_t), intent(in) :: ax
133 type(field_t), intent(inout) :: x
134 integer, intent(in) :: n
135 real(kind=rp), dimension(n), intent(in) :: f
136 type(coef_t), intent(inout) :: coef
137 type(bc_list_t), intent(inout) :: blst
138 type(gs_t), intent(inout) :: gs_h
139 type(ksp_monitor_t) :: ksp_results
140 integer, optional, intent(in) :: niter
141 real(kind=rp), parameter :: one = 1.0
142 real(kind=rp), parameter :: zero = 0.0
143 integer :: i, iter, max_iter
144 real(kind=rp) :: rnorm, rtr, rtr0, rtz2, rtz1
145 real(kind=rp) :: beta, pap, alpha, alphm, norm_fac
146
147 if (present(niter)) then
148 max_iter = niter
149 else
150 max_iter = this%max_iter
151 end if
152 norm_fac = one / sqrt(coef%volume)
153
154 rtz1 = one
155 do i = 1, n
156 x%x(i,1,1,1) = 0.0_rp
157 this%p(i) = 0.0_rp
158 this%r(i) = f(i)
159 end do
160
161 rtr = glsc3(this%r, coef%mult, this%r, n)
162 rnorm = sqrt(rtr)*norm_fac
163 ksp_results%res_start = rnorm
164 ksp_results%res_final = rnorm
165 ksp_results%iter = 0
166 if(abscmp(rnorm, zero)) then
167 ksp_results%converged = .true.
168 return
169 end if
170
171 call this%monitor_start('CG')
172 do iter = 1, max_iter
173 call this%M%solve(this%z, this%r, n)
174 rtz2 = rtz1
175 rtz1 = glsc3(this%r, coef%mult, this%z, n)
176
177 beta = rtz1 / rtz2
178 if (iter .eq. 1) beta = zero
179 call add2s1(this%p, this%z, beta, n)
180
181 call ax%compute(this%w, this%p, coef, x%msh, x%Xh)
182 call gs_h%op(this%w, n, gs_op_add)
183 call blst%apply_scalar(this%w, n)
184
185 pap = glsc3(this%w, coef%mult, this%p, n)
186
187 alpha = rtz1 / pap
188 alphm = -alpha
189 do i = 1, n
190 x%x(i,1,1,1) = x%x(i,1,1,1) + alpha * this%p(i)
191 this%r(i) = this%r(i) + alphm * this%w(i)
192 end do
193
194 rtr = glsc3(this%r, coef%mult, this%r, n)
195 if (iter .eq. 1) rtr0 = rtr
196 rnorm = sqrt(rtr) * norm_fac
197 call this%monitor_iter(iter, rnorm)
198 if (rnorm .lt. this%abs_tol) then
199 exit
200 end if
201 end do
202 call this%monitor_stop()
203 ksp_results%res_final = rnorm
204 ksp_results%iter = iter
205 ksp_results%converged = this%is_converged(iter, rnorm)
206 end function sx_cg_solve
