151 subroutine fluid_user_source_term_init_from_components(this, fields, coef, &
152 source_term_type, eval_vector, eval_pointwise)
153 class(fluid_user_source_term_t), intent(inout) :: this
154 type(field_list_t), intent(in), target :: fields
155 type(coef_t), intent(in), target :: coef
156 character(len=*) :: source_term_type
157 procedure(fluid_source_compute_vector), optional :: eval_vector
158 procedure(fluid_source_compute_pointwise), optional :: eval_pointwise
159
160 call this%free()
161 call this%init_base(fields, coef, 0.0_rp, huge(0.0_rp))
162
163 this%dm => fields%dof(1)
164
165 allocate(this%u(this%dm%Xh%lx, this%dm%Xh%ly, this%dm%Xh%lz, &
166 this%dm%msh%nelv))
167 allocate(this%v(this%dm%Xh%lx, this%dm%Xh%ly, this%dm%Xh%lz, &
168 this%dm%msh%nelv))
169 allocate(this%w(this%dm%Xh%lx, this%dm%Xh%ly, this%dm%Xh%lz, &
170 this%dm%msh%nelv))
171
172 this%u = 0d0
173 this%v = 0d0
174 this%w = 0d0
175
176 if (neko_bcknd_device .eq. 1) then
177 call device_map(this%u, this%u_d, this%dm%size())
178 call device_map(this%v, this%v_d, this%dm%size())
179 call device_map(this%w, this%w_d, this%dm%size())
180 end if
181
182
183 if (trim(source_term_type) .eq. 'user_pointwise' .and. &
184 present(eval_pointwise)) then
185 if (neko_bcknd_device .eq. 1) then
186 call neko_error('Pointwise source terms &
187 &not supported on accelerators')
188 end if
189 this%compute_vector_ => pointwise_eval_driver
190 this%compute_pw_ => eval_pointwise
191 else if (trim(source_term_type) .eq. 'user_vector' .and. &
192 present(eval_vector)) then
193 this%compute_vector_ => eval_vector
194 else
195 call neko_error('Invalid fluid source term '//source_term_type)
196 end if
197 end subroutine fluid_user_source_term_init_from_components

221 subroutine fluid_user_source_term_compute(this, t, tstep)
222 class(fluid_user_source_term_t), intent(inout) :: this
223 real(kind=rp), intent(in) :: t
224 integer, intent(in) :: tstep
225 integer :: n
226
227 if (t .ge. this%start_time .and. t .le. this%end_time) then
228 call this%compute_vector_(this, t)
229 n = this%fields%item_size(1)
230
231 if (neko_bcknd_device .eq. 1) then
232 call device_add2(this%fields%x_d(1), this%u_d, n)
233 call device_add2(this%fields%x_d(2), this%v_d, n)
234 call device_add2(this%fields%x_d(3), this%w_d, n)
235 else
236 call add2(this%fields%items(1)%ptr%x, this%u, n)
237 call add2(this%fields%items(2)%ptr%x, this%v, n)
238 call add2(this%fields%items(3)%ptr%x, this%w, n)
239 end if
240 end if

245 subroutine pointwise_eval_driver(this, t)
246 class(fluid_user_source_term_t), intent(inout) :: this
247 real(kind=rp), intent(in) :: t
248 integer :: j, k, l, e
249 integer :: jj, kk, ll, ee
250
251 select type (this)
252 type is (fluid_user_source_term_t)
253 do e = 1, size(this%u, 4)
254 ee = e
255 do l = 1, size(this%u, 3)
256 ll = l
257 do k = 1, size(this%u, 2)
258 kk = k
259 do j = 1, size(this%u, 1)
260 jj = j
261 call this%compute_pw_(this%u(j,k,l,e), &
262 this%v(j,k,l,e), &
263 this%w(j,k,l,e), &
264 jj, kk, ll, ee, t)
265 end do
266 end do
267 end do
268 end do
269 class default
270 call neko_error('Incorrect source type in pointwise eval driver!')
271 end select
