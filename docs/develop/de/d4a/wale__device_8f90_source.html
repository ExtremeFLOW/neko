60 subroutine wale_compute_device(if_ext, t, tstep, coef, nut, delta, c)
61 logical, intent(in) :: if_ext
62 real(kind=rp), intent(in) :: t
63 integer, intent(in) :: tstep
64 type(coef_t), intent(in) :: coef
65 type(field_t), intent(inout) :: nut
66 type(field_t), intent(in) :: delta
67 real(kind=rp), intent(in) :: c
68 ! This is the alpha tensor in the paper
69 type(field_t), pointer :: g11, g12, g13, g21, g22, g23, g31, g32, g33
70 type(field_t), pointer :: u, v, w
71
72 integer :: temp_indices(9)
73 integer :: e, i
74
75 if (if_ext .eqv. .true.) then
76 u => neko_registry%get_field_by_name("u_e")
77 v => neko_registry%get_field_by_name("v_e")
78 w => neko_registry%get_field_by_name("w_e")
79 else
80 u => neko_registry%get_field_by_name("u")
81 v => neko_registry%get_field_by_name("v")
82 w => neko_registry%get_field_by_name("w")
83 end if
84
85 call neko_scratch_registry%request_field(g11, temp_indices(1), .false.)
86 call neko_scratch_registry%request_field(g12, temp_indices(2), .false.)
87 call neko_scratch_registry%request_field(g13, temp_indices(3), .false.)
88 call neko_scratch_registry%request_field(g21, temp_indices(4), .false.)
89 call neko_scratch_registry%request_field(g22, temp_indices(5), .false.)
90 call neko_scratch_registry%request_field(g23, temp_indices(6), .false.)
91 call neko_scratch_registry%request_field(g31, temp_indices(7), .false.)
92 call neko_scratch_registry%request_field(g32, temp_indices(8), .false.)
93 call neko_scratch_registry%request_field(g33, temp_indices(9), .false.)
94
95
96 ! Compute the derivatives of the velocity (the alpha tensor)
97 call dudxyz (g11%x, u%x, coef%drdx, coef%dsdx, coef%dtdx, coef)
98 call dudxyz (g12%x, u%x, coef%drdy, coef%dsdy, coef%dtdy, coef)
99 call dudxyz (g13%x, u%x, coef%drdz, coef%dsdz, coef%dtdz, coef)
100
101 call dudxyz (g21%x, v%x, coef%drdx, coef%dsdx, coef%dtdx, coef)
102 call dudxyz (g22%x, v%x, coef%drdy, coef%dsdy, coef%dtdy, coef)
103 call dudxyz (g23%x, v%x, coef%drdz, coef%dsdz, coef%dtdz, coef)
104
105 call dudxyz (g31%x, w%x, coef%drdx, coef%dsdx, coef%dtdx, coef)
106 call dudxyz (g32%x, w%x, coef%drdy, coef%dsdy, coef%dtdy, coef)
107 call dudxyz (g33%x, w%x, coef%drdz, coef%dsdz, coef%dtdz, coef)
108
109 call coef%gs_h%op(g11, gs_op_add)
110 call coef%gs_h%op(g12, gs_op_add)
111 call coef%gs_h%op(g13, gs_op_add)
112 call coef%gs_h%op(g21, gs_op_add)
113 call coef%gs_h%op(g22, gs_op_add)
114 call coef%gs_h%op(g23, gs_op_add)
115 call coef%gs_h%op(g31, gs_op_add)
116 call coef%gs_h%op(g32, gs_op_add)
117 call coef%gs_h%op(g33, gs_op_add)
118
119 call device_wale_nut_compute(g11%x_d, g12%x_d, g13%x_d, &
120 g21%x_d, g22%x_d, g23%x_d, &
121 g31%x_d, g32%x_d, g33%x_d, &
122 delta%x_d, nut%x_d, coef%mult_d, &
123 c, neko_eps, g11%dof%size())
124
125 call coef%gs_h%op(nut, gs_op_add)
126 call device_col2(nut%x_d, coef%mult_d, nut%dof%size())
127
128 call neko_scratch_registry%relinquish_field(temp_indices)
129 end subroutine wale_compute_device
130
