68 subroutine dynamic_smagorinsky_compute_device(if_ext, t, tstep, coef, nut, &
69 delta, c_dyn, test_filter, mij, lij, num, den)
70 logical, intent(in) :: if_ext
71 real(kind=rp), intent(in) :: t
72 integer, intent(in) :: tstep
73 type(coef_t), intent(in) :: coef
74 type(field_t), intent(inout) :: nut
75 type(field_t), intent(in) :: delta
76 type(field_t), intent(inout) :: c_dyn
77 type(elementwise_filter_t), intent(inout) :: test_filter
78 type(field_t), intent(inout) :: mij(6), lij(6)
79 type(field_t), intent(inout) :: num, den
80
81 type(field_t), pointer :: u, v, w
82 type(field_t) :: c_dyn_curr
83 ! the strain rate tensor
84 type(field_t), pointer :: s11, s22, s33, s12, s13, s23, s_abs
85 real(kind=rp) :: alpha ! running averaging coefficient
86 integer :: tmp_indices(7)
87 integer :: i
88
89 if (tstep .eq. 1) then
90 alpha = 1.0_rp
91 else
92 alpha = 0.9_rp
93 end if
94
95 if (if_ext .eqv. .true.) then
96 u => neko_registry%get_field_by_name("u_e")
97 v => neko_registry%get_field_by_name("v_e")
98 w => neko_registry%get_field_by_name("w_e")
99 else
100 u => neko_registry%get_field_by_name("u")
101 v => neko_registry%get_field_by_name("v")
102 w => neko_registry%get_field_by_name("w")
103 end if
104
105 call neko_scratch_registry%request_field(s11, tmp_indices(1), .false.)
106 call neko_scratch_registry%request_field(s22, tmp_indices(2), .false.)
107 call neko_scratch_registry%request_field(s33, tmp_indices(3), .false.)
108 call neko_scratch_registry%request_field(s12, tmp_indices(4), .false.)
109 call neko_scratch_registry%request_field(s13, tmp_indices(5), .false.)
110 call neko_scratch_registry%request_field(s23, tmp_indices(6), .false.)
111 call neko_scratch_registry%request_field(s_abs, tmp_indices(7), .false.)
112
113 ! Compute the strain rate tensor
114 call strain_rate(s11%x, s22%x, s33%x, s12%x, s13%x, s23%x, u, v, w, coef)
115
116 call coef%gs_h%op(s11%x, s11%dof%size(), gs_op_add)
117 call coef%gs_h%op(s22%x, s11%dof%size(), gs_op_add)
118 call coef%gs_h%op(s33%x, s11%dof%size(), gs_op_add)
119 call coef%gs_h%op(s12%x, s11%dof%size(), gs_op_add)
120 call coef%gs_h%op(s13%x, s11%dof%size(), gs_op_add)
121 call coef%gs_h%op(s23%x, s11%dof%size(), gs_op_add)
122
123 call device_col2(s11%x_d, coef%mult_d, s11%dof%size())
124 call device_col2(s22%x_d, coef%mult_d, s22%dof%size())
125 call device_col2(s33%x_d, coef%mult_d, s33%dof%size())
126 call device_col2(s12%x_d, coef%mult_d, s12%dof%size())
127 call device_col2(s13%x_d, coef%mult_d, s13%dof%size())
128 call device_col2(s23%x_d, coef%mult_d, s23%dof%size())
129
130 call device_s_abs_compute(s_abs%x_d, s11%x_d, s22%x_d, s33%x_d, &
131 s12%x_d, s13%x_d, s23%x_d, &
132 s11%dof%size())
133
134 call compute_lij_device(lij, u, v, w, test_filter, u%dof%size())
135 call compute_nut_device(nut, c_dyn, num, den, lij, mij, &
136 s11, s22, s33, s12, s13, s23, &
137 s_abs, test_filter, delta, alpha, &
138 coef, u%dof%size())
139
140 call coef%gs_h%op(nut, gs_op_add)
141 call device_col2(nut%x_d, coef%mult_d, nut%dof%size())
142
143 call neko_scratch_registry%relinquish_field(tmp_indices)
144 end subroutine dynamic_smagorinsky_compute_device

155 subroutine compute_lij_device(lij, u, v, w, test_filter, n)
156 type(field_t), intent(inout) :: lij(6)
157 type(field_t), pointer, intent(in) :: u, v, w
158 type(elementwise_filter_t), intent(inout) :: test_filter
159 integer, intent(in) :: n
160 integer :: i
162 type(field_t), pointer :: fu, fv, fw, fuu, fvv, fww, fuv, fuw, fvw
163 integer :: tmp_indices(9)
164
165 call neko_scratch_registry%request_field(fu, tmp_indices(1), .false.)
166 call neko_scratch_registry%request_field(fv, tmp_indices(2), .false.)
167 call neko_scratch_registry%request_field(fw, tmp_indices(3), .false.)
168 call neko_scratch_registry%request_field(fuu, tmp_indices(4), .false.)
169 call neko_scratch_registry%request_field(fvv, tmp_indices(5), .false.)
170 call neko_scratch_registry%request_field(fww, tmp_indices(6), .false.)
171 call neko_scratch_registry%request_field(fuv, tmp_indices(7), .false.)
172 call neko_scratch_registry%request_field(fuw, tmp_indices(8), .false.)
173 call neko_scratch_registry%request_field(fvw, tmp_indices(9), .false.)
174
175 ! Use test filter for the velocity fields
176 call test_filter%apply(fu, u)
177 call test_filter%apply(fv, v)
178 call test_filter%apply(fw, w)
179
180 !! ___ ___
181 !! Compute u_i*u_j and u_i*u_j
182 call device_lij_compute_part1(lij(1)%x_d, lij(2)%x_d, lij(3)%x_d, &
183 lij(4)%x_d, lij(5)%x_d, lij(6)%x_d, &
184 u%x_d, v%x_d, w%x_d, &
185 fu%x_d, fv%x_d, fw%x_d, &
186 fuu%x_d, fvv%x_d, fww%x_d, &
187 fuv%x_d, fuw%x_d, fvw%x_d, n)
188
189 !! Filter u_i*u_j by the test filter
190 call test_filter%apply(fuu, fuu)
191 call test_filter%apply(fvv, fvv)
192 call test_filter%apply(fww, fww)
193 call test_filter%apply(fuv, fuv)
194 call test_filter%apply(fuw, fuw)
195 call test_filter%apply(fvw, fvw)
196
197 !! Assember the final form
198 !! ___ ___ _______
199 !! L_ij = u_i*u_j - u_i*u_j
200 call device_lij_compute_part2(lij(1)%x_d, lij(2)%x_d, lij(3)%x_d, &
201 lij(4)%x_d, lij(5)%x_d, lij(6)%x_d, &
202 fuu%x_d, fvv%x_d, fww%x_d, &
203 fuv%x_d, fuw%x_d, fvw%x_d, n)
204 call neko_scratch_registry%relinquish_field(tmp_indices)
205 end subroutine compute_lij_device

227 subroutine compute_nut_device(nut, c_dyn, num, den, lij, mij, &
228 s11, s22, s33, s12, s13, s23, &
229 s_abs, test_filter, delta, alpha, &
230 coef, n)
231 type(field_t), intent(inout) :: nut, c_dyn
232 type(field_t), intent(inout) :: num, den
233 type(field_t), intent(in) :: lij(6)
234 type(field_t), intent(inout) :: mij(6)
235 type(field_t), intent(inout) :: s11, s22, s33, s12, s13, s23, s_abs
236 type(elementwise_filter_t), intent(inout) :: test_filter
237 type(field_t), intent(in) :: delta
238 real(kind=rp), intent(in) :: alpha
239 type(coef_t), intent(in) :: coef
240 integer, intent(in) :: n
241
242 real(kind=rp) :: delta_ratio2
243 integer :: tmp_indices(13)
244 type(field_t), pointer :: fs11, fs22, fs33, fs12, fs13, fs23, fs_abs, &
245 fsabss11, fsabss22, fsabss33, &
246 fsabss12, fsabss13, fsabss23
247
248 delta_ratio2 = ((test_filter%nx-1.0_rp)/(test_filter%nt-1.0_rp))**2
249
250 call neko_scratch_registry%request_field(fs11, tmp_indices(1), .false.)
251 call neko_scratch_registry%request_field(fs22, tmp_indices(2), .false.)
252 call neko_scratch_registry%request_field(fs33, tmp_indices(3), .false.)
253 call neko_scratch_registry%request_field(fs12, tmp_indices(4), .false.)
254 call neko_scratch_registry%request_field(fs13, tmp_indices(5), .false.)
255 call neko_scratch_registry%request_field(fs23, tmp_indices(6), .false.)
256 call neko_scratch_registry%request_field(fsabss11, tmp_indices(7), .false.)
257 call neko_scratch_registry%request_field(fsabss22, tmp_indices(8), .false.)
258 call neko_scratch_registry%request_field(fsabss33, tmp_indices(9), .false.)
259 call neko_scratch_registry%request_field(fsabss12, tmp_indices(10), .false.)
260 call neko_scratch_registry%request_field(fsabss13, tmp_indices(11), .false.)
261 call neko_scratch_registry%request_field(fsabss23, tmp_indices(12), .false.)
262 call neko_scratch_registry%request_field(fs_abs, tmp_indices(13), .false.)
263
264 !! Compute M_ij
265 !! _____ ____
266 !! Compute s_abs and s_ij
267 call test_filter%apply(fs_abs, s_abs)
268 call test_filter%apply(fs11, s11)
269 call test_filter%apply(fs22, s22)
270 call test_filter%apply(fs33, s33)
271 call test_filter%apply(fs12, s12)
272 call test_filter%apply(fs13, s13)
273 call test_filter%apply(fs23, s23)
274
275 !! _____ ____
276 !! Compute (delta_test/delta)^2 s_abs*s_ij and s_abs*s_ij
277 call device_mij_compute_part1(mij(1)%x_d, mij(2)%x_d, mij(3)%x_d, &
278 mij(4)%x_d, mij(5)%x_d, mij(6)%x_d, &
279 s_abs%x_d, s11%x_d, s22%x_d, s33%x_d, &
280 s12%x_d, s13%x_d, s23%x_d, &
281 fs_abs%x_d, fs11%x_d, fs22%x_d, fs33%x_d, &
282 fs12%x_d, fs13%x_d, fs23%x_d, &
283 fsabss11%x_d, fsabss22%x_d, fsabss33%x_d, &
284 fsabss12%x_d, fsabss13%x_d, fsabss23%x_d, &
285 delta_ratio2, n)
286
287 !! Filter s_abs*s_ij by the test filter
288 call test_filter%apply(fsabss11, fsabss11)
289 call test_filter%apply(fsabss22, fsabss22)
290 call test_filter%apply(fsabss33, fsabss33)
291 call test_filter%apply(fsabss12, fsabss12)
292 call test_filter%apply(fsabss13, fsabss13)
293 call test_filter%apply(fsabss23, fsabss23)
294
295 !! Finalise the compute of Mij and nut
296 call device_mij_nut_compute_part2(mij(1)%x_d, mij(2)%x_d, mij(3)%x_d, &
297 mij(4)%x_d, mij(5)%x_d, mij(6)%x_d, &
298 lij(1)%x_d, lij(2)%x_d, lij(3)%x_d, &
299 lij(4)%x_d, lij(5)%x_d, lij(6)%x_d, &
300 fsabss11%x_d, fsabss22%x_d, fsabss33%x_d, &
301 fsabss12%x_d, fsabss13%x_d, fsabss23%x_d, &
302 num%x_d, den%x_d, c_dyn%x_d, delta%x_d, &
303 s_abs%x_d, nut%x_d, alpha, n)
304 call neko_scratch_registry%relinquish_field(tmp_indices)
305 end subroutine compute_nut_device
