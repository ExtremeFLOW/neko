283  subroutine aggregate_greedy(tamg, lvl_id, max_aggs, facet_neigh, agg_nhbr)
284  type(tamg_hierarchy_t), intent(inout) :: tamg
285  integer, intent(in) :: lvl_id
286  integer, intent(in) :: max_aggs
287  integer, intent(in) :: facet_neigh(:,:)
288  integer, intent(inout), allocatable :: agg_nhbr(:,:)
289  integer, allocatable :: is_aggregated(:)
290  integer, allocatable :: aggregate_size(:)
291  integer :: n_elements, naggs, n_facet, offset_el
292  integer :: i, j, l, ntot, n_agg_facet
293 
294  if (lvl_id .lt. 2) then
295  call neko_error("For now, can only use greedy agg after elms have been aggregated to points (level 1)")
296  else if (lvl_id .eq. 2) then
297  n_facet = 6
298  offset_el = tamg%msh%offset_el
299  else
300  n_facet = 50!TODO: this hard-coded value. how many neighbors can there be?
301  offset_el = 0
302  end if
303 
304  naggs = 0
305  n_elements = tamg%lvl(lvl_id-1)%nnodes
306  allocate( is_aggregated( n_elements ) )
307  allocate( aggregate_size( max_aggs ) )
308 
310  is_aggregated = -1
311 
313  aggregate_size = huge(i)!999999
314 
316  call agg_greedy_first_pass(naggs, max_aggs, n_elements, &
317  facet_neigh, offset_el, n_facet, &
318  is_aggregated, aggregate_size)
319 
320  call aggregation_monitor_phase1(lvl_id, n_elements, naggs, is_aggregated)
321 
323  call agg_greedy_second_pass(naggs, max_aggs, n_elements, &
324  facet_neigh, offset_el, n_facet, &
325  is_aggregated, aggregate_size)
326 
327  call aggregation_monitor_phase2(lvl_id, n_elements, naggs, is_aggregated)
328 
329  if (.true.) then
330  allocate( agg_nhbr(50, max_aggs*2) )!TODO: this hard-coded n_facet value (20)...
331  agg_nhbr = -1
332  call agg_fill_nhbr_info(agg_nhbr, n_agg_facet, n_elements, &
333  facet_neigh, offset_el, n_facet, &
334  is_aggregated, aggregate_size)
335  end if
336 
338  ntot = 0
339  call tamg_lvl_init( tamg%lvl(lvl_id), lvl_id, naggs)
340  do l = 1, naggs
341  call tamg_node_init( tamg%lvl(lvl_id)%nodes(l), l, aggregate_size(l))
342  j = 0
343  do i = 1, n_elements!TODO: this is the lazy expensive way...
344  if (is_aggregated(i) .eq. l) then
345  j = j+1
346  tamg%lvl(lvl_id)%nodes(l)%dofs(j) = i
347  end if
348  end do
349  if (j .ne. tamg%lvl(lvl_id)%nodes(l)%ndofs) then
350  call neko_error("Aggregation problem. Not enough dofs in node.")
351  end if
352  ntot = ntot + aggregate_size(l)
353  end do
354  tamg%lvl(lvl_id)%fine_lvl_dofs = ntot
355  allocate( tamg%lvl(lvl_id)%wrk_in( ntot ) )
356  allocate( tamg%lvl(lvl_id)%wrk_out( ntot ) )
357 
358  call aggregation_monitor_final(lvl_id,ntot,naggs)
359 
360  deallocate( is_aggregated )
361  deallocate( aggregate_size )
362  end subroutine aggregate_greedy
