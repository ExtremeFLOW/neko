275template < const int LX >
276int tune(void *w, void *u, void *dx, void *dy, void *dz,
277 void *dxt, void *dyt, void *dzt, void *h1,
278 void *g11, void *g22, void *g33, void *g12,
279 void *g13, void *g23, int *nelv, int *lx) {
280 hipEvent_t start,stop;
281 float time1,time2;
282 int retval;
283
284 const dim3 nthrds_1d(1024, 1, 1);
285 const dim3 nblcks_1d((*nelv), 1, 1);
286 const dim3 nthrds_kstep((*lx), (*lx), 1);
287 const dim3 nblcks_kstep((*nelv), 1, 1);
288
289 char *env_value = NULL;
290 char neko_log_buf[80];
291
292 env_value=getenv("NEKO_AUTOTUNE");
293
294 sprintf(neko_log_buf, "Autotune Ax helm (lx: %d)", *lx);
295 log_section(neko_log_buf);
296
297 if(env_value) {
298 if( !strcmp(env_value,"1D") ) {
299 CASE_1D(LX);
300 sprintf(neko_log_buf,"Set by env : 1 (1D)");
301 log_message(neko_log_buf);
302 log_end_section();
303 return 1;
304 } else if( !strcmp(env_value,"KSTEP") ) {
305 CASE_KSTEP(LX);
306 sprintf(neko_log_buf,"Set by env : 2 (KSTEP)");
307 log_message(neko_log_buf);
308 log_end_section();
309 return 2;
310 } else {
311 sprintf(neko_log_buf, "Invalid value set for NEKO_AUTOTUNE");
312 log_error(neko_log_buf);
313 }
314 }
315
316 HIP_CHECK(hipEventCreate(&start));
317 HIP_CHECK(hipEventCreate(&stop));
318
319 HIP_CHECK(hipEventRecord(start,0));
320
321 for(int i = 0; i < 100; i++) {
322 CASE_1D(LX);
323 }
324
325 HIP_CHECK(hipEventRecord(stop,0));
326 HIP_CHECK(hipEventSynchronize(stop));
327 HIP_CHECK(hipEventElapsedTime(&time1, start, stop));
328
329 HIP_CHECK(hipEventRecord(start,0));
330
331 for(int i = 0; i < 100; i++) {
332 CASE_KSTEP(LX);
333 }
334
335 HIP_CHECK(hipEventRecord(stop,0));
336 HIP_CHECK(hipEventSynchronize(stop));
337 HIP_CHECK(hipEventElapsedTime(&time2, start, stop));
338
339 if(time1 < time2) {
340 retval = 1;
341 } else {
342 retval = 2;
343 }
344
345 sprintf(neko_log_buf, "Chose : %d (%s)", retval,
346 (retval > 1 ? "KSTEP" : "1D"));
347 log_message(neko_log_buf);
348 log_end_section();
349 return retval;
350}
