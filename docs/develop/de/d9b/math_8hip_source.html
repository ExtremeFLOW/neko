668 /*
669 * Reduction buffer
670 */
671 int red_s = 0;
672 real * bufred = NULL;
673 real * bufred_d = NULL;
674
675 void hip_redbuf_check_alloc(int nb) {
676 if ( nb >= red_s) {
677 red_s = nb+1;
678 if (bufred != NULL) {
679 HIP_CHECK(hipHostFree(bufred));
680 HIP_CHECK(hipFree(bufred_d));
681 }
682 HIP_CHECK(hipHostMalloc(&bufred,red_s*sizeof(real),hipHostMallocDefault));
683 HIP_CHECK(hipMalloc(&bufred_d, red_s*sizeof(real)));
684 }
685 }

690 void hip_global_reduce_add(real * bufred, void * bufred_d, int n,
691 const hipStream_t stream) {
692 #ifdef HAVE_RCCL
693 device_nccl_allreduce(bufred_d, bufred_d, n, sizeof(real),
694 DEVICE_NCCL_SUM, stream);
695 HIP_CHECK(hipMemcpyAsync(bufred, bufred_d, n*sizeof(real),
696 hipMemcpyDeviceToHost, stream));
697 HIP_CHECK(hipStreamSynchronize(stream));
698 #elif HAVE_DEVICE_MPI
699 HIP_CHECK(hipStreamSynchronize(stream));
700 device_mpi_allreduce(bufred_d, bufred, n, sizeof(real), DEVICE_MPI_SUM);
701 #else
702 HIP_CHECK(hipMemcpyAsync(bufred, bufred_d, n*sizeof(real),
703 hipMemcpyDeviceToHost, stream));
704 HIP_CHECK(hipStreamSynchronize(stream));
705 #endif
706 }
