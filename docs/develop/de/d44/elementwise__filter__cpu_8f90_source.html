55  subroutine build_1d_cpu(fh, fht, trnsfr, nx, filter_type)
56  integer, intent(in) :: nx
57  real(kind=rp), intent(inout) :: fh(nx, nx), fht(nx, nx)
58  real(kind=rp), intent(in) :: trnsfr(nx)
59  real(kind=rp) :: diag(nx, nx), rmult(nx), lj(nx), zpts(nx)
60  type(matrix_t) :: phi, pht
61  integer :: n, i, j, k
62  real(kind=rp) :: z
63  character(len=*), intent(in) :: filter_type
64 
65  call phi%init(nx, nx)
66  call pht%init(nx, nx)
67 
68  call zwgll(zpts, rmult, nx)
69 
70  n = nx-1
71  do j = 1, nx
72  z = zpts(j)
73  call legendre_poly(lj, z, n)
74  select case (filter_type)
75  case("Boyd")
76  pht%x(1,j) = lj(1)
77  pht%x(2,j) = lj(2)
78  do k=3,nx
79  pht%x(k,j) = lj(k)-lj(k-2)
80  end do
81  case("nonBoyd")
82  pht%x(:,j) = lj
83  end select
84  end do
85 
86  call trsp(phi%x, nx, pht%x, nx)
87  pht%x = phi%x
88 
89  call pht%inverse(0) ! "0" for cpu implementation
90 
91  diag = 0.0_rp
92 
93  do i=1,nx
94  diag(i,i) = trnsfr(i)
95  end do
96 
97  call mxm (diag, nx, pht%x, nx, fh, nx) ! -1
98  call mxm (phi%x, nx, fh, nx, pht%x, nx) ! V D V
99 
100  call copy (fh, pht%x, nx*nx)
101  call trsp (fht, nx, fh, nx)
102 
103  end subroutine build_1d_cpu
104 
105 end module elementwise_filter_cpu
