160 subroutine fluid_scheme_compressible_euler_init(this, msh, lx, params, user, &
161 chkp)
162 class(fluid_scheme_compressible_euler_t), target, intent(inout) :: this
163 type(mesh_t), target, intent(inout) :: msh
164 integer, intent(in) :: lx
165 type(json_file), target, intent(inout) :: params
166 type(user_t), target, intent(in) :: user
167 type(chkp_t), target, intent(inout) :: chkp
168 character(len=12), parameter :: scheme = 'compressible'
169 integer :: rk_order
170
171 call this%free()
172
173 ! Initialize base class
174 call this%scheme_init(msh, lx, params, scheme, user)
175
176 call euler_rhs_factory(this%euler_rhs)
177
178 associate(xh_lx => this%Xh%lx, xh_ly => this%Xh%ly, xh_lz => this%Xh%lz, &
179 dm_xh => this%dm_Xh, nelv => this%msh%nelv)
180
181 call this%drho%init(dm_xh, 'drho')
182 call this%dm_x%init(dm_xh, 'dm_x')
183 call this%dm_y%init(dm_xh, 'dm_y')
184 call this%dm_z%init(dm_xh, 'dm_z')
185 call this%dE%init(dm_xh, 'dE')
186 call this%h%init(dm_xh, 'h')
187
188 end associate
189
190 if (neko_bcknd_device .eq. 1) then
191 associate(p => this%p, rho => this%rho, &
192 u => this%u, v => this%v, w => this%w, &
193 m_x => this%m_x, m_y => this%m_y, m_z => this%m_z, &
194 effective_visc => this%effective_visc)
195 call device_memcpy(p%x, p%x_d, p%dof%size(), &
196 host_to_device, sync = .false.)
197 call device_memcpy(rho%x, rho%x_d, rho%dof%size(), &
198 host_to_device, sync = .false.)
199 call device_memcpy(u%x, u%x_d, u%dof%size(), &
200 host_to_device, sync = .false.)
201 call device_memcpy(v%x, v%x_d, v%dof%size(), &
202 host_to_device, sync = .false.)
203 call device_memcpy(w%x, w%x_d, w%dof%size(), &
204 host_to_device, sync = .false.)
205 call device_memcpy(m_x%x, m_x%x_d, m_x%dof%size(), &
206 host_to_device, sync = .false.)
207 call device_memcpy(m_y%x, m_y%x_d, m_y%dof%size(), &
208 host_to_device, sync = .false.)
209 call device_memcpy(m_z%x, m_z%x_d, m_z%dof%size(), &
210 host_to_device, sync = .false.)
211 call device_memcpy(effective_visc%x, effective_visc%x_d, effective_visc%dof%size(), &
212 host_to_device, sync = .false.)
213 end associate
214 end if
215
216 ! Initialize the diffusion operator
217 call ax_helm_factory(this%Ax, full_formulation = .false.)
218
219 ! Compute h
220 call this%compute_h()
221
222 ! Initialize regularization
223 call this%setup_regularization(params)
224
225 ! Initialize Runge-Kutta scheme
226 call json_get_or_default(params, 'case.numerics.time_order', rk_order, 4)
227 call this%rk_scheme%init(rk_order)
228
229 ! Set up boundary conditions
230 call this%setup_bcs(user, params)

232 end subroutine fluid_scheme_compressible_euler_init

270 subroutine fluid_scheme_compressible_euler_step(this, time, dt_controller)
271 use entropy_viscosity, only: entropy_viscosity_t
272 class(fluid_scheme_compressible_euler_t), target, intent(inout) :: this
273 type(time_state_t), intent(in) :: time
274 type(time_step_controller_t), intent(in) :: dt_controller
275 type(field_t), pointer :: temp
276 integer :: temp_indices(1)
277 ! number of degrees of freedom
278 integer :: n
279 integer :: i
280 class(bc_t), pointer :: b
281
282 n = this%dm_Xh%size()
283 call neko_scratch_registry%request_field(temp, temp_indices(1), .false.)
284 b => null()
285
286 call profiler_start_region('Fluid compressible', 1)
287 associate(u => this%u, v => this%v, w => this%w, p => this%p, &
288 m_x=> this%m_x, m_y => this%m_y, m_z => this%m_z, &
289 xh => this%Xh, msh => this%msh, ax => this%Ax, &
290 c_xh => this%c_Xh, dm_xh => this%dm_Xh, gs_xh => this%gs_Xh, &
291 e => this%E, rho => this%rho, mu => this%mu, &
292 f_x => this%f_x, f_y => this%f_y, f_z => this%f_z, &
293 drho => this%drho, dm_x => this%dm_x, dm_y => this%dm_y, &
294 dm_z => this%dm_z, de => this%dE, &
295 euler_rhs => this%euler_rhs, h => this%h, &
296 t => time%t, tstep => time%tstep, dt => time%dt, &
297 c_avisc_low => this%c_avisc_low, rk_scheme => this%rk_scheme)
298
299 ! Compute artificial viscosity
300 call this%regularization%compute(time, time%tstep, time%dt)
301
302 ! Execute RHS step with effective viscosity field
303 call euler_rhs%step(rho, m_x, m_y, m_z, e, &
304 p, u, v, w, ax, &
305 c_xh, gs_xh, h, this%effective_visc, &
306 rk_scheme, dt)
307
309 call this%bcs_density%apply(rho, time)
310
312 ! Update u, v, w
313 call field_copy(u, m_x, n)
314 call field_invcol2(u, rho, n)
315 call field_copy(v, m_y, n)
316 call field_invcol2(v, rho, n)
317 call field_copy(w, m_z, n)
318 call field_invcol2(w, rho, n)
319
321 call this%bcs_vel%apply_vector(u%x, v%x, w%x, &
322 dm_xh%size(), time, strong = .true.)
323
324
325 call field_copy(m_x, u, n)
326 call field_col2(m_x, rho, n)
327 call field_copy(m_y, v, n)
328 call field_col2(m_y, rho, n)
329 call field_copy(m_z, w, n)
330 call field_col2(m_z, rho, n)
331
333 call field_col3(temp, u, u, n)
334 call field_addcol3(temp, v, v, n)
335 call field_addcol3(temp, w, w, n)
336 call field_col2(temp, rho, n)
337 call field_cmult(temp, 0.5_rp, n)
338 call field_copy(p, e, n)
339 call field_sub2(p, temp, n)
340 call field_cmult(p, this%gamma - 1.0_rp, n)
341
343 call this%bcs_prs%apply(p, time)
344 ! Ensure pressure is positive
345 call field_cpwmax2(p, 1.0e-12_rp, n)
346 ! E = p / (gamma - 1) + 0.5 * rho * (u^2 + v^2 + w^2)
347 call field_copy(e, p, n)
348 call field_cmult(e, 1.0_rp / (this%gamma - 1.0_rp), n)
349 ! temp = 0.5 * rho * (u^2 + v^2 + w^2)
350 call field_add2(e, temp, n)
351
353 call this%compute_entropy()
354
356 if (allocated(this%regularization)) then
357 select type (reg => this%regularization)
358 type is (entropy_viscosity_t)
359 call reg%update_lag()
360 end select
361 end if
362
364 call this%compute_max_wave_speed()
365
366 do i = 1, this%bcs_vel%size()
367 b => this%bcs_vel%get(i)
368 b%updated = .false.
369 end do
370
371 do i = 1, this%bcs_prs%size()
372 b => this%bcs_prs%get(i)
373 b%updated = .false.
374 end do
375
376 do i = 1, this%bcs_density%size()
377 b => this%bcs_density%get(i)
378 b%updated = .false.
379 end do
380 nullify(b)
381
382 end associate
383 call profiler_end_region('Fluid compressible', 1)
384
385 call neko_scratch_registry%relinquish_field(temp_indices)
386
387 end subroutine fluid_scheme_compressible_euler_step

393 subroutine fluid_scheme_compressible_euler_setup_bcs(this, user, params)
394 class(fluid_scheme_compressible_euler_t), target, intent(inout) :: this
395 type(user_t), target, intent(in) :: user
396 type(json_file), intent(inout) :: params
397 integer :: i, n_bcs, zone_index, j, zone_size, global_zone_size, ierr
398 class(bc_t), pointer :: bc_i
399 type(json_core) :: core
400 type(json_value), pointer :: bc_object
401 type(json_file) :: bc_subdict
402 logical :: found
403 integer, allocatable :: zone_indices(:)
404 character(len=LOG_SIZE) :: log_buf
405
406 ! Process boundary conditions
407 if (params%valid_path('case.fluid.boundary_conditions')) then
408 call params%info('case.fluid.boundary_conditions', n_children = n_bcs)
409 call params%get_core(core)
410 call params%get('case.fluid.boundary_conditions', bc_object, found)
411
412 !
413 ! Velocity bcs
414 !
415 call this%bcs_vel%init(n_bcs)
416
417 do i = 1, n_bcs
418 ! Extract BC configuration
419 call json_extract_item(core, bc_object, i, bc_subdict)
420 call json_get(bc_subdict, "zone_indices", zone_indices)
421
422 ! Validate zones
423 do j = 1, size(zone_indices)
424 zone_size = this%msh%labeled_zones(zone_indices(j))%size
425 call mpi_allreduce(zone_size, global_zone_size, 1, &
426 mpi_integer, mpi_max, neko_comm, ierr)
427
428 if (global_zone_size .eq. 0) then
429 write(log_buf,'(A,I0,A)') "Error: Zone ", zone_indices(j), &
430 " has zero size"
431 call neko_error(log_buf)
432 end if
433 end do
434
435 ! Create BC
436 bc_i => null()
437 call velocity_bc_factory(bc_i, this, bc_subdict, this%c_Xh, user)
438
439 ! Add to appropriate lists
440 if (associated(bc_i)) then
441 call this%bcs_vel%append(bc_i)
442 end if
443 end do
444
445 !
446 ! Pressure bcs
447 !
448 call this%bcs_prs%init(n_bcs)
449
450 do i = 1, n_bcs
451 ! Create a new json containing just the subdict for this bc
452 call json_extract_item(core, bc_object, i, bc_subdict)
453 bc_i => null()
454 call pressure_bc_factory(bc_i, this, bc_subdict, this%c_Xh, user)
455
456 ! Not all bcs require an allocation for pressure in particular,
457 ! so we check.
458 if (associated(bc_i)) then
459 call this%bcs_prs%append(bc_i)
460 end if
461 end do
462
463 !
464 ! Density bcs
465 !
466 call this%bcs_density%init(n_bcs)
467
468 do i = 1, n_bcs
469 ! Create a new json containing just the subdict for this bc
470 call json_extract_item(core, bc_object, i, bc_subdict)
471 bc_i => null()
472 call density_bc_factory(bc_i, this, bc_subdict, this%c_Xh, user)
473
474 ! Not all bcs require an allocation for pressure in particular,
475 ! so we check.
476 if (associated(bc_i)) then
477 call this%bcs_density%append(bc_i)
478 end if
479 end do
480 else
481 ! Check that there are no labeled zones, i.e. all are periodic.
482 do i = 1, size(this%msh%labeled_zones)
483 if (this%msh%labeled_zones(i)%size .gt. 0) then
484 call neko_error("No boundary_conditions entry in the case file!")
485 end if
486 end do
487
488 ! For a pure periodic case, we still need to initilise the bc lists
489 ! to a zero size to avoid issues with apply() in step()
490 call this%bcs_prs%init()
491 call this%bcs_vel%init()
492 call this%bcs_density%init()
493
494 end if
495 end subroutine fluid_scheme_compressible_euler_setup_bcs

501 subroutine compute_h(this)
502 class(fluid_scheme_compressible_euler_t), intent(inout) :: this
503 integer :: e, i, j, k
504 integer :: im, ip, jm, jp, km, kp
505 real(kind=rp) :: di, dj, dk, ndim_inv
506 integer :: lx_half, ly_half, lz_half
507
508 lx_half = this%c_Xh%Xh%lx / 2
509 ly_half = this%c_Xh%Xh%ly / 2
510 lz_half = this%c_Xh%Xh%lz / 2
511
512 do e = 1, this%c_Xh%msh%nelv
513 do k = 1, this%c_Xh%Xh%lz
514 km = max(1, k-1)
515 kp = min(this%c_Xh%Xh%lz, k+1)
516
517 do j = 1, this%c_Xh%Xh%ly
518 jm = max(1, j-1)
519 jp = min(this%c_Xh%Xh%ly, j+1)
520
521 do i = 1, this%c_Xh%Xh%lx
522 im = max(1, i-1)
523 ip = min(this%c_Xh%Xh%lx, i+1)
524
525 di = (this%c_Xh%dof%x(ip, j, k, e) - &
526 this%c_Xh%dof%x(im, j, k, e))**2 &
527 + (this%c_Xh%dof%y(ip, j, k, e) - &
528 this%c_Xh%dof%y(im, j, k, e))**2 &
529 + (this%c_Xh%dof%z(ip, j, k, e) - &
530 this%c_Xh%dof%z(im, j, k, e))**2
531
532 dj = (this%c_Xh%dof%x(i, jp, k, e) - &
533 this%c_Xh%dof%x(i, jm, k, e))**2 &
534 + (this%c_Xh%dof%y(i, jp, k, e) - &
535 this%c_Xh%dof%y(i, jm, k, e))**2 &
536 + (this%c_Xh%dof%z(i, jp, k, e) - &
537 this%c_Xh%dof%z(i, jm, k, e))**2
538
539 dk = (this%c_Xh%dof%x(i, j, kp, e) - &
540 this%c_Xh%dof%x(i, j, km, e))**2 &
541 + (this%c_Xh%dof%y(i, j, kp, e) - &
542 this%c_Xh%dof%y(i, j, km, e))**2 &
543 + (this%c_Xh%dof%z(i, j, kp, e) - &
544 this%c_Xh%dof%z(i, j, km, e))**2
545
546 di = sqrt(di) / (ip - im)
547 dj = sqrt(dj) / (jp - jm)
548 dk = sqrt(dk) / (kp - km)
549 this%h%x(i,j,k,e) = (di * dj * dk)**(1.0_rp / 3.0_rp)
550
551 end do
552 end do
553 end do
554 end do
555
556 if (neko_bcknd_device .eq. 1) then
557 call device_memcpy(this%h%x, this%h%x_d, this%h%dof%size(),&
558 host_to_device, sync = .false.)
559 call this%gs_Xh%op(this%h, gs_op_add)
560 call device_col2(this%h%x_d, this%c_Xh%mult_d, this%h%dof%size())
561 else
562 call this%gs_Xh%op(this%h, gs_op_add)
563 call col2(this%h%x, this%c_Xh%mult, this%h%dof%size())
564 end if
565
566 end subroutine compute_h

577 subroutine setup_regularization(this, params)
578 use entropy_viscosity, only: entropy_viscosity_t, entropy_viscosity_set_fields
579 class(fluid_scheme_compressible_euler_t), target, intent(inout) :: this
580 type(json_file), intent(inout) :: params
581 type(json_file) :: reg_json
582 type(json_core) :: json_core_inst
583 type(json_value), pointer :: reg_params
584 character(len=:), allocatable :: buffer
585 real(kind=rp) :: c_avisc_entropy_val
586 character(len=:), allocatable :: regularization_type
587
588 call json_get_or_default(params, 'case.numerics.c_avisc_low', &
589 this%c_avisc_low, 0.5_rp)
590 call json_get_or_default(params, 'case.numerics.c_avisc_entropy', &
591 c_avisc_entropy_val, 1.0_rp)
592
593 call json_core_inst%initialize()
594 call json_core_inst%create_object(reg_params, '')
595 call json_core_inst%add(reg_params, 'c_avisc_entropy', c_avisc_entropy_val)
596 call json_core_inst%add(reg_params, 'c_avisc_low', this%c_avisc_low)
597 call json_core_inst%print_to_string(reg_params, buffer)
598 call json_core_inst%destroy(reg_params)
599
600 call reg_json%initialize()
601 call reg_json%load_from_string(buffer)
602
603 regularization_type = 'entropy_viscosity'
604
605 call regularization_factory(this%regularization, regularization_type, reg_json, &
606 this%c_Xh, this%dm_Xh, this%effective_visc)
607
608 select type (reg => this%regularization)
