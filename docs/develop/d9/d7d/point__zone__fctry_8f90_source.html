56  module subroutine point_zone_factory(object, json, dof)
57  class(point_zone_t), allocatable, intent(inout) :: object
58  type(json_file), intent(inout) :: json
59  type(dofmap_t), intent(inout), optional :: dof
60  character(len=:), allocatable :: type_name
61  character(len=:), allocatable :: type_string
62 
63  call json_get(json, "geometry", type_name)
64 
65  if (trim(type_name) .eq. "box") then
66  allocate(box_point_zone_t::object)
67  else if (trim(type_name) .eq. "sphere") then
68  allocate(sphere_point_zone_t::object)
69  else if (trim(type_name) .eq. "cylinder") then
70  allocate(cylinder_point_zone_t::object)
71  else
72  type_string = concat_string_array(pointz_known_types, &
73  new_line('A') // "- ", .true.)
74  call neko_error("Unknown point zone type: " &
75  // trim(type_name) // ". Known types are: " &
76  // type_string)
77  end if
78 
79  if (present(dof)) then
80  call object%init(json, dof%size())
81  call object%map(dof)
82  call object%finalize()
83  else
84  ! This is for initializing zones inside a combine point zone,
85  ! as they don't need to be mapped
86  call object%init(json, 1)
87  call object%finalize()
88  end if
89 
90 
91  end subroutine point_zone_factory
