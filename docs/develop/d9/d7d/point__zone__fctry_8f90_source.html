56 module subroutine point_zone_factory(object, json, dof)
57 class(point_zone_t), allocatable, intent(inout) :: object
58 type(json_file), intent(inout) :: json
59 type(dofmap_t), intent(inout), optional :: dof
60 character(len=:), allocatable :: type_name
61 character(len=:), allocatable :: type_string
62
63 call json_get(json, "geometry", type_name)
64
65 call point_zone_allocator(object, type_name)
66
67 if (present(dof)) then
68 call object%init(json, dof%size())
69 call object%map(dof)
70 call object%finalize()
71 else
72 ! This is for initializing zones inside a combine point zone,
73 ! as they don't need to be mapped
74 call object%init(json, 1)
75 call object%finalize()
76 end if
77 end subroutine point_zone_factory

79 module subroutine point_zone_allocator(object, type_name)
80 class(point_zone_t), allocatable, intent(inout) :: object
81 character(len=:), allocatable, intent(in) :: type_name
82 integer :: i
83
84 if (allocated(object)) deallocate(object)
85
86 select case (trim(type_name))
87 case ('box')
88 allocate(box_point_zone_t::object)
89 case ('sphere')
90 allocate(sphere_point_zone_t::object)
91 case ('cylinder')
92 allocate(cylinder_point_zone_t::object)
93 case default
94 do i = 1, point_zone_registry_size
95 if (trim(type_name) .eq. trim(point_zone_registry(i)%type_name)) then
96 call point_zone_registry(i)%allocator(object)
97 return
98 end if
99 end do
100 call neko_error("Unknown point zone type: " // trim(type_name))
101 end select
102 end subroutine point_zone_allocator

109 module subroutine register_point_zone(type_name, allocator)
110 character(len=*), intent(in) :: type_name
111 procedure(point_zone_allocate), pointer, intent(in) :: allocator
112 type(allocator_entry), allocatable :: temp(:)
113 integer :: i
114
115 do i = 1, size(pointz_known_types)
116 if (trim(type_name) .eq. trim(pointz_known_types(i))) then
117 call neko_type_registration_error("point zone", type_name, .true.)
118 end if
119 end do
120
121 do i = 1, point_zone_registry_size
122 if (trim(type_name) .eq. trim(point_zone_registry(i)%type_name)) then
123 call neko_type_registration_error("point zone", type_name, .false.)
124 end if
125 end do
126
127 ! Expand registry
128 if (point_zone_registry_size .eq. 0) then
129 allocate(point_zone_registry(1))
130 else
131 allocate(temp(point_zone_registry_size + 1))
132 temp(1:point_zone_registry_size) = point_zone_registry
133 call move_alloc(temp, point_zone_registry)
134 end if
135
136 point_zone_registry_size = point_zone_registry_size + 1
137 point_zone_registry(point_zone_registry_size)%type_name = type_name
138 point_zone_registry(point_zone_registry_size)%allocator => allocator
139
140 end subroutine register_point_zone
