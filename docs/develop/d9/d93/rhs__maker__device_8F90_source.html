270 type(field_t), intent(inout) :: u,v, w
271 type(field_t), intent(inout) :: uu, vv, ww

273 real(kind=rp), dimension(3), intent(in) :: ab

279 wwlag%lf(1)%x_d, wwlag%lf(2)%x_d, ab(1), ab(2), ab(3), nab, &
280 uu%dof%size())
281#elif HAVE_CUDA
