95 subroutine fluid_stats_simcomp_init_from_json(this, json, case)
96 class(fluid_stats_simcomp_t), target, intent(inout) :: this
97 type(json_file), intent(inout) :: json
98 class(case_t), intent(inout), target :: case
99 character(len=:), allocatable :: filename
100 character(len=20), allocatable :: fields(:)
101 character(len=:), allocatable :: hom_dir
102 character(len=:), allocatable :: stat_set
103 real(kind=rp) :: start_time
104 type(field_t), pointer :: u, v, w, p
105 type(coef_t), pointer :: coef
106
107 call this%init_base(json, case)
108 call json_get_or_default(json, 'avg_direction', &
109 hom_dir, 'none')
110 call json_get_or_default(json, 'start_time', &
111 start_time, 0.0_rp)
112 call json_get_or_default(json, 'set_of_stats', &
113 stat_set, 'full')
114
115
116 u => neko_registry%get_field("u")
117 v => neko_registry%get_field("v")
118 w => neko_registry%get_field("w")

121
122 if (json%valid_path("output_filename")) then
123 call json_get(json, "output_filename", filename)
124 call fluid_stats_simcomp_init_from_components(this, u, v, w, p, coef, &
125 start_time, hom_dir, stat_set,filename)

127 call fluid_stats_simcomp_init_from_components(this, u, v, w, p, coef, &

194 subroutine fluid_stats_simcomp_restart(this, time)
195 class(fluid_stats_simcomp_t), intent(inout) :: this
196 type(time_state_t), intent(in) :: time
197 character(len=NEKO_FNAME_LEN) :: fname
198 character(len=5) :: prefix,suffix
199 integer :: last_slash_pos
200 real(kind=rp) :: t
201 t = time%t
202 if (t .gt. this%time) this%time = t
203 if (this%default_fname) then

205 write (prefix, '(I5)') this%stats_output%file_%get_counter()

207 last_slash_pos = &

216 end if

224 subroutine fluid_stats_simcomp_compute(this, time)
225 class(fluid_stats_simcomp_t), intent(inout) :: this
226 type(time_state_t), intent(in) :: time
227 real(kind=rp) :: delta_t, t
228 real(kind=rp) :: sample_start_time, sample_time
229 character(len=LOG_SIZE) :: log_buf
230 integer :: ierr
231
232 if (time%start_time .gt. this%start_time) then
233 write(log_buf, '(A)') 'Simulation start time is later than the ' &
234 // 'fluid stats start time.'
235 call neko_log%warning(log_buf)
236 write(log_buf, '(A,E15.7)') 'Simulation start time:', time%start_time
237 call neko_log%warning(log_buf)
238 write(log_buf, '(A,E15.7)') 'Fluid stats start time:', this%start_time
239 call neko_log%warning(log_buf)
240 write(log_buf, '(A)') 'Assigning the statistics start time to ' &
241 // 'the simulation start time.'
242 call neko_log%warning(log_buf)
243 this%start_time = time%start_time
244 this%time = time%start_time
245 end if
246
247 t = time%t
248
249 if (t .ge. this%start_time) then
250 delta_t = t - this%time !This is only a real number
251
252 call mpi_barrier(neko_comm, ierr)
253
254 sample_start_time = mpi_wtime()
255
256 call this%stats%update(delta_t)
257 call mpi_barrier(neko_comm, ierr)
258 this%time = t
259
260 sample_time = mpi_wtime() - sample_start_time
261
262 call neko_log%section('Fluid stats')
263 write(log_buf, '(A,E15.7)') 'Sampling at time:', t
264 call neko_log%message(log_buf)
265 write(log_buf, '(A33,E15.7)') 'Simulationtime since last sample:', &
266 delta_t
267 call neko_log%message(log_buf)
268 write(log_buf, '(A,E15.7)') 'Sampling time (s):', sample_time
269 call neko_log%message(log_buf)
270 call neko_log%end_section()
271 end if
272
273 end subroutine fluid_stats_simcomp_compute
