177 type(mesh_t), intent(in) :: msh

196 coef%G11_d, coef%G22_d, coef%G33_d, &
197 coef%G12_d, coef%G13_d, coef%G23_d, &
198 msh%nelv, xh%lx)

202 coef%G11_d, coef%G22_d, coef%G33_d, &
203 coef%G12_d, coef%G13_d, coef%G23_d, &
204 msh%nelv, xh%lx)
205#endif
206
207 if (coef%ifh2) then
208 call device_addcol4(w_d ,coef%h2_d, coef%B_d, u_d, coef%dof%size())
209 end if
