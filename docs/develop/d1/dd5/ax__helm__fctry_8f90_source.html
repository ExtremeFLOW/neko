53 module subroutine ax_helm_factory(object, full_formulation)
54 class(ax_t), allocatable, intent(inout) :: object
55 logical, intent(in) :: full_formulation
56
57 if (allocated(object)) then
58 deallocate(object)
59 end if
60
61 if (full_formulation) then
62 if (neko_bcknd_sx .eq. 1 .or. neko_bcknd_xsmm .eq. 1) then
63 call neko_error("Full stress formulation is only available &
64 &on the CPU and device")
65 else if (neko_bcknd_device .eq. 1) then
66 allocate(ax_helm_full_device_t::object)
67 else
68 allocate(ax_helm_full_cpu_t::object)
69 end if
70 else
71 if (neko_bcknd_sx .eq. 1) then
72 allocate(ax_helm_sx_t::object)
73 else if (neko_bcknd_xsmm .eq. 1) then
74 allocate(ax_helm_xsmm_t::object)
75 else if (neko_bcknd_device .eq. 1) then
76 allocate(ax_helm_device_t::object)
77 else
78 allocate(ax_helm_cpu_t::object)
79 end if
80 end if
