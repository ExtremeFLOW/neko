46 void device_mpi_allreduce(void *buf_d, void *buf, int count, int nbytes, int op) {
47 
48  if (nbytes == sizeof(float)) {
49  if (op == DEVICE_MPI_SUM)
50  MPI_Allreduce(buf_d, buf, count, MPI_FLOAT, MPI_SUM, NEKO_COMM);
51  else if (op == DEVICE_MPI_MAX)
52  MPI_Allreduce(buf_d, buf, count, MPI_FLOAT, MPI_MAX, NEKO_COMM);
53  else {
54  fprintf(stderr, __FILE__ ": Invalid reduction op)\n");
55  exit(1);
56  }
57  }
58  else if (nbytes == sizeof(double)) {
59  if (op == DEVICE_MPI_SUM)
60  MPI_Allreduce(buf_d, buf, count, MPI_DOUBLE, MPI_SUM, NEKO_COMM);
61  else if (op == DEVICE_MPI_MAX)
62  MPI_Allreduce(buf_d, buf, count, MPI_DOUBLE, MPI_MAX, NEKO_COMM);
63  else {
64  fprintf(stderr, __FILE__ ": Invalid reduction op)\n");
65  exit(1);
66  }
67  }
68  else {
69  fprintf(stderr, __FILE__ ": Invalid data type)\n");
70  exit(1);
71  }
72 }
