78  elemental function smooth_step_kernel(x, edge0, edge1) result(res)
79  real(kind=rp), intent(in) :: x
80  real(kind=rp), intent(in) :: edge0, edge1

82 
83  t = clamp_kernel((x - edge0) / (edge1 - edge0), 0.0_rp, 1.0_rp)
