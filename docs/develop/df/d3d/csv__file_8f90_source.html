305  function csv_file_count_lines(this) result(n)
306  class(csv_file_t), intent(in) :: this
307 
308  integer :: n
309  integer :: ierr, file_unit
310 
311  call this%check_exists()
312 
313  open(file = trim(this%fname), status = 'old', newunit = file_unit, &
314  iostat = ierr)
315  if (ierr .ne. 0) call neko_error("Error while opening " // trim(this%fname))
316  rewind(file_unit)
317 
318  n = 0
319 
320  ! Keep reading (ierr = 0) until we reach the end (ierr != 0)
321  do
322  read (file_unit, *, iostat = ierr)
323  if (ierr .ne. 0) exit
324  n = n + 1
325  end do
326  rewind(file_unit)
