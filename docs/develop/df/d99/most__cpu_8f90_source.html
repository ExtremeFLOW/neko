101 subroutine select_bc_operators(bc_type, bc_value, q, ts, ti, kappa, &
102 utau, z0h, hi, Pr)
103 character(len=*), intent(in) :: bc_type
104 real(kind=rp), intent(in) :: hi, ti, kappa, utau, z0h, bc_value, pr
105 real(kind=rp), intent(inout) :: q,ts
106 select case (bc_type)
107 case ("neumann")
108 ! ts not used
109 q = bc_value
110 f_ptr => f_neumann
111 dfdl_ptr => dfdl_neumann
112 case ("dirichlet")
113 ts = bc_value
114 q = (kappa/pr)*utau*(ts - ti)/log(hi/z0h)
115 f_ptr => f_dirichlet
116 dfdl_ptr => dfdl_dirichlet
117 case default
118 call neko_error("Invalid specified temperature b.c. type ('neumann' or 'dirichlet'?)")
119 end select
120 end subroutine select_bc_operators

123 subroutine compute_ri_b(bc_type, g_dot_n, hi, ti, ts, magu, kappa, q, Pr, Ri_b)
124 character(len=*), intent(in) :: bc_type
125 real(kind=rp), intent(in) :: hi, ti, ts, pr
126 real(kind=rp), intent(in) :: magu, kappa, g_dot_n
127 real(kind=rp), intent(inout) :: q, ri_b
128
129 select case (bc_type)
130 case ("neumann")
131 ri_b = - g_dot_n*hi / ti*q*pr / (magu**3*kappa**2)
132 case ("dirichlet")
133 ri_b = g_dot_n*hi/ti*(ti - ts)/magu**2
134 case default
135 call neko_error("Invalid specified temperature b.c. type ('neumann' or 'dirichlet'?)")
136 end select
137 end subroutine compute_ri_b

140 subroutine set_stability_regime(Ri_b, Ri_threshold)
141 real(kind=rp), intent(in) :: ri_b, ri_threshold
142
143 if (ri_b > ri_threshold) then
144 slaw_m_ptr => slaw_m_stable
145 slaw_h_ptr => slaw_h_stable
146 corr_m_ptr => corr_m_stable
147 corr_h_ptr => corr_h_stable
148 elseif (ri_b < -ri_threshold) then
149 slaw_m_ptr => slaw_m_convective
150 slaw_h_ptr => slaw_h_convective
151 corr_m_ptr => corr_m_convective
152 corr_h_ptr => corr_h_convective

154 slaw_m_ptr => slaw_m_neutral

161 subroutine most_compute_cpu(u, v, w, temp, ind_r, ind_s, ind_t, ind_e, &
162 n_x, n_y, n_z, h, tau_x, tau_y, tau_z, n_nodes, lx, nelv, &
163 kappa, mu, rho, g_vec, Pr, z0, z0h_in, bc_type, bc_value, tstep, &
164 Ri_b_diagn, L_ob_diagn, utau_diagn, magu_diagn, ti_diagn, ts_diagn,&
165 q_diagn, h_x_idx, h_y_idx, h_z_idx)
166 integer, intent(in) :: n_nodes, lx, nelv, tstep
167 real(kind=rp), dimension(lx, lx, lx, nelv), intent(in) :: u, v, w, temp
168 integer, intent(in), dimension(n_nodes) :: ind_r, ind_s, ind_t, ind_e
169 real(kind=rp), dimension(n_nodes), intent(in) :: n_x, n_y, n_z, h
170 real(kind=rp), intent(in) :: kappa, z0, z0h_in, bc_value, mu, rho, pr
171 real(kind=rp), dimension(3), intent(in) :: g_vec
172 real(kind=rp) :: g_dot_n
173 character(len=*), intent(in) :: bc_type
174 real(kind=rp), dimension(n_nodes), intent(inout) :: tau_x, tau_y, tau_z
175 real(kind=rp) :: ui, vi, wi, hi
176 real(kind=rp) :: normu, z0h
177 real(kind=rp) :: l_upper, l_lower, l_old
178 real(kind=rp) :: f, dfdl, fd_h, l_new, l_sign
179 integer :: i, count
180 integer, parameter :: max_count = 50
181 real(kind=rp), parameter :: tol = 0.001_rp
182 real(kind=rp), parameter :: nr_step = 0.001_rp
183 real(kind=rp), parameter :: ri_threshold = 0.0001_rp
184 character(len=LOG_SIZE) :: log_buf
185 real(kind=rp) :: utau, ri_b, l_ob, magu, q, ti, ts
186 real(kind=rp), dimension(n_nodes), intent(inout) :: ri_b_diagn, l_ob_diagn
187 real(kind=rp), dimension(n_nodes), intent(inout) :: utau_diagn, magu_diagn
188 real(kind=rp), dimension(n_nodes), intent(inout) :: ti_diagn, ts_diagn
189 real(kind=rp), dimension(n_nodes), intent(inout) :: q_diagn
190 integer, dimension(n_nodes), intent(in) :: h_x_idx
191 integer, dimension(n_nodes), intent(in) :: h_y_idx
192 integer, dimension(n_nodes), intent(in) :: h_z_idx
193
194 do i=1, n_nodes
195 ! Sample the variables
196 ui = u(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
197 vi = v(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
198 wi = w(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
199 ti = temp(ind_r(i), ind_s(i), ind_t(i), ind_e(i))
200 hi = h(i)
201
202 ! Project on horizontal directions
203 normu = ui * n_x(i) + vi * n_y(i) + wi * n_z(i)
204 ui = ui - normu * n_x(i)
205 vi = vi - normu * n_y(i)
206 wi = wi - normu * n_z(i)
207
208 ! Compute velocity magnitude
209 magu = sqrt(ui**2 + vi**2 + wi**2)
210 magu = max(magu, 1.0e-6_rp)
211 utau = magu*kappa / log(hi/z0)
212
213 ! Compute thermal roughness length from Zilitinkevich, 1995
214 if (z0h_in < 0) then
215 ! z0h_in is interpreted as -C_Zil (Zilitinkevich constant) for z0h
216 z0h = z0 * exp(z0h_in*sqrt((utau*z0)/(mu/rho)))
217 else
218 z0h = z0h_in
219 end if
220
221 ! Get q, Ri_b, f_ptr, dfdl_ptr based on bc_type
222 ! Maybe redundant, but needed to initialise Rib
223 call select_bc_operators(bc_type, bc_value, q, ts, ti, kappa, utau, z0h, hi, pr)
224
225 ! Compute g along the normal (generalisation for hills and similar)
226 g_dot_n = abs(g_vec(1)*n_x(i) + g_vec(2)*n_y(i) + g_vec(3)*n_z(i))
227
228 ! Compute Richardson and set stability accordingly
229 call compute_ri_b(bc_type, g_dot_n, hi, ti, ts, magu, kappa, q, pr, ri_b)
230 call set_stability_regime(ri_b, ri_threshold)
231
232 if (abs((ri_b)) <= ri_threshold) then
233 ! Neutral (L_ob undefined)
234 l_ob = 0.0_rp
235 else
236 ! Determine target regime sign
237 if ((ri_b) > 0.0_rp) then
238 l_ob = hi / max(ri_b, ri_threshold) ! Stable guess
239 l_sign = 1.0_rp
240 else
241 l_ob = hi / min(ri_b, -ri_threshold) ! Convective guess
242 l_sign = -1.0_rp
243 end if
244
245 l_old = 1.0e10_rp
246 count = 0
247
248 ! Find Obukhov length
249 do while ((abs(l_old - l_ob)/abs(l_ob) > tol) .and. (count < max_count))
250 ! Switch between stable and convective based on bulk Richardson (Ri_b)
251 l_old = l_ob
252 count = count + 1
253 fd_h = nr_step*l_ob
254 l_upper = l_ob + fd_h
255 l_lower = l_ob - fd_h
256
257 ! Compute L_ob based on stability and bc_type
258 f = f_ptr(ri_b, hi, z0, z0h, pr, l_ob, slaw_m_ptr, slaw_h_ptr)
259 dfdl = dfdl_ptr(l_upper, l_lower, hi, z0, z0h, pr, l_ob, &
260 slaw_m_ptr, slaw_h_ptr, fd_h)
261 if (abs(dfdl) < 1.0e-12_rp) call neko_error("Division by zero in dfdl")
262 l_new = l_ob - f/dfdl
263 ! Avoid regime crossing during Newton iter (otherwise crash)
264 if (l_new*l_sign <= 0.0_rp) then
265 ! "damp update" (stay on same side)
266 l_new = 0.5_rp * l_ob
267 end if
268 ! Bound L_ob
269 l_ob = sign(max(abs(l_new), 1.0e-8_rp), l_sign)
270 l_ob = sign(min(abs(l_ob), 1.0e8_rp), l_sign)
271 end do
272
273 if (abs(l_ob) > 5e5_rp .or. abs(l_ob) < 1e-6_rp) then
274 count = max_count
275 call neko_warning("Obukhov length did not converge (MOST wall model)")
276 end if
277
278 if (.not. associated(f_ptr) .or. .not. associated(dfdl_ptr)) then
279 call neko_error("Unassociated pointer for f or dfdl")
280 end if
281 end if
282
283 ! Based on stability and bc_type, compute utau/q
284 select case (bc_type)
285 case ("neumann")
286 ! Compute u* with the new Obukhov length
287 utau = kappa*magu/slaw_m_ptr(hi, l_ob, z0)
288 case ("dirichlet")
289 ! Compute u* with the new Obukhov length
290 utau = kappa*magu/slaw_m_ptr(hi, l_ob, z0)
291 ! and compute q from here
292 q = kappa/pr*utau*(ts - ti)/slaw_h_ptr(hi, l_ob, z0h)
293 case default
294 call neko_error("Invalid specified temperature b.c. type ('neumann' or 'dirichlet'?)")
295 end select
296
297 ! Distribute according to the velocity vector and bound magu to avoid 0 division
298 magu = max(magu, 1.0e-6_rp)
299 tau_x(i) = -rho * utau**2 * ui / magu
300 tau_y(i) = -rho * utau**2 * vi / magu
301 tau_z(i) = -rho * utau**2 * wi / magu
302
303 ri_b_diagn(i) = ri_b
304 l_ob_diagn(i) = l_ob
305 utau_diagn(i) = utau
306 magu_diagn(i) = magu
307 ti_diagn(i) = ti
308 ts_diagn(i) = temp(ind_r(i)-h_x_idx(i), ind_s(i)-h_y_idx(i), ind_t(i)-h_z_idx(i), ind_e(i))
309 q_diagn(i) = q
310 end do
311 end subroutine most_compute_cpu

333 function corr_m_stable(z, L_ob) result(corr)
334 real(kind=rp), intent(in) :: z, l_ob
335 real(kind=rp) :: corr
336 real(kind=rp) :: a, b, c, d
337 real(kind=rp) :: zeta
338 zeta = z/l_ob
339 ! Coefficients specific to Cheng & Brutsaert (2005)
340 a = 1.0_rp
341 b = 2.0_rp/3.0_rp
342 c = 5.0_rp
343 d = 0.35_rp
344 corr = - a*zeta - b*(zeta-c/d)*exp(-d*zeta) - b*c/d
345 end function corr_m_stable

381 function corr_m_convective(z, L_ob) result(corr)
382 real(kind=rp), intent(in) :: z, l_ob
383 real(kind=rp) :: xi, pi, zeta
384 real(kind=rp) :: corr
385
386 zeta = z/l_ob
