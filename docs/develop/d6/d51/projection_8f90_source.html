122 subroutine projection_init(this, n, L, activ_step, reorthogonalize_basis)
123 class(projection_t), target, intent(inout) :: this
124 integer, intent(in) :: n
125 integer, intent(in) :: L
126 integer, optional, intent(in) :: activ_step
127 logical, optional, intent(in) :: reorthogonalize_basis
128 integer :: i
129 integer(c_size_t) :: ptr_size
130 type(c_ptr) :: ptr
131 real(c_rp) :: dummy
132
133 call this%free()
134
135 this%L = l
136
137 if (present(activ_step)) then
138 this%activ_step = activ_step
139 else
140 this%activ_step = 5
141 end if
142
143 if (present(reorthogonalize_basis)) then
144 this%prj_reorthogonalize_basis = reorthogonalize_basis
145 end if
146
147 this%m = 0
148
149 ! Return if the space is 0, to avoid allocating zero sized
150 ! arrays, which are not supported for all backends
151 if (this%L .le. 0) then
152 return
153 end if
154
155 allocate(this%xx(n, this%L))
156 allocate(this%bb(n, this%L))
157 allocate(this%xbar(n))
158 call rzero(this%xbar, n)
159 do i = 1, this%L
160 call rzero(this%xx(1, i), n)
161 call rzero(this%bb(1, i), n)
162 end do
163 if (neko_bcknd_device .eq. 1) then
164
165 allocate(this%xx_d(this%L))
166 allocate(this%bb_d(this%L))
167
168 call device_map(this%xbar, this%xbar_d, n)
169 call device_alloc(this%alpha_d, int(c_sizeof(dummy)*this%L, c_size_t))
170
171 call device_rzero(this%xbar_d, n)
172 call device_rzero(this%alpha_d, this%L)
173
174 do i = 1, this%L
175 this%xx_d(i) = c_null_ptr
176 call device_map(this%xx(:, i), this%xx_d(i), n)
177 call device_rzero(this%xx_d(i), n)
178 this%bb_d(i) = c_null_ptr
179 call device_map(this%bb(:, i), this%bb_d(i), n)
180 call device_rzero(this%bb_d(i), n)
181 end do
182
183 ptr_size = c_sizeof(c_null_ptr) * this%L
184 call device_alloc(this%xx_d_d, ptr_size)
185 ptr = c_loc(this%xx_d)
186 call device_memcpy(ptr, this%xx_d_d, ptr_size, &
187 host_to_device, sync = .false.)
188 call device_alloc(this%bb_d_d, ptr_size)
189 ptr = c_loc(this%bb_d)
190 call device_memcpy(ptr, this%bb_d_d, ptr_size, &
191 host_to_device, sync = .false.)
192 end if
193
194
195 end subroutine projection_init

240 subroutine projection_pre_solving(this, b, tstep, coef, n, dt_controller, &
241 string, Ax, gs_h, bclst)
242 class(projection_t), intent(inout) :: this
243 integer, intent(inout) :: n
244 real(kind=rp), intent(inout), dimension(n) :: b
245 integer, intent(in) :: tstep
246 class(coef_t), intent(inout) :: coef
247 type(time_step_controller_t), intent(in) :: dt_controller
248 class(bc_list_t), optional, intent(inout) :: bclst
249 type(gs_t), optional, intent(inout) :: gs_h
250 class(ax_t), optional, intent(in) :: Ax
251 character(len=*), optional :: string
252
253 if (tstep .gt. this%activ_step .and. this%L .gt. 0) then
254 if (dt_controller%is_variable_dt) then
255 ! the time step at which dt is changed
256 if (dt_controller%dt_last_change .eq. 0) then
257 call this%clear(n)
258 else if (dt_controller%dt_last_change .gt. this%activ_step - 1) then
259 ! Re-orthogonalize basis if requested
260 call this%reortho_basis(ax, coef, gs_h, bclst, n)
261 ! activate projection some steps after dt is changed
262 ! note that dt_last_change start from 0
263 call this%project_on(b, coef, n)
264 if (present(string)) then
265 call this%log_info(string, tstep)
266 end if
267 end if
268 else
269 ! Re-orthogonalize basis if requested
270 call this%reortho_basis(ax, coef, gs_h, bclst, n)
271 call this%project_on(b, coef, n)
272 if (present(string)) then
273 call this%log_info(string, tstep)
274 end if
275 end if
276 end if
277
278 end subroutine projection_pre_solving

280 subroutine projection_post_solving(this, x, Ax, coef, bclst, gs_h, n, tstep, &
281 dt_controller)
282 class(projection_t), intent(inout) :: this
283 integer, intent(inout) :: n
284 class(ax_t), intent(inout) :: Ax
285 class(coef_t), intent(inout) :: coef
286 class(bc_list_t), intent(inout) :: bclst
287 type(gs_t), intent(inout) :: gs_h
288 real(kind=rp), intent(inout), dimension(n) :: x
289 integer, intent(in) :: tstep
290 type(time_step_controller_t), intent(in) :: dt_controller
291
292 if (tstep .gt. this%activ_step .and. this%L .gt. 0) then
293 if (.not.(dt_controller%is_variable_dt) .or. &
294 (dt_controller%dt_last_change .gt. this%activ_step - 1)) then
295 call this%project_back(x, ax, coef, bclst, gs_h, n)
296 end if
297 end if
298
299 end subroutine projection_post_solving

301 subroutine bcknd_project_on(this, b, coef, n)
302 class(projection_t), intent(inout) :: this
303 integer, intent(inout) :: n
304 class(coef_t), intent(inout) :: coef
305 real(kind=rp), intent(inout), dimension(n) :: b
306 call profiler_start_region('Project on', 16)
307 if (neko_bcknd_device .eq. 1) then
308 call device_project_on(this, b, coef, n)

311 end if

315 subroutine bcknd_project_back(this, x, Ax, coef, bclst, gs_h, n)
316 class(projection_t), intent(inout) :: this
317 integer, intent(inout) :: n
318 class(ax_t), intent(inout) :: Ax
319 class(coef_t), intent(inout) :: coef
320 class(bc_list_t), intent(inout) :: bclst
321 type(gs_t), intent(inout) :: gs_h
322 real(kind=rp), intent(inout), dimension(n) :: x
323 type(c_ptr) :: x_d
324
325 call profiler_start_region('Project back', 17)
326
327 if (neko_bcknd_device .eq. 1) then
328 x_d = device_get_ptr(x)
329 ! Restore desired solution
330 if (this%m .gt. 0) call device_add2(x_d, this%xbar_d, n)
331 if (this%m .eq. this%L) then
332 this%m = 1
333 else
334 this%m = min(this%m+1, this%L)
335 end if
336
337 call device_copy(this%xx_d(this%m), x_d, n) ! Update (X,B)
338
339 else
340 if (this%m .gt. 0) call add2(x, this%xbar, n) ! Restore desired solution
341 if (this%m .eq. this%L) then
342 this%m = 1
343 else
344 this%m = min(this%m+1, this%L)
345 end if
346
347 call copy(this%xx(1, this%m), x, n) ! Update (X,B)
348 end if
349
350 call ax%compute(this%bb(1, this%m), x, coef, coef%msh, coef%Xh)
351 call gs_h%gs_op_vector(this%bb(1, this%m), n, gs_op_add)
352 call bclst%apply_scalar(this%bb(1, this%m), n)
353
354 call proj_ortho(this, coef, n)
355 call profiler_end_region('Project back', 17)
356 end subroutine bcknd_project_back

359 class(projection_t), intent(inout) :: this
360 class(ax_t), intent(in) :: Ax
361 class(coef_t), intent(in) :: coef
362 type(gs_t), intent(inout) :: gs_h
363 type(bc_list_t), intent(inout) :: blst
364 integer, intent(in) :: n

367 if (.not. this%prj_reorthogonalize_basis) return

369 call profiler_start_region('Project reortho basis')

371 call device_reorthogonalize_basis(this, ax, coef, gs_h, blst, n)
372 else

378 subroutine cpu_reorthogonalize_basis(this, Ax, coef, gs_h, blst, n)
379 class(projection_t), intent(inout) :: this
380 class(ax_t), intent(in) :: Ax
381 class(coef_t), intent(in) :: coef
382 type(gs_t), intent(inout) :: gs_h
383 type(bc_list_t), intent(inout) :: blst
384 integer, intent(in) :: n
385 character(len=1000) :: msg
386
387 integer :: i, j
388 real(kind=rp) :: alpha, s, norm_fac
389 real(kind=rp) :: start_time, end_time, time
390
391 if (this%m .le. 0) return
392
393 associate(xx => this%xx, bb => this%bb)
394 start_time = mpi_wtime()
395
396 ! Recompute B = A_new * X using the new mesh metrics
397 do i = 1, this%m
398 call ax%compute(bb(1,i), xx(1,i), coef, coef%msh, coef%Xh)
399 call gs_h%gs_op_vector(bb(1,i), n, gs_op_add)
400 call blst%apply_scalar(bb(1,i), n)
401 end do
402
403 ! Modified Gram-Schmidt
404 do i = 1, this%m
405
406 ! Orthogonalize against previous vectors
407 do j = 1, i - 1
408 alpha = glsc3(xx(1,i), bb(1,j), coef%mult, n)
409 call add2s2(xx(1,i), xx(1,j), -alpha, n)
410 call add2s2(bb(1,i), bb(1,j), -alpha, n)
411 end do
412
413 s = glsc3(xx(1,i), bb(1,i), coef%mult, n)
414 norm_fac = 1.0_rp / sqrt(s)
415 call cmult(xx(1,i), norm_fac, n)
416 call cmult(bb(1,i), norm_fac, n)
417
418 end do
419 end_time = mpi_wtime()
420 time = end_time - start_time
421 write(msg, '(A, E15.7)') "Projection basis reorthogonalization (s): ", time
422 call neko_log%message(trim(msg))
423 end associate
424 end subroutine cpu_reorthogonalize_basis

426 subroutine device_reorthogonalize_basis(this, Ax, coef, gs_h, blst, n)
427 class(projection_t), intent(inout) :: this
428 class(ax_t), intent(in) :: Ax
429 class(coef_t), intent(in) :: coef
430 type(gs_t), intent(inout) :: gs_h
431 type(bc_list_t), intent(inout) :: blst
432 integer, intent(in) :: n
433 character(len=1000) :: msg
434
435 integer :: i, j
436 real(kind=rp) :: alpha, s, norm_fac
437 real(kind=rp) :: start_time, end_time, time
438
439 if (this%m .le. 0) return
440
441 associate(xx_d => this%xx_d, bb_d => this%bb_d)
442 start_time = mpi_wtime()
443
444 ! Recompute B = A_new * X using the new mesh metrics
445 do i = 1, this%m
446 call ax%compute(this%bb(1,i), this%xx(1,i), coef, coef%msh, coef%Xh)
447 call gs_h%gs_op_vector(this%bb(1,i), n, gs_op_add)
448 call blst%apply_scalar(this%bb(1,i), n)
449 end do
450
451 ! Modified Gram-Schmidt
452 do i = 1, this%m
453
454 ! Orthogonalize against previous vectors
455 do j = 1, i - 1
456 alpha = device_glsc3(xx_d(i), bb_d(j), coef%mult_d, n)
457 call device_add2s2(xx_d(i), xx_d(j), -alpha, n)
458 call device_add2s2(bb_d(i), bb_d(j), -alpha, n)
459 end do
460
461 s = device_glsc3(xx_d(i), bb_d(i), coef%mult_d, n)
462 norm_fac = 1.0_rp / sqrt(s)
463 call device_cmult(xx_d(i), norm_fac, n)
464 call device_cmult(bb_d(i), norm_fac, n)
465
466 end do
467 end_time = mpi_wtime()
468 time = end_time - start_time
469 write(msg, '(A, E15.7)') "Projection basis reorthogonalization (s): ", time
470 call neko_log%message(trim(msg))
471 end associate
472 end subroutine device_reorthogonalize_basis

475 subroutine cpu_project_on(this, b, coef, n)
476 class(projection_t), intent(inout) :: this
477 integer, intent(inout) :: n
478 class(coef_t), intent(inout) :: coef
479 real(kind=rp), intent(inout), dimension(n) :: b
480 integer :: i, j, k, l, ierr
481 real(kind=rp) :: work(this%L), alpha(this%L), s
482
483 associate(xbar => this%xbar, xx => this%xx, &
484 bb => this%bb)
485
486 if (this%m .le. 0) return
487
488 !First round of CGS
489 call rzero(alpha, this%m)
490 this%proj_res = sqrt(glsc3(b, b, coef%mult, n) / coef%volume)
491 this%proj_m = this%m
492 do i = 1, n, neko_blk_size
493 j = min(neko_blk_size, n-i+1)
494 do k = 1, this%m
495 s = 0.0_rp
496 do l = 0, (j-1)
497 s = s + xx(i+l, k) * coef%mult(i+l,1,1,1) * b(i+l)
498 end do
499 alpha(k) = alpha(k) + s
500 end do
501 end do
502
503 !First one outside loop to avoid zeroing xbar and bbar
504 call mpi_allreduce(mpi_in_place, alpha, this%m, &
505 mpi_real_precision, mpi_sum, neko_comm, ierr)
506
507 call rzero(work, this%m)
508
509 do i = 1, n, neko_blk_size
510 j = min(neko_blk_size, n-i+1)
511 do l = 0, (j-1)
512 xbar(i+l) = alpha(1) * xx(i+l,1)
513 b(i+l) = b(i+l) - alpha(1) * bb(i+l,1)
514 end do
515 do k = 2, this%m
516 do l = 0, (j-1)
517 xbar(i+l) = xbar(i+l) + alpha(k) * xx(i+l,k)
518 b(i+l) = b(i+l)- alpha(k) * bb(i+l,k)
519 end do
520 end do
521 !Second round of CGS
522 do k = 1, this%m
523 s = 0.0_rp
524 do l = 0, (j-1)
525 s = s + xx(i+l,k) * coef%mult(i+l,1,1,1) * b(i+l)
526 end do
527 work(k) = work(k) + s
528 end do
529 end do
530
531 call mpi_allreduce(work, alpha, this%m, &
532 mpi_real_precision, mpi_sum, neko_comm, ierr)
533
534 do i = 1, n, neko_blk_size
535 j = min(neko_blk_size, n-i+1)
536 do k = 1, this%m
537 do l = 0, (j-1)
538 xbar(i+l) = xbar(i+l) + alpha(k) * xx(i+l,k)
539 b(i+l) = b(i+l) - alpha(k) * bb(i+l,k)
540 end do
541 end do
542 end do
543 end associate
544 end subroutine cpu_project_on

546 subroutine device_project_on(this, b, coef, n)
547 class(projection_t), intent(inout) :: this
548 integer, intent(inout) :: n
549 class(coef_t), intent(inout) :: coef
550 real(kind=rp), intent(inout), dimension(n) :: b
551 real(kind=rp) :: alpha(this%L)
552 type(c_ptr) :: b_d
553 integer :: i
554 b_d = device_get_ptr(b)
555
556 associate(xbar_d => this%xbar_d, xx_d => this%xx_d, xx_d_d => this%xx_d_d, &
557 bb_d => this%bb_d, bb_d_d => this%bb_d_d, alpha_d => this%alpha_d)
558
559 if (this%m .le. 0) return
560
561
562
563 this%proj_res = sqrt(device_glsc3(b_d, b_d, coef%mult_d, n)/coef%volume)
564 this%proj_m = this%m
565 if (neko_device_mpi .and. (neko_bcknd_opencl .ne. 1)) then
566 call device_proj_on(alpha_d, b_d, xx_d_d, bb_d_d, &
567 coef%mult_d, xbar_d, this%m, n)
568 else
569 if (neko_bcknd_opencl .eq. 1) then
570 do i = 1, this%m
571 alpha(i) = device_glsc3(b_d, xx_d(i), coef%mult_d, n)
572 end do
573 else
574 call device_glsc3_many(alpha, b_d, xx_d_d, coef%mult_d, this%m, n)
575 call device_memcpy(alpha, alpha_d, this%m, &
576 host_to_device, sync = .false.)
577 end if
578 call device_rzero(xbar_d, n)

631 subroutine device_proj_ortho(this, xx_d, bb_d, w_d, n)
632 type(projection_t), intent(inout) :: this
633 integer, intent(in) :: n
634 type(c_ptr), dimension(this%L) :: xx_d, bb_d
635 type(c_ptr), intent(in) :: w_d
636 real(kind=rp) :: nrm, scl
637 real(kind=rp) :: alpha(this%L)
638 integer :: i
639
640 associate(m => this%m, xx_d_d => this%xx_d_d, &
641 bb_d_d => this%bb_d_d, alpha_d => this%alpha_d)
642
643 if (m .le. 0) return
644
645 if (neko_device_mpi .and. (neko_bcknd_opencl .ne. 1)) then
646 call device_project_ortho(alpha_d, bb_d(m), xx_d_d, bb_d_d, &
647 w_d, xx_d(m), this%m, n, nrm)
648 else
649 if (neko_bcknd_opencl .eq. 1)then
650 do i = 1, m
651 alpha(i) = device_glsc3(bb_d(m), xx_d(i), w_d,n)
652 end do
653 else
654 call device_glsc3_many(alpha, bb_d(m), xx_d_d, w_d, m, n)
655 end if
656 nrm = sqrt(alpha(m))
657 call cmult(alpha, -1.0_rp,m)
658 if (neko_bcknd_opencl .eq. 1)then
659 do i = 1, m - 1
660 call device_add2s2(xx_d(m), xx_d(i), alpha(i), n)
661 call device_add2s2(bb_d(m), bb_d(i), alpha(i), n)
662
663 alpha(i) = device_glsc3(bb_d(m), xx_d(i), w_d, n)
664 end do
665 else
666 call device_memcpy(alpha, alpha_d, this%m, &
667 host_to_device, sync = .false.)
668 call device_add2s2_many(xx_d(m), xx_d_d, alpha_d, m-1, n)
669 call device_add2s2_many(bb_d(m), bb_d_d, alpha_d, m-1, n)
670
671 call device_glsc3_many(alpha, bb_d(m), xx_d_d, w_d, m, n)
672 end if
673 call cmult(alpha, -1.0_rp,m)
674 if (neko_bcknd_opencl .eq. 1)then
675 do i = 1, m - 1
676 call device_add2s2(xx_d(m), xx_d(i), alpha(i), n)
677 call device_add2s2(bb_d(m), bb_d(i), alpha(i), n)
678 alpha(i) = device_glsc3(bb_d(m), xx_d(i), w_d, n)
679 end do
680 else
681 call device_memcpy(alpha, alpha_d, m, &
682 host_to_device, sync = .false.)
683 call device_add2s2_many(xx_d(m), xx_d_d, alpha_d, m-1, n)
684 call device_add2s2_many(bb_d(m), bb_d_d, alpha_d, m-1, n)
685 call device_glsc3_many(alpha, bb_d(m), xx_d_d, w_d, m, n)
686 end if
687 end if
688
689 alpha(m) = device_glsc3(xx_d(m), w_d, bb_d(m), n)
690 alpha(m) = sqrt(alpha(m))
691
692 if (alpha(m) .gt. this%tol*nrm) then !New vector is linearly independent
693 scl = 1.0_rp / alpha(m)
694 call device_cmult(xx_d(m), scl, n)
695 call device_cmult(bb_d(m), scl, n)
696
697
698 else !New vector is not linearly independent, forget about it
699 if (pe_rank .eq. 0) then
700 call neko_warning('New vector not linearly indepependent!')
701 end if
702 m = m - 1 !Remove column
703 end if
704
705 end associate

707 end subroutine device_proj_ortho

710 subroutine cpu_proj_ortho(this, xx, bb, w, n)
711 type(projection_t), intent(inout) :: this
712 integer, intent(in) :: n
713 real(kind=rp), dimension(n, this%L), intent(inout) :: xx, bb
714 real(kind=rp), dimension(n), intent(in) :: w
715 real(kind=rp) :: nrm, scl1, scl2, c, s
716 real(kind=rp) :: alpha(this%L), beta(this%L)
717 integer :: i, j, k, l, h, ierr
718
719 associate(m => this%m)
720
721 if (m .le. 0) return !No vectors to ortho-normalize
722
723 ! AX = B
724 ! Calculate dx, db: dx = x-XX^Tb, db=b-BX^Tb
725 call rzero(alpha, m)
726 do i = 1, n, neko_blk_size
727 j = min(neko_blk_size, n-i+1)
728 do k = 1, m !First round CGS
729 s = 0.0_rp
730 c = 0.0_rp
731 do l = 0, (j-1)
732 s = s + xx(i+l,k) * w(i+l) * bb(i+l,m)
733 c = c + bb(i+l,k) * w(i+l) * xx(i+l,m)
734 end do
735 alpha(k) = alpha(k) + 0.5_rp * (s + c)
736 end do
737 end do
738
739 call mpi_allreduce(mpi_in_place, alpha, this%m, &
740 mpi_real_precision, mpi_sum, neko_comm, ierr)
741
742 nrm = sqrt(alpha(m)) !Calculate A-norm of new vector
743
744
745 do i = 1, n, neko_blk_size
746 j = min(neko_blk_size, n-i+1)
747 do k = 1, m-1
748 do l = 0, (j-1)
749 xx(i+l,m) = xx(i+l,m) - alpha(k) * xx(i+l,k)
750 bb(i+l,m) = bb(i+l,m) - alpha(k) * bb(i+l,k)
751 end do
752 end do
753 end do
754 call rzero(beta,m)
755
756 do i = 1, n, neko_blk_size
757 j = min(neko_blk_size, n-i+1)
758 do k = 1, m-1
759 s = 0.0_rp
760 c = 0.0_rp
761 do l = 0, (j-1)
762 s = s + xx(i+l,k) * w(i+l) * bb(i+l,m)
763 c = c + bb(i+l,k) * w(i+l) * xx(i+l,m)
764 end do
765 beta(k) = beta(k) + 0.5_rp * (s + c)
766 end do
767 end do
768
769 call mpi_allreduce(mpi_in_place, beta, this%m-1, &
770 mpi_real_precision, mpi_sum, neko_comm, ierr)
771
772 alpha(m) = 0.0_rp
773
774 do i = 1, n, neko_blk_size
775 j = min(neko_blk_size,n-i+1)
776 do k = 1, m-1
777 do l = 0, (j-1)
778 xx(i+l,m) = xx(i+l,m) - beta(k) * xx(i+l,k)
779 bb(i+l,m) = bb(i+l,m) - beta(k) * bb(i+l,k)
780 end do
781 end do
782 s = 0.0_rp
783 do l = 0, (j-1)
784 s = s + xx(i+l,m) * w(i+l) * bb(i+l,m)
785 end do
786 alpha(m) = alpha(m) + s
787 end do
788 do k = 1, m-1
789 alpha(k) = alpha(k) + beta(k)
790 end do
791
792 !alpha(m) = glsc3(xx(1,m), w, bb(1,m), n)
793 call mpi_allreduce(mpi_in_place, alpha(m), 1, &
794 mpi_real_precision, mpi_sum, neko_comm, ierr)
795 alpha(m) = sqrt(alpha(m))
796 !dx and db now stored in last column of xx and bb
797
798 if (alpha(m) .gt. this%tol*nrm) then !New vector is linearly independent
799 !Normalize dx and db
800 scl1 = 1.0_rp / alpha(m)
801 do i = 0, (n - 1)
802 xx(1+i,m) = scl1 * xx(1+i,m)
803 bb(1+i,m) = scl1 * bb(1+i,m)
804 end do
805
806 else !New vector is not linearly independent, forget about it
807 k = m !location of rank deficient column
808 if (pe_rank .eq. 0) then
809 call neko_warning('New vector not linearly indepependent!')
810 end if
811 m = m - 1 !Remove column
812 end if
813
814 end associate
815
816 end subroutine cpu_proj_ortho
