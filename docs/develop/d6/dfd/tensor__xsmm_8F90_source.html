82 subroutine tnsr3d_el_xsmm(v, nv, u, nu, A, Bt, Ct)
83 integer, intent(in) :: nv, nu
84 real(kind=rp), intent(inout) :: v(nv*nv*nv), u(nu*nu*nu)
85 real(kind=rp), intent(inout) :: a(nv,nu),bt(nu, nv),ct(nu,nv)
86 real(kind=rp) :: work(0:nu**2*nv), work2(0:nu*nv**2)
87 integer :: i, nunu, nvnu, nvnv
88
89 nvnu = nv * nu
90 nunu = nu * nu
91 nvnv = nv * nv
92
93 call mxm(a, nv, u(1), nu ,work, nunu)
94 do i = 0,nu-1
95 call mxm(work(nvnu*i), nv, bt, nu, work2(nv*nv*i), nv)
96 end do
97 call mxm(work2, nvnv, ct, nu, v(1), nv)
98
99 end subroutine tnsr3d_el_xsmm

101 subroutine tnsr3d_xsmm(v, nv, u, nu, A, Bt, Ct, nelv)
102 integer, intent(in) :: nv, nu, nelv
103 real(kind=rp), intent(inout) :: v(nv*nv*nv,nelv)
104 real(kind=rp), intent(in) :: u(nu*nu*nu,nelv)
105 real(kind=rp), intent(in) :: a(nv,nu), bt(nu, nv), ct(nu,nv)
106 real(kind=rp) :: work(0:nu**2*nv), work2(0:nu*nv**2)
107 integer :: ie, i, nunu, nvnu, nvnv
108
109 nvnu = nv * nu
110 nunu = nu * nu
111 nvnv = nv * nv
112
113 do ie = 1,nelv
114 call mxm(a, nv, u(1,ie), nu, work, nunu)
115 do i = 0,nu-1
116 call mxm(work(nvnu*i), nv, bt, nu, work2(nv*nv*i), nv)
117 end do
118 call mxm(work2, nvnv, ct, nu, v(1,ie), nv)
119 end do
120
121 end subroutine tnsr3d_xsmm

123 subroutine tnsr1_3d_xsmm(v, nv, nu, A, Bt, Ct, nelv)
124 integer, intent(in) :: nv, nu, nelv
125 real(kind=rp), intent(inout) :: v(nv*nv*nv*nelv)
126 real(kind=rp), intent(inout) :: a(nv,nu), bt(nu, nv), ct(nu,nv)
127 real(kind=rp) :: work(0:nu**2*nv), work2(0:nu*nv**2)
128 integer :: e, e0, ee, es, iu, iv, i, nu3, nv3
129
130 e0 = 1
131 es = 1
132 ee = nelv
133
134 if (nv.gt.nu) then
135 e0 = nelv
136 es = -1
137 ee = 1
138 endif
139
140 nu3 = nu**3
141 nv3 = nv**3
142
143 do e = e0,ee,es
144 iu = 1 + (e-1)*nu3
145 iv = 1 + (e-1)*nv3
146 call mxm(a, nv, v(iu), nu, work, nu*nu)
147 do i = 0,nu-1
148 call mxm(work(nv*nu*i), nv, bt, nu, work2(nv*nv*i), nv)
149 end do
150 call mxm(work2, nv*nv, ct, nu, v(iv), nv)
151 end do
152 end subroutine tnsr1_3d_xsmm
