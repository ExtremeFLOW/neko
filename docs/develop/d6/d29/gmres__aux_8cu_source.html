53
54 const dim3 nthrds(1024, 1, 1);

58
59 if (gmres_bf1 != NULL && gmres_bf_len < nb) {
60 CUDA_CHECK(cudaFreeHost(gmres_bf1));

62 gmres_bf1 = NULL;
63 }
64
65 if (gmres_bf1 == NULL){

69 }

71 gmres_part2_kernel<real>
