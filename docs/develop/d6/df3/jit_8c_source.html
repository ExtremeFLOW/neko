50void opencl_kernel_jit(const char *kernel, cl_program *program)
51{
52 cl_int err;
53
54 (*program) = clCreateProgramWithSource((cl_context) glb_ctx, 1,
55 &kernel, NULL, &err);
56 CL_CHECK(err);
57
58 err = clBuildProgram((*program), 1, (cl_device_id *) &glb_device_id,
59 NULL, NULL, NULL);
60
61 if (err == CL_BUILD_PROGRAM_FAILURE) {
62 size_t log_size;
63 clGetProgramBuildInfo((*program), glb_device_id,
64 CL_PROGRAM_BUILD_LOG, 0, NULL, &log_size);

69 CL_PROGRAM_BUILD_LOG, log_size, log, NULL);

73 }
