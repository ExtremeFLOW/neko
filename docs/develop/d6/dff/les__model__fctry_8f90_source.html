55 module subroutine les_model_factory(object, type_name, dofmap, coef, json)
56 class(les_model_t), allocatable, intent(inout) :: object
57 character(len=*), intent(in) :: type_name
58 type(dofmap_t), intent(in) :: dofmap
59 type(coef_t), intent(in) :: coef
60 type(json_file), intent(inout) :: json
61 character(len=:), allocatable :: type_string
62
63 if (allocated(object)) then
64 deallocate(object)
65 else if (trim(type_name) .eq. 'vreman') then
66 allocate(vreman_t::object)
67 else if (trim(type_name) .eq. 'smagorinsky') then
68 allocate(smagorinsky_t::object)
69 else if (trim(type_name) .eq. 'dynamic_smagorinsky') then
70 allocate(dynamic_smagorinsky_t::object)
71 else if (trim(type_name) .eq. 'sigma') then
72 allocate(sigma_t::object)
73 else
74 type_string = concat_string_array(les_known_types, &
75 new_line('A') // "- ", .true.)
76 call neko_error("Unknown LES model type: " &
77 // trim(type_name) // ". Known types are: " &
78 // type_string)
79 stop
80
81 end if
82 call object%init(dofmap, coef, json)
83 end subroutine les_model_factory
