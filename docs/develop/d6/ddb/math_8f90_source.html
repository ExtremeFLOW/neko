461  subroutine vcross(u1, u2, u3, v1, v2, v3, w1, w2, w3, n)
462  integer, intent(in) :: n
463  real(kind=rp), dimension(n), intent(in) :: v1, v2, v3
464  real(kind=rp), dimension(n), intent(in) :: w1, w2, w3
465  real(kind=rp), dimension(n), intent(out) :: u1, u2, u3
466  integer :: i
467 
468  do i = 1, n
469  u1(i) = v2(i)*w3(i) - v3(i)*w2(i)
470  u2(i) = v3(i)*w1(i) - v1(i)*w3(i)
471  u3(i) = v1(i)*w2(i) - v2(i)*w1(i)
472  end do
473 
474  end subroutine vcross
