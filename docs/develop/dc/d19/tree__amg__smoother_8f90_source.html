178  subroutine amg_cheby_solve(this, x, f, n, amg, niter)
179  class(amg_cheby_t), intent(inout) :: this
180  integer, intent(in) :: n
181  real(kind=rp), dimension(n), intent(inout) :: x

183  class(tamg_hierarchy_t), intent(inout) :: amg
184  type(ksp_monitor_t) :: ksp_results

191  call this%comp_eig(amg, n)
192  end if

194  if (present(niter)) then
195  max_iter = niter
196  else
197  max_iter = this%max_iter
198  end if
199 
200  associate( w => this%w, r => this%r, d => this%d, blst=>amg%blst)
201  call copy(r, f, n)
202  w = 0d0
203  call amg%matvec(w, x, this%lvl)
204  call sub2(r, w, n)
205 
206  thet = this%tha
207  delt = this%dlt
208  s1 = thet / delt
209  rhok = 1.0_rp / s1
210 
211  call copy(d, r, n)
212  call cmult(d, 1.0_rp/thet, n)
213 
214  do iter = 1, max_iter
215  call add2(x,d,n)
216 
217  w = 0d0
218  call amg%matvec(w, d, this%lvl)
219  call sub2(r, w, n)
220 
221  rhokp1 = 1.0_rp / (2.0_rp * s1 - rhok)
222  tmp1 = rhokp1 * rhok
223  tmp2 = 2.0_rp * rhokp1 / delt
224  rhok = rhokp1
225 
226  call cmult(d, tmp1, n)
227  call add2s2(d, r, tmp2, n)
228  end do
