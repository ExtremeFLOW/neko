269  subroutine fluid_scheme_init_common(this, msh, lx, params, scheme, user, &
270  kspv_init)
271  implicit none
272  class(fluid_scheme_t), target, intent(inout) :: this
273  type(mesh_t), target, intent(inout) :: msh
274  integer, intent(in) :: lx
275  character(len=*), intent(in) :: scheme
276  type(json_file), target, intent(inout) :: params
277  type(user_t), target, intent(in) :: user
278  logical, intent(in) :: kspv_init
279  type(dirichlet_t) :: bdry_mask
280  character(len=LOG_SIZE) :: log_buf
281  real(kind=rp), allocatable :: real_vec(:)
282  real(kind=rp) :: real_val, kappa, b, z0
283  logical :: logical_val
284  integer :: integer_val, ierr
285  type(json_file) :: wm_json

288  !
289  ! SEM simulation fundamentals
290  !
291 
292  this%msh => msh
293 
294  if (msh%gdim .eq. 2) then
295  call this%Xh%init(gll, lx, lx)
296  else
297  call this%Xh%init(gll, lx, lx, lx)
298  end if
299 
300  call this%dm_Xh%init(msh, this%Xh)
301 
302  call this%gs_Xh%init(this%dm_Xh)
303 
304  call this%c_Xh%init(this%gs_Xh)
305 
306  ! Local scratch registry
307  this%scratch = scratch_registry_t(this%dm_Xh, 10, 2)
308 
309  ! Case parameters
310  this%params => params
311 
312 
313  !
314  ! First section of fluid log
315  !
316 
317  call neko_log%section('Fluid')
318  write(log_buf, '(A, A)') 'Type : ', trim(scheme)
319  call neko_log%message(log_buf)
320 
321  !
322  ! Material properties
323  !
324  call this%set_material_properties(params, user)
325 
326  !
327  ! Turbulence modelling and variable material properties
328  !
329  if (params%valid_path('case.fluid.nut_field')) then
330  call json_get(params, 'case.fluid.nut_field', this%nut_field_name)
331  this%variable_material_properties = .true.
332  else
333  this%nut_field_name = ""
334  end if
335 
336  ! Fill mu and rho field with the physical value
337 
338  call this%mu_field%init(this%dm_Xh, "mu")
339  call this%rho_field%init(this%dm_Xh, "mu")
340  call field_cfill(this%mu_field, this%mu, this%mu_field%size())
341  call field_cfill(this%rho_field, this%rho, this%mu_field%size())
342 
343  ! Since mu, rho is a field, and the none-stress simulation fetches
344  ! data from the host arrays, we need to mirror the constant
345  ! material properties on the host
346  if (neko_bcknd_device .eq. 1) then
347  call cfill(this%mu_field%x, this%mu, this%mu_field%size())
348  call cfill(this%rho_field%x, this%rho, this%rho_field%size())
349  end if
350 
351 
352  ! Projection spaces
353  call json_get_or_default(params, &
354  'case.fluid.velocity_solver.projection_space_size', &
355  this%vel_projection_dim, 20)
356  call json_get_or_default(params, &
357  'case.fluid.pressure_solver.projection_space_size', &
358  this%pr_projection_dim, 20)
359  call json_get_or_default(params, &
360  'case.fluid.velocity_solver.projection_hold_steps', &
361  this%vel_projection_activ_step, 5)
362  call json_get_or_default(params, &
363  'case.fluid.pressure_solver.projection_hold_steps', &
364  this%pr_projection_activ_step, 5)
365 
366 
367  call json_get_or_default(params, 'case.fluid.freeze', this%freeze, .false.)
368 
369  if (params%valid_path("case.fluid.flow_rate_force")) then
370  this%forced_flow_rate = .true.
371  end if
372 
373 
374  if (lx .lt. 10) then
375  write(log_buf, '(A, I1)') 'Poly order : ', lx-1
376  else if (lx .ge. 10) then
377  write(log_buf, '(A, I2)') 'Poly order : ', lx-1
378  else
379  write(log_buf, '(A, I3)') 'Poly order : ', lx-1
380  end if
381  call neko_log%message(log_buf)
382  this%glb_n_points = int(this%msh%glb_nelv, i8)*int(this%Xh%lxyz, i8)
383  this%glb_unique_points = int(glsum(this%c_Xh%mult, this%dm_Xh%size()), i8)
384 
385  write(log_buf, '(A, I0)') 'GLL points : ', this%glb_n_points
386  call neko_log%message(log_buf)
387  write(log_buf, '(A, I0)') 'Unique pts.: ', this%glb_unique_points
388  call neko_log%message(log_buf)
389 
390  write(log_buf, '(A,ES13.6)') 'rho :', this%rho
391  call neko_log%message(log_buf)
392  write(log_buf, '(A,ES13.6)') 'mu :', this%mu
393  call neko_log%message(log_buf)
394 
395  call json_get(params, 'case.numerics.dealias', logical_val)
396  write(log_buf, '(A, L1)') 'Dealias : ', logical_val
397  call neko_log%message(log_buf)
398 
399  call json_get_or_default(params, 'case.output_boundary', logical_val, &
400  .false.)
401  write(log_buf, '(A, L1)') 'Save bdry : ', logical_val
402  call neko_log%message(log_buf)
403  if (logical_val) then
404  write(log_buf, '(A)') 'bdry keys: '
405  call neko_log%message(log_buf)
406  write(log_buf, '(A)') 'No-slip wall ''w'' = 1'
407  call neko_log%message(log_buf)
408  write(log_buf, '(A)') 'Inlet/velocity dirichlet ''v'' = 2'
409  call neko_log%message(log_buf)
410  write(log_buf, '(A)') 'Outlet ''o'' = 3'
411  call neko_log%message(log_buf)
412  write(log_buf, '(A)') 'Symmetry ''sym'' = 4'
413  call neko_log%message(log_buf)
414  write(log_buf, '(A)') 'Outlet-normal ''on'' = 5'
415  call neko_log%message(log_buf)
416  write(log_buf, '(A)') 'Periodic = 6'
417  call neko_log%message(log_buf)
418  write(log_buf, '(A)') 'Dirichlet on velocity components: '
419  call neko_log%message(log_buf)
420  write(log_buf, '(A)') ' ''d_vel_u, d_vel_v, d_vel_w'' = 7'
421  call neko_log%message(log_buf)
422  write(log_buf, '(A)') 'Pressure dirichlet ''d_pres'' = 8'
423  call neko_log%message(log_buf)
424  write(log_buf, '(A)') '''d_vel_(u,v,w)'' and ''d_pres'' = 8'
425  call neko_log%message(log_buf)
426  write(log_buf, '(A)') 'Shear stress ''sh'' = 9'
427  call neko_log%message(log_buf)
428  write(log_buf, '(A)') 'Wall modelling ''wm'' = 10'
429  call neko_log%message(log_buf)
430  write(log_buf, '(A)') 'No boundary condition set = 0'
431  call neko_log%message(log_buf)
432  end if
433 
434 
435  !
436  ! Setup velocity boundary conditions
437  !
438  allocate(this%bc_labels(neko_msh_max_zlbls))
439  this%bc_labels = "not"
440 
441  if (params%valid_path('case.fluid.boundary_types')) then
442  call json_get(params, &
443  'case.fluid.boundary_types', &
444  this%bc_labels)
445  end if
446 
447  call bc_list_init(this%bclst_vel)
448 
449  call this%bc_sym%init_base(this%c_Xh)
450  call this%bc_sym%mark_zone(msh%sympln)
451  call this%bc_sym%mark_zones_from_list(msh%labeled_zones, &
452  'sym', this%bc_labels)
453  call this%bc_sym%finalize()
454  call this%bc_sym%init(this%c_Xh)
455  call bc_list_add(this%bclst_vel, this%bc_sym)
456 
457  ! Shear stress conditions
458  call this%bc_sh%init_base(this%c_Xh)
459  call this%bc_sh%mark_zones_from_list(msh%labeled_zones, &
460  'sh', this%bc_labels)
461  call this%bc_sh%finalize()
462  ! This marks the dirichlet and neumann conditions inside, and finalizes
463  ! them.
464  call this%bc_sh%init_shear_stress(this%c_Xh)
465 
466  call bc_list_add(this%bclst_vel, this%bc_sh%symmetry)
467 
468  ! Read stress value, default to [0 0 0]
469  if (this%bc_sh%msk(0) .gt. 0) then
470  call params%get('case.fluid.shear_stress.value', real_vec, logical_val)
471  if (.not. logical_val .and. this%bc_sh%msk(0) .gt. 0) then
472  call neko_warning("No stress values provided for sh boundaries, &
473  & defaulting to 0. Use fluid.shear_stress.value to set.")
474  allocate(real_vec(3))
475  real_vec = 0.0_rp
476  else if (size(real_vec) .ne. 3) then
477  call neko_error ("The shear stress vector provided in &
478  &fluid.shear_stress.value should have 3 components.")
479  end if
480  call this%bc_sh%set_stress(real_vec(1), real_vec(2), real_vec(3))
481  end if
482 
483  call bc_list_init(this%bclst_vel_neumann)
484  call bc_list_add(this%bclst_vel_neumann, this%bc_sh)
485 
486  !
487  ! Inflow
488  !
489  if (params%valid_path('case.fluid.inflow_condition')) then
490  call json_get(params, 'case.fluid.inflow_condition.type', string_val1)
491  if (trim(string_val1) .eq. "uniform") then
492  allocate(inflow_t::this%bc_inflow)
493  else if (trim(string_val1) .eq. "blasius") then
494  allocate(blasius_t::this%bc_inflow)
495  else if (trim(string_val1) .eq. "user") then
496  allocate(usr_inflow_t::this%bc_inflow)
497  else
498  call neko_error('Invalid inflow condition '//string_val1)
499  end if
500 
501  call this%bc_inflow%init_base(this%c_Xh)
502  call this%bc_inflow%mark_zone(msh%inlet)
503  call this%bc_inflow%mark_zones_from_list(msh%labeled_zones, &
504  'v', this%bc_labels)
505  call this%bc_inflow%finalize()
506  call bc_list_add(this%bclst_vel, this%bc_inflow)
507 
508  if (trim(string_val1) .eq. "uniform") then
509  call json_get(params, 'case.fluid.inflow_condition.value', real_vec)
510  select type (bc_if => this%bc_inflow)
511  type is (inflow_t)
512  call bc_if%set_inflow(real_vec)
513  end select
514  else if (trim(string_val1) .eq. "blasius") then
515  select type (bc_if => this%bc_inflow)
516  type is (blasius_t)
517  call json_get(params, 'case.fluid.blasius.delta', real_val)
518  call json_get(params, 'case.fluid.blasius.approximation', &
519  string_val2)
520  call json_get(params, 'case.fluid.blasius.freestream_velocity', &
521  real_vec)
522 
523  call bc_if%set_params(real_vec, real_val, string_val2)
524 
525  end select
526  else if (trim(string_val1) .eq. "user") then
527  end if
528  end if
529 
530  !
531  ! Wall models
532  !
533 
534  call this%bc_wallmodel%init_base(this%c_Xh)
535  call this%bc_wallmodel%mark_zones_from_list(msh%labeled_zones,&
536  'wm', this%bc_labels)
537  call this%bc_wallmodel%finalize()
538 
539  if (this%bc_wallmodel%msk(0) .gt. 0) then
540  call json_extract_object(params, 'case.fluid.wall_modelling', wm_json)
541  call this%bc_wallmodel%init_wall_model_bc(wm_json, this%mu / this%rho)
542  else
543  call this%bc_wallmodel%shear_stress_t%init_shear_stress(this%c_Xh)
544  end if
545 
546  call bc_list_add(this%bclst_vel, this%bc_wallmodel%symmetry)
547  call bc_list_add(this%bclst_vel_neumann, this%bc_wallmodel)
548 
549  call this%bc_wall%init_base(this%c_Xh)
550  call this%bc_wall%mark_zone(msh%wall)
551  call this%bc_wall%mark_zones_from_list(msh%labeled_zones, &
552  'w', this%bc_labels)
553  call this%bc_wall%finalize()
554  call bc_list_add(this%bclst_vel, this%bc_wall)
555 
556  ! Setup field dirichlet bc for u-velocity
557  call this%user_field_bc_vel%bc_u%init_base(this%c_Xh)
558  call this%user_field_bc_vel%bc_u%mark_zones_from_list(msh%labeled_zones, &
559  'd_vel_u', this%bc_labels)
