78  class(ksp_t), allocatable, target, intent(inout) :: object
79  integer, intent(in), value :: n
80  character(len=*), intent(in) :: type_name
81  integer, intent(in) :: max_iter
82  real(kind=rp), optional :: abstol
83  class(pc_t), optional, intent(inout), target :: m
84  character(len=:), allocatable :: type_string
85 
86  if (allocated(object)) then
87  call krylov_solver_destroy(object)
88  deallocate(object)
89  end if
90 
91  if (trim(type_name) .eq. 'cg') then
92  if (neko_bcknd_sx .eq. 1) then
93  allocate(sx_cg_t::object)
94  else if (neko_bcknd_device .eq. 1) then
95  allocate(cg_device_t::object)
96  else
97  allocate(cg_t::object)
98  end if
99  else if (trim(type_name) .eq. 'cpldcg') then
100  allocate(cg_cpld_t::object)
101  if (neko_bcknd_device .eq. 1) then
102  call neko_error('Coupled CG only supported for CPU')
103  end if
104  else if (trim(type_name) .eq. 'pipecg') then
105  if (neko_bcknd_sx .eq. 1) then
106  allocate(sx_pipecg_t::object)
107  else if (neko_bcknd_device .eq. 1) then
108  if (neko_bcknd_opencl .eq. 1) then
109  call neko_error('PipeCG not supported for OpenCL')
110  end if
111  allocate(pipecg_device_t::object)
112  else
113  allocate(pipecg_t::object)
114  end if
115  else if (trim(type_name) .eq. 'fusedcg') then
116  if (neko_bcknd_device .eq. 1) then
117  if (neko_bcknd_opencl .eq. 1) then
118  call neko_error('FusedCG not supported for OpenCL')
119  end if
120  allocate(fusedcg_device_t::object)
121  else
122  call neko_error('FusedCG only supported for CUDA/HIP')
123  end if
124  else if (trim(type_name) .eq. 'fcpldcg') then
125  if (neko_bcknd_device .eq. 1) then
126  if (neko_bcknd_opencl .eq. 1) then
127  call neko_error('Coupled FusedCG not supported for OpenCL')
128  end if
129  allocate(fusedcg_cpld_device_t::object)
130  else
131  call neko_error('Coupled FusedCG only supported for CUDA/HIP')
132  end if
133  else if (trim(type_name) .eq. 'cacg') then
134  allocate(cacg_t::object)
135  else if (trim(type_name) .eq. 'gmres') then
136  if (neko_bcknd_sx .eq. 1) then
137  allocate(sx_gmres_t::object)
138  else if (neko_bcknd_device .eq. 1) then
139  allocate(gmres_device_t::object)
140  else
141  allocate(gmres_t::object)
142  end if
143  else if (trim(type_name) .eq. 'bicgstab') then
144  allocate(bicgstab_t::object)
145  else
146  type_string = concat_string_array(known_types, new_line('A') // "- ", &
147  .true.)
148  call neko_error("Unknown Krylov solver type: " &
149  // trim(type_name) // ". Known types are: " &
150  // type_string)
151  end if
152 
153  ! This select type is in principle not necessary,but we have it due to
154  ! issues with compilers, when it was not there. However, at some point we
155  ! should check if we can get away with just having one obj%init statement.
156  ! Same applies to the code in the "destroy" routine below.
157  if (present(abstol) .and. present(m)) then
158  select type (obj => object)
159  type is (cg_t)
160  call obj%init(n, max_iter, m = m, abs_tol = abstol)
161  type is (sx_cg_t)
162  call obj%init(n, max_iter, m = m, abs_tol = abstol)
163  type is (cg_cpld_t)
164  call obj%init(n, max_iter, m = m, abs_tol = abstol)
165  type is (cg_device_t)
166  call obj%init(n, max_iter, m = m, abs_tol = abstol)
167  type is (pipecg_t)
168  call obj%init(n, max_iter, m = m, abs_tol = abstol)
169  type is (sx_pipecg_t)
170  call obj%init(n, max_iter, m = m, abs_tol = abstol)
171  type is (pipecg_device_t)
172  call obj%init(n, max_iter, m = m, abs_tol = abstol)
173  type is (fusedcg_device_t)
174  call obj%init(n, max_iter, m = m, abs_tol = abstol)
175  type is (fusedcg_cpld_device_t)
176  call obj%init(n, max_iter, m = m, abs_tol = abstol)
177  type is (cacg_t)
178  call obj%init(n, max_iter, m = m, abs_tol = abstol)
179  type is (gmres_t)
180  call obj%init(n, max_iter, m = m, abs_tol = abstol)
181  type is (sx_gmres_t)
182  call obj%init(n, max_iter, m = m, abs_tol = abstol)
183  type is (gmres_device_t)
184  call obj%init(n, max_iter, m = m, abs_tol = abstol)
185  type is (bicgstab_t)
186  call obj%init(n, max_iter, m = m, abs_tol = abstol)
187  end select
188  else if (present(abstol)) then
189  select type (obj => object)
190  type is (cg_t)
191  call obj%init(n, max_iter, abs_tol = abstol)
192  type is (sx_cg_t)
193  call obj%init(n, max_iter, abs_tol = abstol)
194  type is (cg_cpld_t)
195  call obj%init(n, max_iter, abs_tol = abstol)
196  type is (cg_device_t)
197  call obj%init(n, max_iter, abs_tol = abstol)
198  type is (pipecg_t)
199  call obj%init(n, max_iter, abs_tol = abstol)
200  type is (sx_pipecg_t)
201  call obj%init(n, max_iter, abs_tol = abstol)
202  type is (pipecg_device_t)
203  call obj%init(n, max_iter, abs_tol = abstol)
204  type is (fusedcg_device_t)
205  call obj%init(n, max_iter, abs_tol = abstol)
206  type is (fusedcg_cpld_device_t)
207  call obj%init(n, max_iter, abs_tol = abstol)
208  type is (cacg_t)
209  call obj%init(n, max_iter, abs_tol = abstol)
210  type is (gmres_t)
211  call obj%init(n, max_iter, abs_tol = abstol)
212  type is (sx_gmres_t)
213  call obj%init(n, max_iter, abs_tol = abstol)
214  type is (gmres_device_t)
215  call obj%init(n, max_iter, abs_tol = abstol)
216  type is (bicgstab_t)
217  call obj%init(n, max_iter, abs_tol = abstol)
218  end select
219  else if (present(m)) then
220  select type (obj => object)
221  type is (cg_t)
222  call obj%init(n, max_iter, m = m)
223  type is (sx_cg_t)
224  call obj%init(n, max_iter, m = m)
225  type is (cg_cpld_t)
226  call obj%init(n, max_iter, m = m)
227  type is (cg_device_t)
228  call obj%init(n, max_iter, m = m)
229  type is (pipecg_t)
230  call obj%init(n, max_iter, m = m)
231  type is (sx_pipecg_t)
232  call obj%init(n, max_iter, m = m)
233  type is (pipecg_device_t)
234  call obj%init(n, max_iter, m = m)
235  type is (fusedcg_device_t)
236  call obj%init(n, max_iter, m = m)
237  type is (fusedcg_cpld_device_t)
238  call obj%init(n, max_iter, m = m)
239  type is (cacg_t)
240  call obj%init(n, max_iter, m = m)
241  type is (gmres_t)
242  call obj%init(n, max_iter, m = m)
243  type is (sx_gmres_t)
244  call obj%init(n, max_iter, m = m)
245  type is (gmres_device_t)
246  call obj%init(n, max_iter, m = m)
247  type is (bicgstab_t)
248  call obj%init(n, max_iter, m = m)
249  end select
250  else
251  select type (obj => object)
252  type is (cg_t)
253  call obj%init(n, max_iter)
254  type is (sx_cg_t)
255  call obj%init(n, max_iter)
256  type is (cg_cpld_t)
257  call obj%init(n, max_iter)
258  type is (cg_device_t)

267  call obj%init(n, max_iter)
268  type is (fusedcg_cpld_device_t)
269  call obj%init(n, max_iter)
270  type is (cacg_t)
271  call obj%init(n, max_iter)
272  type is (gmres_t)
273  call obj%init(n, max_iter)
274  type is (sx_gmres_t)
275  call obj%init(n, max_iter)
276  type is (gmres_device_t)
277  call obj%init(n, max_iter)
278  type is (bicgstab_t)
279  call obj%init(n, max_iter)
280  end select
281  end if

283  end subroutine krylov_solver_factory
