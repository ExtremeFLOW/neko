151 subroutine scratch_registry_free(this)
152 class(scratch_registry_t), intent(inout):: this
153 integer :: i
154
155 if (allocated(this%inuse)) then
156 if(any(this%inuse)) then
157 call neko_error("scratch_registry::free: "&
158 // "Cannot free scratch registry with in-use entries.")
159 end if
160 deallocate(this%inuse)
161 end if
162
163 if (allocated(this%registry)) then
164 do i = 1, this%n_available
165 call this%registry(i)%free()
166 end do
167
168 deallocate(this%registry)
169 end if
170
171 if (associated(this%dof)) nullify(this%dof)
172
173 ! Reset to default values
174 this%n_available = 0
175 this%n_inuse = 0
176 this%expansion_size = 10
177

239 subroutine expand(this)
240 class(scratch_registry_t), intent(inout) :: this
241 type(registry_entry_t), allocatable :: temp(:)
242 logical, allocatable :: temp2(:)
243
244 allocate(temp(this%get_size() + this%expansion_size))
245 temp(1:this%n_available) = this%registry(1:this%n_available)
246
247 call move_alloc(temp, this%registry)

251 temp2(this%n_available+1:) = .false.

256 subroutine request_field(this, f, index)
257 class(scratch_registry_t), target, intent(inout) :: this
258 type(field_t), pointer, intent(inout) :: f
259 integer, intent(inout) :: index
260 character(len=10) :: name
261
262 if (.not. associated(this%dof)) then
263 call neko_error("scratch_registry::request_field: "&
264 // "No dofmap assigned to scratch registry.")
265 end if
266
267 associate(n_available => this%n_available, n_inuse => this%n_inuse)
268
269 do index = 1, this%get_size()
270 if (.not. this%inuse(index)) then
271
272 if (.not. this%registry(index)%is_allocated()) then
273 write(name, "(A3,I0.3)") "wrk", index
274 call this%registry(index)%init_field(this%dof, trim(name))
275 n_available = n_available + 1
276 else if (this%registry(index)%get_type() .ne. 'field') then
277 cycle
278 end if
279
280 f => this%registry(index)%field_ptr
281 call field_rzero(f)
282 this%inuse(index) = .true.
283 this%n_inuse = this%n_inuse + 1
284 return
285 end if
286 end do
287
288 ! all existing fields in use, we need to expand to add a new one
289 index = n_available + 1
290 call this%expand()
291 n_available = n_available + 1
292 n_inuse = n_inuse + 1
293 this%inuse(n_available) = .true.
294 write (name, "(A3,I0.3)") "wrk", index
295 call this%registry(n_available)%init_field(this%dof, trim(name))
296 f => this%registry(n_available)%field_ptr
297
298 end associate
299 end subroutine request_field
