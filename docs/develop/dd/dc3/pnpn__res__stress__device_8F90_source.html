245 call neko_scratch_registry%request_field(s12, temp_indices(12), .false.)

250 lxyz = c_xh%Xh%lxyz
251 nelv = c_xh%msh%nelv
252 gdim = c_xh%msh%gdim

255 call device_invcol1(c_xh%h1_d, n)

265 call device_col2(wa3%x_d, mu%x_d, n)
266
267
268 ! The strain rate tensor
269 call strain_rate(s11%x, s22%x, s33%x, s12%x, s13%x, s23%x, &
270 u_e, v_e, w_e, c_xh)
271
272
273 ! Gradient of viscosity * 2
274 !call opgrad(ta1%x, ta2%x, ta3%x, mu%x, c_Xh)
275 call dudxyz(ta1%x, mu%x, c_xh%drdx, c_xh%dsdx, c_xh%dtdx, c_xh)

277 call dudxyz(ta3%x, mu%x, c_xh%drdz, c_xh%dsdz, c_xh%dtdz, c_xh)

289 f_x%x_d, f_y%x_d, f_z%x_d, &
290 c_xh%B_d, c_xh%h1_d, rho%x_d, n)

310 call cdtp(wa3%x, ta3%x, c_xh%drdz, c_xh%dsdz, c_xh%dtdz, c_xh)
311
312 ! The laplacian of the pressure
313 call ax%compute(p_res%x, p%x, c_xh, p%msh, p%Xh)
314
315#ifdef HAVE_HIP

317#elif HAVE_CUDA

358 class(ax_t), intent(in) :: Ax
359 type(mesh_t), intent(inout) :: msh
360 type(space_t), intent(inout) :: Xh
361 type(field_t), intent(inout) :: p, u, v, w
362 type(field_t), intent(inout) :: u_res, v_res, w_res
363 type(field_t), intent(in) :: f_x, f_y, f_z
364 type(coef_t), intent(inout) :: c_Xh
365 type(field_t), intent(in) :: mu
366 type(field_t), intent(in) :: rho
367 real(kind=rp), intent(in) :: bd
368 real(kind=rp), intent(in) :: dt

376 call device_copy(c_xh%h2_d, rho%x_d, n)

380

382

388 call neko_scratch_registry%request_field(ta2, temp_indices(2), .false.)
389 call neko_scratch_registry%request_field(ta3, temp_indices(3), .false.)
390
391 ! Pressure gradient
392 call opgrad(ta1%x, ta2%x, ta3%x, p%x, c_xh)
393
394 ! Sum all the terms

396 call pnpn_vel_res_update_hip(u_res%x_d, v_res%x_d, w_res%x_d, &
397 ta1%x_d, ta2%x_d, ta3%x_d, f_x%x_d, f_y%x_d, f_z%x_d, n)
398#elif HAVE_CUDA

400 ta1%x_d, ta2%x_d, ta3%x_d, f_x%x_d, f_y%x_d, f_z%x_d, n)
401#elif HAVE_OPENCL
402 call pnpn_vel_res_update_opencl(u_res%x_d, v_res%x_d, w_res%x_d, &
403 ta1%x_d, ta2%x_d, ta3%x_d, f_x%x_d, f_y%x_d, f_z%x_d, n)
404#endif
405
406 call neko_scratch_registry%relinquish_field(temp_indices)

408 end subroutine pnpn_vel_res_stress_device_compute
