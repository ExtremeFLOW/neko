109 subroutine most_init(this, scheme_name, coef, msk, facet, &
110 h_index, json)
111 class(most_t), intent(inout) :: this
112 character(len=*), intent(in) :: scheme_name
113 type(coef_t), intent(in) :: coef
114 integer, intent(in) :: msk(:)
115 integer, intent(in) :: facet(:)
116 integer, intent(in) :: h_index
117 type(json_file), intent(inout) :: json
118 real(kind=rp) :: kappa, z0, z0h_in, pr
119 character(len=:), allocatable :: bc_type
120 character(len=:), allocatable :: scalar_name
121 real(kind=rp) :: bc_value
122 real(kind=rp), allocatable :: g_tmp(:)
123 real(kind=rp) :: g(3)
124 logical :: if_time_dependent
125
126 call json_get_or_lookup_or_default(json, "kappa", kappa, 0.4_rp)
127 call json_get_or_lookup_or_default(json, "Pr", pr, 1.0_rp)
128 call json_get_or_lookup(json, "z0", z0)
129 ! If z0h is specified and positive, z0h will be constant and equal to
130 ! what's specified in the case file.
131 ! If z0h is specified and negative, the Zilitinkevich 1995 formulation
132 ! is used, with the specified value acting as -C_Zil.
133 ! If z0h is not specified, assign to have the same value as z0.
134 call json_get_or_lookup_or_default(json, "z0h", z0h_in, z0)
135 call json_get(json, "type_of_temp_bc", bc_type)
136 call json_get(json, "scalar_field", scalar_name)
137 call json_get_or_lookup(json, "bottom_bc_flux_or_temp", bc_value)
138 call json_get_or_default(json, "time_dependent_temp_bc", &
139 if_time_dependent, .false.)
140
141 if (if_time_dependent) then
142 call neko_const_registry%add_real_scalar(bc_value, "bc_value")
143 end if
144
145 call json_get_or_lookup(json, "g", g_tmp)
146 if (size(g_tmp) == 3) then
147 g = g_tmp
148 else
149 call neko_error("MOST WM: The gravity vector should have exactly 3 components")
150 end if
151 deallocate(g_tmp)
152
153 call this%init_from_components(scheme_name, scalar_name, coef, msk, facet, h_index, &
154 kappa, g, pr, z0, z0h_in, bc_type, bc_value)
155 deallocate(bc_type)
156 deallocate(scalar_name)
157 end subroutine most_init

220 subroutine most_finalize(this, msk, facet)
221 class(most_t), intent(inout) :: this
222 integer, intent(in) :: msk(:)
223 integer, intent(in) :: facet(:)
224 integer :: i, fid
225
226
227 call this%finalize_base(msk, facet)
228
229 call this%Ri_b%init(this%n_nodes)
230 call this%L_ob%init(this%n_nodes)
231 call this%utau%init(this%n_nodes)
232 call this%magu%init(this%n_nodes)
233 call this%ti%init(this%n_nodes)
234 call this%ts%init(this%n_nodes)
235 call this%q%init(this%n_nodes)
236
237 allocate(this%h_x_idx(this%n_nodes))
238 allocate(this%h_y_idx(this%n_nodes))
239 allocate(this%h_z_idx(this%n_nodes))
240 if (neko_bcknd_device .eq. 1) then
241 call device_map(this%h_x_idx, this%h_x_idx_d, this%n_nodes)
242 call device_map(this%h_y_idx, this%h_y_idx_d, this%n_nodes)
243 call device_map(this%h_z_idx, this%h_z_idx_d, this%n_nodes)
244 end if
245
246 do i = 1, this%n_nodes
247 fid = this%facet(i)
248 select case (fid)
249 case (1)
250 this%h_x_idx(i) = this%h_index
251 this%h_y_idx(i) = 0
252 this%h_z_idx(i) = 0
253 case (2)
254 this%h_x_idx(i) = - this%h_index
255 this%h_y_idx(i) = 0
256 this%h_z_idx(i) = 0
257 case (3)
258 this%h_x_idx(i) = 0
259 this%h_y_idx(i) = this%h_index
260 this%h_z_idx(i) = 0
261 case (4)
262 this%h_x_idx(i) = 0
263 this%h_y_idx(i) = - this%h_index
264 this%h_z_idx(i) = 0
265 case (5)
266 this%h_x_idx(i) = 0
267 this%h_y_idx(i) = 0
268 this%h_z_idx(i) = this%h_index
269 case (6)
270 this%h_x_idx(i) = 0
271 this%h_y_idx(i) = 0
272 this%h_z_idx(i) = - this%h_index
273 case default
274 call neko_error("The face index is not correct (most.f90)")
275 end select
276
277 if (neko_bcknd_device .eq. 1) then
278 call device_memcpy(this%h_x_idx, this%h_x_idx_d, this%n_nodes, &
279 host_to_device, sync = .false.)
280 call device_memcpy(this%h_y_idx, this%h_y_idx_d, this%n_nodes, &
281 host_to_device, sync = .false.)
282 call device_memcpy(this%h_z_idx, this%h_z_idx_d, this%n_nodes, &
283 host_to_device, sync = .false.)
284 end if
285
286 end do
287
288 call device_memcpy(this%h_x_idx, this%h_x_idx_d, this%n_nodes, &
289 host_to_device, sync = .false.)
290 call device_memcpy(this%h_y_idx, this%h_y_idx_d, this%n_nodes, &
291 host_to_device, sync = .false.)
292 call device_memcpy(this%h_z_idx, this%h_z_idx_d, this%n_nodes, &
293 host_to_device, sync = .true.)
294 end subroutine most_finalize

309 subroutine most_init_from_components(this, scheme_name, scalar_name, &
310 coef, msk, facet, h_index, kappa, g, Pr, z0, &
311 z0h_in, bc_type, bc_value)
312 class(most_t), intent(inout) :: this
313 character(len=*), intent(in) :: scheme_name
314 character(len=*), intent(in) :: bc_type
315 character(len=*), intent(in) :: scalar_name
316 type(coef_t), intent(in) :: coef
317 integer, intent(in) :: msk(:)
318 integer, intent(in) :: facet(:)
319 integer, intent(in) :: h_index
320 real(kind=rp), intent(in) :: g(3)
321 real(kind=rp) :: g_mag, g_dot_n, cos_alpha, max_ang
322 integer :: i
323 real(kind=rp), intent(in) :: kappa
324 real(kind=rp), intent(in) :: z0, z0h_in, bc_value, pr
325 character(len=LOG_SIZE) :: log_buf
326
327 call this%init_base(scheme_name, coef, msk, facet, h_index)
328
329 this%kappa = kappa
330 this%g = g
331 this%Pr = pr
332 this%z0 = z0
333 this%z0h_in = z0h_in
334 this%bc_type = bc_type
335 this%bc_value = bc_value
336 this%scalar_name = scalar_name
337
339 g_mag = sqrt(sum(g**2))
340 if (g_mag < 1.0e-6_rp) then
341 call neko_error("MOST WM: Gravity magnitude is zero. Check your input configuration.")
342 end if
343
345 max_ang = 0.0_rp
346 do i = 1, this%n_nodes
347 g_dot_n = abs(g(1)*this%n_x%x(i) + g(2)*this%n_y%x(i) + g(3)*this%n_z%x(i))
348 cos_alpha = g_dot_n / g_mag
349 max_ang = max(max_ang, acos(min(1.0_rp, cos_alpha)))
350 end do
351 max_ang = max_ang * 180.0_rp / (4.0_rp * atan(1.0_rp))
352 if (max_ang > 8.0_rp) then
353 write(log_buf, '(A, F6.2, A)') "MOST WM: Significant gravity-normal misalignment (max ", &
354 max_ang, " deg). Stability corrections will use projected gravity."
355 call neko_warning(trim(log_buf))
356 end if
357
359 if (any(this%h%x(1:this%n_nodes) .le. this%z0)) then
360 call neko_error("MOST WM: Sampling height h must be greater than roughness z0.")
361 else if ( (this%z0h_in .gt. 0.0_rp) .and. (any(this%h%x(1:this%n_nodes) .le. this%z0h_in)) ) then
362 call neko_error("MOST WM: Sampling height h must be greater than thermal roughness z0h.")
363 else if (this%z0 .eq. 0.0_rp) then
364 call neko_error("MOST WM: Roughness z0 must be greater than 0.")
365 else if (this%z0h_in .eq. 0.0_rp) then
366 call neko_error("MOST WM: Thermal roughness z0h must be greater than 0.")
367 end if
368
369 end subroutine most_init_from_components
