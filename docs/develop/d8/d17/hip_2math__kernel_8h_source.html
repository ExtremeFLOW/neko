89__device__ __forceinline__
90void face_gather_nonlinear_index(int *index, const int idx, const int lx,
91 const int ly, const int lz) {
92 const int idx2 = idx - 1;
93 index[3] = idx2 / (lx * ly * lz);
94 index[2] = (idx2 - (lx * ly * lz) * index[3]) / (lx * ly);
95 index[1] = (idx2 - (lx * ly * lz) * index[3] - (lx * ly) * index[2]) / lx;
96 index[0] = (idx2 - (lx * ly * lz) * index[3] - (lx * ly) * index[2]) -
97 lx * index[1];
98 index[0]++;
99 index[1]++;
100 index[2]++;
101 index[3]++;
102}

104__device__ __forceinline__
105int face_gather_idx(const int i, const int j, const int k, const int l,
106 const int n1, const int n2, const int nf) {
107 return ((i) + (n1) * (((j) - 1) + (n2) * (((k) - 1) + (nf) * (((l) - 1))))) - 1;
108}

113template< typename T >
114__global__ void face_masked_gather_copy_kernel(T * __restrict__ a,
115 const T * __restrict__ b,
116 const int * __restrict__ mask,
117 const int * __restrict__ facet,
118 const int n1,
119 const int n2,
120 const int lx,
121 const int ly,
122 const int lz,

125
126 const int idx = blockIdx.x * blockDim.x + threadIdx.x;
127 const int str = blockDim.x * gridDim.x;

129 for (int m = idx; m < n_mask; m += str) {
130 const int f = facet[m + 1];
131 face_gather_nonlinear_index(index, mask[m + 1], lx, ly, lz);
132
133 switch (f) {
134 case 1:
135 case 2:
136 a[m] = b[face_gather_idx(index[1], index[2], f, index[3], n1, n2, 6)];
137 break;
138 case 3:
139 case 4:
140 a[m] = b[face_gather_idx(index[0], index[2], f, index[3], n1, n2, 6)];
141 break;
142 case 5:
143 case 6:
144 a[m] = b[face_gather_idx(index[0], index[1], f, index[3], n1, n2, 6)];
145 break;
146 }
147 }

764template< typename T>

806template< typename T >
807__global__ void reduce_kernel(T * bufred, const int n) {
808
809 T sum = 0;
810 const int idx = blockIdx.x * blockDim.x + threadIdx.x;
811 const int str = blockDim.x * gridDim.x;
812 for (int i = idx; i<n ; i += str)
813 {
814 sum += bufred[i];
815 }
816
817 __shared__ T shared[64];
818 unsigned int lane = threadIdx.x % warpSize;
819 unsigned int wid = threadIdx.x / warpSize;
820
821 sum = reduce_warp<T>(sum);
822 if (lane == 0)
823 shared[wid] = sum;
824 __syncthreads();
825
826 sum = (threadIdx.x < blockDim.x / warpSize) ? shared[lane] : 0;
827 if (wid == 0)
828 sum = reduce_warp<T>(sum);
829
830 if (threadIdx.x == 0)
831 bufred[blockIdx.x] = sum;
