55 module subroutine precon_factory(pc, type_name)
56 class(pc_t), allocatable, intent(inout) :: pc
57 character(len=*), intent(in) :: type_name
58 character(len=:), allocatable :: type_string
59
60 if (allocated(pc)) then
61 call precon_destroy(pc)
62 deallocate(pc)
63 end if
64
65 select case (trim(type_name))
66 case ('jacobi')
67 if (neko_bcknd_sx .eq. 1) then
68 allocate(sx_jacobi_t::pc)
69 else if (neko_bcknd_device .eq. 1) then
70 allocate(device_jacobi_t::pc)
71 else
72 allocate(jacobi_t::pc)
73 end if
74 case ('hsmg')
75 allocate(hsmg_t::pc)
76 case ('phmg')
77 allocate(phmg_t::pc)
78 case('ident')
79 if (neko_bcknd_device .eq. 1) then
80 allocate(device_ident_t::pc)
81 else
82 allocate(ident_t::pc)
83 end if
84 case default
85 call neko_type_error("preconditioner", type_name, pc_known_types)
86 end select
87
88 end subroutine precon_factory

91 module subroutine precon_destroy(pc)
92 class(pc_t), allocatable, intent(inout) :: pc
93
94 if (allocated(pc)) then
95 select type (pcp => pc)
96 type is (jacobi_t)
97 call pcp%free()
98 type is (sx_jacobi_t)
99 call pcp%free()
100 type is (device_jacobi_t)
101 call pcp%free()
102 type is (hsmg_t)
103 call pcp%free()
104 type is (phmg_t)
105 call pcp%free()
106 end select
107 end if
108
109 end subroutine precon_destroy
