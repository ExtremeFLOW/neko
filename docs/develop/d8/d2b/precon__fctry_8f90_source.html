55 module subroutine precon_factory(pc, type_name)
56 class(pc_t), allocatable, intent(inout) :: pc
57 character(len=*), intent(in) :: type_name
58 character(len=:), allocatable :: type_string
59
60 if (allocated(pc)) then
61 call precon_destroy(pc)
62 deallocate(pc)
63 end if
64
65 if (trim(type_name) .eq. 'jacobi') then
66 if (neko_bcknd_sx .eq. 1) then
67 allocate(sx_jacobi_t::pc)
68 else if (neko_bcknd_device .eq. 1) then
69 allocate(device_jacobi_t::pc)
70 else
71 allocate(jacobi_t::pc)
72 end if
73 else if (type_name(1:4) .eq. 'hsmg') then
74 allocate(hsmg_t::pc)
75 else if (type_name(1:4) .eq. 'phmg') then
76 if (neko_bcknd_device .eq. 1) then
77 call neko_error('Hybrid ph multigrid only supported for CPU')
78 end if
79 allocate(phmg_t::pc)
80 else if(trim(type_name) .eq. 'ident') then
81 if (neko_bcknd_device .eq. 1) then
82 allocate(device_ident_t::pc)
83 else
84 allocate(ident_t::pc)
85 end if
86 else
87 type_string = concat_string_array(pc_known_types, &
88 new_line('A') // "- ", .true.)
89 call neko_error("Unknown preconditioner type: " &
90 // trim(type_name) // ". Known types are: " &
91 // type_string)
92 end if
93
94 end subroutine precon_factory
95
97 module subroutine precon_destroy(pc)
98 class(pc_t), allocatable, intent(inout) :: pc
99
100 if (allocated(pc)) then
101 select type (pcp => pc)
102 type is (jacobi_t)
103 call pcp%free()
104 type is (sx_jacobi_t)
105 call pcp%free()
106 type is (device_jacobi_t)
107 call pcp%free()
108 type is (hsmg_t)
109 call pcp%free()
110 type is (phmg_t)
111 call pcp%free()
112 end select
113 end if
114
115 end subroutine precon_destroy
116
117end submodule precon_fctry
