79 subroutine data_streamer_simcomp_init_from_json(this, json, case)
80 class(data_streamer_simcomp_t), intent(inout), target :: this
81 type(json_file), intent(inout) :: json
82 class(case_t), intent(inout), target :: case
83 character(len=NEKO_VARNAME_LEN), allocatable :: which_fields(:)
84 character(len=:), allocatable :: name
85 real(kind=rp) :: start_time
86 logical :: stream_mesh
87
88 call json_get_or_default(json, "name", name, "data_streamer_simcomp")
89 call json_get(json, 'fields', which_fields)
90
91 call json_get_or_lookup_or_default(json, 'start_time', start_time, &
92 -1.0_rp)
93
94 call json_get_or_default(json, 'stream_mesh', stream_mesh, .false.)
95
96 call this%init_base(json, case)
97 call this%init_from_components(name, which_fields, start_time, stream_mesh)
98
99 end subroutine data_streamer_simcomp_init_from_json

105 subroutine data_streamer_simcomp_init_from_components(this, name, &
106 which_fields, start_time, stream_mesh)
107 class(data_streamer_simcomp_t), intent(inout) :: this
108 character(len=*), intent(in) :: name
109 character(len=*), intent(in) :: which_fields(:)
110 real(kind=rp), intent(in) :: start_time
111 logical, intent(in) :: stream_mesh
112 type(field_t), pointer :: f
113
114 this%name = name
115 this%field_names = which_fields
116 this%start_time = start_time
117
118 call this%dstream%init(this%case%fluid%c_Xh)
119
120 if (stream_mesh) then
121 ! Stream the mesh coordinates of the first field in which_fields. We
122 ! assume that all fields share the same mesh. We don't use the dofmap
123 ! in case%fluid in case the fields we are streaming are masked.
124 f => neko_registry%get_field_by_name(which_fields(1))
125 call neko_log%message("Using field " // trim(f%name) // &
126 " for streaming mesh", lvl = neko_log_debug)
127 call neko_log%message("Streaming mesh: x-coordinates", &
128 lvl = neko_log_debug)
129 call this%dstream%stream(f%dof%x)
130 call neko_log%message("Streaming mesh: y-coordinates", &
131 lvl = neko_log_debug)
132 call this%dstream%stream(f%dof%y)
133 call neko_log%message("Streaming mesh: z-coordinates", &
134 lvl = neko_log_debug)
135 call this%dstream%stream(f%dof%z)
136 end if
137
138 end subroutine data_streamer_simcomp_init_from_components

153 subroutine data_streamer_simcomp_compute(this, time)
154 class(data_streamer_simcomp_t), intent(inout) :: this
155 type(time_state_t), intent(in) :: time
156
157 integer :: i
158 type(field_t), pointer :: f
159
160 if (time%t .ge. this%start_time) then
161 do i = 1, size(this%field_names)
162
163 ! Sync from GPU to CPU
164 f => neko_registry%get_field_by_name(this%field_names(i))
165 call f%copy_from(device_to_host, .true.)
166
167 call neko_log%message("Streaming field: " // this%field_names(i))
168
169 ! Stream the field
170 call this%dstream%stream(f%x)
171 end do
172 end if
173
174 end subroutine data_streamer_simcomp_compute
