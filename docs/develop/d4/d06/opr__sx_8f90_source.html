101 subroutine opr_sx_curl(w1, w2, w3, u1, u2, u3, work1, work2, c_Xh)

118 if (gdim .eq. 3) then

121 else

123 end if
124 ! this%work1=du/dz ; this%work2=dw/dx
125 if (gdim .eq. 3) then

132 call sub3(w2%x, work1%x, work2%x, n)
