170 subroutine field_registry_add_field(this, dof, fld_name, ignore_existing)
171 class(field_registry_t), intent(inout) :: this
172 type(dofmap_t), target, intent(in) :: dof
173 character(len=*), target, intent(in) :: fld_name
174 logical, optional, intent(in) :: ignore_existing
175 logical :: ignore_existing_
176
177 ignore_existing_ = .false.
178 if (present(ignore_existing)) then
179 ignore_existing_ = ignore_existing
180 end if
181
182 if (this%field_exists(fld_name)) then
183 if (ignore_existing_) then
184 return
185 else
186 call neko_error("Field with name " // fld_name // &
187 " is already registered")
188 end if
189 end if
190
191 if (this%n_fields() == size(this%fields)) then
192 call this%expand()
193 end if
194
195 this%n_fields_ = this%n_fields_ + 1
196
197 ! initialize the field at the appropraite index
198 call this%fields(this%n_fields_)%init( dof, fld_name)
199
200 end subroutine field_registry_add_field

205 subroutine field_registry_add_alias(this, alias, fld_name)
206 class(field_registry_t), intent(inout) :: this
207 character(len=*), target, intent(in) :: alias
208 character(len=*), target, intent(in) :: fld_name
209
210 if (this%field_exists(alias)) then
211 call neko_error("Cannot create alias. Field " // alias // &
212 " already exists in the registry")
213 end if
214
215 if (this%field_exists(fld_name)) then
216 if (this%n_aliases_ == size(this%aliases)) then
217 call this%expand_aliases()
218 end if
219

221
222 call this%aliases(this%n_aliases_)%initialize()
223 call this%aliases(this%n_aliases_)%add("alias", alias)
224 call this%aliases(this%n_aliases_)%add("target", fld_name)
225 else

227 " could not be found in the registry")

279 recursive function field_registry_get_field_by_name(this, name) result(f)
280 class(field_registry_t), target, intent(in) :: this
281 character(len=*), intent(in) :: name
282 character(len=:), allocatable :: alias
283 character(len=:), allocatable :: alias_target
284 type(field_t), pointer :: f
285 logical :: found
286 integer :: i
287 type(json_file), pointer :: alias_json ! need this for some reason
288
289 found = .false.
290
291 do i = 1, this%n_fields()
292 if (this%fields(i)%name == trim(name)) then
293 f => this%fields(i)
294 found = .true.
295 exit
296 end if
297 end do
298
299 do i = 1, this%n_aliases()
300 alias_json => this%aliases(i)
301 call json_get(alias_json, "alias", alias)
302 if (alias == trim(name)) then
303 call json_get(alias_json, "target", alias_target)
304 f => this%get_field_by_name(alias_target)
305 found = .true.
306 exit
307 end if
308 end do
309
310 if (.not. found) then
311 if (pe_rank .eq. 0) then
312 write(error_unit,*) "Current field_registry contents:"
313
314 do i=1, this%n_fields()
315 write(error_unit,*) "- ", this%fields(i)%name
316 end do
317 end if
318 call neko_error("Field " // name // &
319 " could not be found in the registry")
320 end if
321 end function field_registry_get_field_by_name
