323  subroutine fill_lvl_map(amg)
324  type(tamg_hierarchy_t), intent(inout) :: amg
325  integer :: i, j, k, l, nid, n
326  do j = 1, amg%lvl(1)%nnodes

328  nid = amg%lvl(1)%nodes(j)%dofs(k)

337  do k = 1, amg%lvl(l)%nodes(j)%ndofs
338  if (nid .eq. amg%lvl(l)%nodes(j)%dofs(k)) then
339  amg%lvl(l)%map_f2c_dof(i) = amg%lvl(l)%nodes(j)%gid

341  end do
342  end do
343  end do
