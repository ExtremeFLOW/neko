105  class(global_interpolation_t), intent(inout) :: this

129  2*lx, 2*ly, 2*lz, & ! Mesh size for bounding box computation
130  0.01, & ! relative size to expand bounding boxes by
131  lx*ly*lz*nelv, lx*ly*lz*nelv, & ! local/global hash mesh sizes

174  subroutine global_interpolation_find_common(this)
175  class(global_interpolation_t), intent(inout) :: this
176  !!Perhaps this should be kind dp
177  real(kind=rp) :: xdiff, ydiff, zdiff
178  character(len=8000) :: log_buf
179  real(kind=rp), allocatable :: x_check(:)
180  real(kind=rp), allocatable :: y_check(:)
181  real(kind=rp), allocatable :: z_check(:)
182  logical :: isdiff
183  integer :: i
184 
185 #ifdef HAVE_GSLIB
186 
187  ! gslib find points, which element they belong, to process etc.
188  call fgslib_findpts(this%gs_handle, &
189  this%error_code, 1, &
190  this%proc_owner, 1, &
191  this%el_owner, 1, &
192  this%rst, this%mesh%gdim, &
193  this%dist2, 1, &
194  this%xyz(1,1), this%mesh%gdim, &
195  this%xyz(2,1), this%mesh%gdim, &
196  this%xyz(3,1), this%mesh%gdim, this%n_points)
197 
198  do i=1,this%n_points
199 
200  !
201  ! Check validity of points
202  !
203  if (this%error_code(i) .eq. 1) then
204  if (this%dist2(i) .gt. this%tol) then
205  write(*,*) 'Point with coords: ',&
206  this%xyz(1,i),&
207  this%xyz(2,i),&
208  this%xyz(3,i),&
209  'Did not converge to tol. Absolute differences squared: ',&
210  this%dist2(i)
211  end if
212  end if
213 
214  if (this%error_code(i) .eq. 2) &
215  write(*,*) 'Point with coords: ',&
216  this%xyz(1,i), this%xyz(2,i), this%xyz(3,i),&
217  'Outside the mesh!',&
218  ' Interpolation on these points will return 0.0. dist2: ', &
219  this%dist2(i),&
220  'el_owner, rst coords: ',&
221  this%el_owner(i), this%rst(1,i), this%rst(2,i), this%rst(3,i)
222 
223  end do
224 
225  allocate(x_check(this%n_points))
226  allocate(y_check(this%n_points))
227  allocate(z_check(this%n_points))
228 
229  call fgslib_findpts_eval(this%gs_handle, x_check, &
230  1, this%error_code, 1, &
231  this%proc_owner, 1, this%el_owner, 1, &
232  this%rst, this%mesh%gdim, &
233  this%n_points, this%dof%x)
234 
235  call fgslib_findpts_eval(this%gs_handle, y_check, &
236  1, this%error_code, 1, &
237  this%proc_owner, 1, this%el_owner, 1, &
238  this%rst, this%mesh%gdim, &
239  this%n_points, this%dof%y)
240 
241  call fgslib_findpts_eval(this%gs_handle, z_check, &
242  1, this%error_code, 1, &
243  this%proc_owner, 1, this%el_owner, 1, &
244  this%rst, this%mesh%gdim, &
245  this%n_points, this%dof%z)
246 
247 
248  do i=1,this%n_points
249 
250  !
251  ! Check validity of points
252  !
253  isdiff = .false.
254  xdiff = (x_check(i)-this%xyz(1,i))**2
255  if ( xdiff .gt. this%tol) isdiff = .true.
256  ydiff = (y_check(i)-this%xyz(2,i))**2
257  if ( ydiff .gt. this%tol) isdiff = .true.
258  zdiff = (z_check(i)-this%xyz(3,i))**2
259  if ( zdiff .gt. this%tol) isdiff = .true.
260 
261  if (isdiff) then
262  write(*,*) 'Points with coords: ',&
263  this%xyz(1,i),this%xyz(2,i),this%xyz(3,i), &
264  'Differ from interpolated coords: ',&
265  x_check(i), y_check(i), z_check(i),&
266  'Distance squared: ',&
267  xdiff, ydiff, zdiff
268  end if
269  end do
270 
271  deallocate(x_check)
272  deallocate(y_check)
273  deallocate(z_check)
274 
275 
276 #else
277  call neko_error('Neko needs to be built with GSLIB support')
278 #endif
279  end subroutine global_interpolation_find_common
